006    /**
007     * Allows strings to be localized from a myriad of different places.
008     * Four string files are searched to determine the localized version of the string:
009     * <ol>
010     *   <li>a file with the same name as this class
011     *   <li>a file of the same name as the bundle that this class is in (as determined by the
012     *       NSExecutable key in the Info plist for the class's bundle)
013     *   <li>a file named "LocalizedStrings"
014     *   <li>a file named by the DIV GVCLocalizedFileNameVariable
015     * </ol>
016     * The search is done in this class' bundle, unless a DIV exists with the
017     * GVCLocalizedBundleVariable key, in which case, it uses the bundle indicated by that DIV.
018     *
019     * @author Copyright (c) 2001-2005  Global Village Consulting, Inc.  All rights reserved.
020     * This software is published under the terms of the Educational Community License (ECL) version 1.0,
021     * a copy of which has been included with this distribution in the LICENSE.TXT file.
022     * @version $Revision: 6$
023     */
024    public class Localization
025    {
026        public static final String GVCLocalizedFileNameVariable = "localizedFileName";
027        public static final String GVCLocalizedBundleVariable = "localizedBundle";
028        public static final String GVCDefaultLocalizedFileName = "Localizable";
029    
030    
031        /**
032         * Static methods only.  You'll never need to instantiate this class.
033         */
034        private Localization()
035        {
036            super();
037        }
038    
039    
040    
041        /**
042         * Helper method to support <code>localizedStringForString</code> and
043         * <code>localizedStringExistsForString</code>.<p>
044         *
045         * Four string files are searched to determine the localized version of the string (all are postpended with the ".strings" extension):
046         * <ol>
047         *   <li>a file with the same name as <code>anObject</code>'s class
048         *   <li>a file of the same name as the bundle that <code>anObject</code>'s class is in (as determined by the NSExecutable key in the Info plist for the class's bundle)
049         *   <li>a file named "Localizable"
050         *   <li>a file named by the DIV GVCLocalizedFileNameVariable
051         * </ol>
052         * The search is done in <code>anObject</code>'s bundle, unless a DIV exists with the <code>GVCLocalizedBundleVariable</code> key, in which case, it uses the bundle indicated by that DIV.
053         *
054         * @param anObject the object that controls where to look for the localized string
055         * @param stringToLocalize the string that this will localize
056         * @return the localized string or <code>null</code> if it could not be found
057         */
058        protected static String findLocalizedStringForString(Object anObject, String stringToLocalize)
059        {
060            /** require
061            [valid_anObject_param] anObject != null;
062            [valid_stringToLocalize_param] stringToLocalize != null; **/
063    
064            String localizedString = null;
065            NSBundle localBundle;
066    
067            if (DynamicInstanceVariable.instanceVariableExists(anObject, GVCLocalizedBundleVariable))
068            {
069                localBundle = (NSBundle)DynamicInstanceVariable.instanceVariable(anObject, GVCLocalizedBundleVariable);
070            }
071            else
072            {
073                localBundle = NSBundle.bundleForClass(anObject.getClass());
074            }
075    
076            // First look in a file with the same name as the class of anObject
077            if (NSBundleAdditions.localizedStringExistsForKey(localBundle, stringToLocalize, ClassAdditions.unqualifiedClassName(anObject.getClass())))
078            {
079                localizedString = NSBundleAdditions.localizedStringForKey(localBundle, stringToLocalize, ClassAdditions.unqualifiedClassName(anObject.getClass()));
080            }
081    
082            // Then look in a file with the same name as the bundle
083            if (localizedString == null)
084            {
085                String classAppName = (String)localBundle.name();
086                if (NSBundleAdditions.localizedStringExistsForKey(localBundle, stringToLocalize, classAppName))
087                {
088                    localizedString = NSBundleAdditions.localizedStringForKey(localBundle, stringToLocalize, classAppName);
089                }
090            }
091    
092            // Then look in the default file name
093            if (localizedString == null)
094            {
095                if (NSBundleAdditions.localizedStringExistsForKey(localBundle, stringToLocalize, GVCDefaultLocalizedFileName))
096                {
097                    localizedString = NSBundleAdditions.localizedStringForKey(localBundle, stringToLocalize, GVCDefaultLocalizedFileName);
098                }
099            }
100    
101            // Finally look in the file noted by the DIV GVCLocalizedFileName
102            if (localizedString == null)
103            {
104                if (DynamicInstanceVariable.instanceVariableExists(anObject, GVCLocalizedFileNameVariable))
105                {
106                    String divFileName = (String)DynamicInstanceVariable.instanceVariable(anObject, GVCLocalizedFileNameVariable);
107        
108                    if (NSBundleAdditions.localizedStringExistsForKey(localBundle, stringToLocalize, divFileName))
109                    {
110                        localizedString = NSBundleAdditions.localizedStringForKey(localBundle, stringToLocalize, divFileName);
111                    }
112                }
113            }
114    
115            return localizedString;
116        }
117    
118    
119    
120        /**
121         * Determines if a localized version of the passed string can be located.
122         *
123         * @param anObject the object that controls where to look for the localized string.
124         * @param stringToLocalize the string for which to search.
125         * @return <code>true</code> if the string <code>stringToLocalize</code> exists in any of the string files that we search, <code>false</code> otherwise.
126         */
127        public static boolean localizedStringExistsForString(Object anObject, String stringToLocalize)
128        {
129            /** require
130            [valid_anObject_param] anObject != null;
131            [valid_stringToLocalize_param] stringToLocalize != null; **/
132    
133            return findLocalizedStringForString(anObject, stringToLocalize) != null;
134        }
135    
136    
137    
138        /**
139         * Returns a localized version of the passed string.
140         *
141         * @param anObject the object that controls where to look for the localized string
142         * @param stringToLocalize the string that this will localize
143         * @return the localized version of <code>stringToLocalize</code>
144         */
145        public static String localizedStringForString(Object anObject, String stringToLocalize)
146        {
147            /** require
148            [valid_anObject_param] anObject != null;
149            [valid_stringToLocalize_param] stringToLocalize != null;
150            localizedStringExistsForString(anObject, stringToLocalize); **/
151    
152            return findLocalizedStringForString(anObject, stringToLocalize);
153    
154            /** ensure [valid_result] Result != null; **/
155        }
