001    package net.global_village.foundation;
002    
003    import org.apache.log4j.*;
004    
005    
006    /**
007     * Based on code from Dr Heinz Max Kabutz's The Java Specialists' Newsletter [Issue 137].  Rather than cut and paste
008     * the standard code:
009     * <pre>
010     * private final static Logger logger = Logger.getLogger(Application.class.getName());
011     * </pre>
012     * and risk forgetting to update the classname, use this  cut and paste safe class:
013     * <pre>
014     * private final static Logger logger = LoggerFactory.makeLogger();
015     * </pre>
016     * 
017     * @see <a href="http://www.javaspecialists.co.za/archive/newsletter.do?issue=137">The Java Specialists' Newsletter [Issue 137]</a>
018     * @author Copyright (c) 2006-2006 Harte-Hanks Shoppers, Inc.
019     */
020    public class LoggerFactory
021    {
022    
023          public static Logger makeLogger()
024        {
025            Throwable t = new Throwable();
026            StackTraceElement directCaller = t.getStackTrace()[1];
027            return Logger.getLogger(directCaller.getClassName());
028        }
029    }
