011    /**
012     * VersionableArchivableKeyedLookup differs from VersionableArchivableObject in that a known key is associated with each
013     * object. This key, which must be unique where isActive is true, is used by code to retrieve a specific instance of
014     * this class while maintaining a name that has meaning to the user. These keys must be pre-determined and defined in
015     * the program code. For example:
016     * <p>
017     * 
018     * <pre>
019     *   
020     *   public static final String cancelledStatusKey = @&quot;cancelled&quot;;
021     *    
022     *   public static MyVersionableArchivableKeyedLookup cancelledStatusWithEditingContext( EOEditingContext editingContext )
023     *   {
024     *       return VersionableArchivableKeyedLookup.objectForKeyWithEditingContextAndEntityName( cancelledStatusKey, editingContext, &quot;MyVersionableArchivableKeyedLookup&quot; );
025     *   }
026     * </pre>
027     * 
028     * @author Copyright (c) 2001-2005 Global Village Consulting, Inc. All rights reserved. This software is published under
029     *         the terms of the Educational Community License (ECL) version 1.0, a copy of which has been included with this
030     *         distribution in the LICENSE.TXT file.
031     * @version $Revision: 5$
032     */  
033    public class VersionableArchivableKeyedLookup extends _VersionableArchivableKeyedLookup
034    {
035        /**
036         * Returns the object of entity where key == aKey and isActive == YES.  Throws an exception if there are no matching objects, or more than one matching object.  Subclasses may want to wrap this method to pass in their own entity name.
037         *
038         * @param editingContext the <code>EOEditingContext</code> to fetch from
039         * @param aKey key of specific object to look for.
040         * @param entityName the kind of objects you want to fetch with this method
041         * @return object associated with <code>aKey</code>
042         */
043        static public VersionableArchivableKeyedLookup objectForKey( EOEditingContext editingContext, String entityName, String aKey)
044        {
045            /** require
046            [aKey_not_null] aKey != null;
047            [editingContext_not_null] editingContext != null;
048            [entityName_not_null] entityName != null;   **/
049            
050            return (VersionableArchivableKeyedLookup) EOUtilities.objectWithQualifierFormat( editingContext, entityName, "key = %@ AND isActive = %@", new NSArray( new Object [] { aKey, GVCBoolean.trueBoolean() } ) );
051    
052            /** ensure [Result_not_null] Result != null; **/
053        }
054    }
