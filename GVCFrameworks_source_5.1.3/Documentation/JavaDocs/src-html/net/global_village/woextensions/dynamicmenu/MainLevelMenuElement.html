030        /**
031         * Designated Constructor, takes in all the must have attributes of the class
032         *
033         * @param aKey the key by which this menu can be accessed
034         * @param aLinkHTML the HTML to use for the link, can be text or an image reference, allows for CSS and 
035                  HTML tags to be used
036         * @param aDestination either the action to have the parent component perform or some URI relative or absolute
037         * @param aFrameName the name of the frame to appear in
038         * @param someAltText the alternate text, for image links and used in textual menus too
039         * @param aContentAlignment either leftAlignment, rightAlignment, or centerAlignment
040         * @param aKeyboardShortCut the IE specific navigational aid, a single character
041         * @param aSecondaryMenu a menu to display when this element is "moused over"
042         * @param isDestinationDynamic flag to differentiate static from dynamic destinations
043         */
044        public MainLevelMenuElement(String aKey,
045                                    String aDestination,
046                                    String aLinkHTML,
047                                    String aFrameName,
048                                    String someAltText,
049                                    String aContentAlignment,
050                                    String aKeyboardShortCut,
051                                    SecondaryMenu aSecondaryMenu,
052                                    boolean isDestinationDynamic)
053        {
054            super(aKey,
055                  aDestination,
056                  aLinkHTML,
057                  aFrameName,
058                  someAltText,
059                  isDestinationDynamic);
060            /** require
061            [valid_aDestination_param] aDestination != null;
062            [valid_aLinkHTML_param] aLinkHTML != null;
063            [valid_aFrameName_param] aFrameName != null;
064            [valid_someAltText_param] someAltText != null;
065            [valid_aContentAlignment_param] aContentAlignment != null;
066            [content_alignment_is_valid] (aContentAlignment.equals(DynamicMenu.centerAlignment)) || (aContentAlignment.equals(DynamicMenu.rightAlignment)) || (aContentAlignment.equals(DynamicMenu.leftAlignment));
067            [valid_aKeyboardShortCut_param] aKeyboardShortCut != null;
068            [shortcut_is_valid] aKeyboardShortCut.length() == 1 || aKeyboardShortCut.equals(MainLevelMenuElement.defaultKeyboardShortCut); /# this last precondition caused problem for instance when unit testing.  Odd exceptions occur when it is uncommented. Added OR clause. #/ **/
069    
070            // Don't use the setters so that the invariant works...
071            contentAlignment = aContentAlignment;
072            keyboardShortCut = aKeyboardShortCut;
073    
074            setSubMenu(aSecondaryMenu);
075    
076            setLeftPosition(DynamicMenu.defaultOffset);
077            setTopPosition(DynamicMenu.defaultOffset);
078            setKeyboardShortCut(aKeyboardShortCut);
079            setWidth(MainLevelMenuElement.defaultWidth);
080        }

284        /**
285         * Returns properly formatted String version of this object as calls to BrotherCake's Ultimate
286           Dropdown Menu javascript<br>
287           example: addMainItem("http://www.domain.com/","Home",100,"left","_top","Home page",0,0,"h");
288         *
289         * @param aContext used to generate correct URI's
290         * @return String version of this object as calls to BrotherCake's javascript
291         */
292        public String javascript(WOContext aContext)
293        {
294            /** require [valid_param] aContext != null; **/
295    
296            String javascript;
297    
298            javascript = "addMainItem(";
299    
300            if (isDestinationDynamic())
301            {
302                javascript = javascript + StringUtilities.quotedString(actionNameAsURI(aContext));
303            }
304            else
305            {
306                javascript = javascript + StringUtilities.quotedString(uri());
307            }
308            
309            javascript = javascript 
310                          + ","
311                          + StringUtilities.quotedString(linkHTML())
312                          + ","
313                          + width()
314                          + ","
315                          + StringUtilities.quotedString(contentAlignment())
316                          + ","
317                          + StringUtilities.quotedString(frameName())
318                          + ","
319                          + StringUtilities.quotedString(altText())
320                          + ","
321                          + topPosition()
322                          + ","
323                          + leftPosition()
324                          + ","
325                          + StringUtilities.quotedString(keyboardShortCut())
326                          + ");";
327    
328            if (hasSubMenu())
329            {
330                javascript = javascript + DynamicMenu.newLineString + subMenu().javascript(aContext) + DynamicMenu.newLineString;
331            }
332            
333            return javascript;
334    
335            /** ensure [valid_result] Result != null; **/
336        }

356        /**
357         * Conformance to NSKeyValueCoding. This method first checks itself for a matching key.
358         * If one does not exist, it looks in the submenu for a menu element that has a
359         * key that matches.
360         *
361         * @param aKey the key to look up
362         * @return the value of the key
363         */
364        public Object valueForKey(String aKey)
365        {
366            try
367            {
368                return NSKeyValueCoding.DefaultImplementation.valueForKey(this, aKey);
369            }
370            catch (NSKeyValueCoding.UnknownKeyException e)
371            {
372                return subMenu().valueForKey(aKey);
373            }
374        }
375    
376    
377        /**
378         * Conformance to NSKeyValueCoding. This method first checks itself for a matching key.
379         * If one does not exist, it looks in the elements array for a menu element that has a
380         * key that matches.
381         *
382         * @param value the value to set
383         * @param aKey the key to look up
384         */
385        public void takeValueForKey(Object value, String aKey)
386        {
387            NSKeyValueCoding.DefaultImplementation.takeValueForKey(this, value, aKey);
388        }
