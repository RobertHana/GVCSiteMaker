001    package net.global_village.foundation;
002    
003    /**
004     ExceptionConverter changes a checked exception into an unchecked (Runtime) exception.  It preserves the stack trace of the original exception so that the location of the error can be pinpointed.
005     <br>
006     <b>Example Usage</b>
007     <br>
008     <code>
009     try {
010       f();
011     } catch(IOException ex) {
012       System.out.println("Printing out plain ol' IOException:");
013       System.out.println("---");
014       ex.printStackTrace();
015       System.out.println("---");
016       throw new ExceptionConverter(ex);
017     }
018     </code>
019     <br>
020     <br>
021     * @author Copyright (c) 2001-2005  Global Village Consulting, Inc.  All rights reserved.
022     * This software is published under the terms of the Educational Community License (ECL) version 1.0,
023     * a copy of which has been included with this distribution in the LICENSE.TXT file.
024     <br>
025      This material from The Java(tm) Specialists' Newsletter by Maximum Solutions (South Africa) - The Java(tm) Specialists.<br>
026      Please contact <a href="mailto:h.kabutz@computer.org>Dr. Heinz M. Kabutz (h.kabutz@computer.org)</a> for more information.
027     <br>
028      Back issues are available on-line at http://www.smotricz.com/kabutz.  This code is taken from <a href="http://www.javaspecialists.eu/archive/Issue033.html">Issue 33</a>
029    */
030    public class ExceptionConverter extends RuntimeException
031    {
032    
033        /** we keep a handle to the wrapped exception */
034        private final Throwable ex;
035    
036        public ExceptionConverter(Throwable exception)
037        {
038            this.ex = exception;
039        }
040    
041    
042    
043        /** and allow the user of ExceptionConverter to get a handle to it. */
044        public Throwable getException()
045        {
046            return ex;
047        }
048    
049    
050    
051        /** We print the message of the checked exception */
052        public String getMessage()
053        {
054            return ex.getMessage();
055        }
056    
057    
058    
059        /** and make sure we also produce a localized version */
060        public String getLocalizedMessage()
061        {
062            return ex.getLocalizedMessage();
063        }
064    
065    
066    
067        /** The toString() is changed to be prefixed with ExceptionConverter */
068        public String toString()
069        {
070            return "ExceptionConverter: " + ex;
071        }
072    
073    
074    
075        /** we have to override this as well */
076        public void printStackTrace()
077        {
078            printStackTrace(System.err);
079        }
080    
081    
082    
083        /** here we prefix, with s.print(), not s.println(), the stack trace with "ExceptionConverter:" */
084        public void printStackTrace(java.io.PrintStream s)
085        {
086            synchronized (s)
087            {
088                s.print("ExceptionConverter: ");
089                ex.printStackTrace(s);
090            }
091        }
092    
093    
094    
095        /** Again, we prefix the stack trace with "ExceptionConverter:" */
096        public void printStackTrace(java.io.PrintWriter s)
097        {
098            synchronized (s)
099            {
100                s.print("ExceptionConverter: ");
101                ex.printStackTrace(s);
102            }
103        }
104    
105    
106    
107        /** requests to fill in the stack trace we will have to ignore (I think)  We can't throw an exception here, as this method is called by the constructor of Throwable */
108        public Throwable fillInStackTrace()
109        {
110            return this;
111        }
112    
113    }
