016    /**
017     * <p>An Ant Task facilitating the use of <a href="http://www.rubicode.com/Software/EOGenerator/">
018     * EOGenerator</a> to generate the Java source for EOEnterpriseObjects.  See the
019     * EOGenerator documentation for details on the code generation process.</p>
020     * 
021     * <p>Here is an example usage of the task:
022     * <pre>
023     *  &lt;target name="test" depends="setProps"&gt;
024     *       &lt;eogenerator
025     *           isDebugging="on"
026     *           model="TestModel1.eomodeld"
027     *           baseClassDestination="GeneratedEOs"&gt;
028     *           subClassDestination="testSrc"
029     *           &lt;referencedModels&gt;
030     *               &lt;model location="../EOPrototypes/EOPrototypes.eomodeld" /&gt;
031     *           &lt;/referencedModels&gt;
032     *       &lt;/eoGenerator&gt;
033     *   &lt;/target&gt;
034     * </pre>
035     * </p>
036     * 
037     * <p>These properties are recognized by the eogenerator task:<br/>
038     * <table border="1">
039     *   <tr align="left">
040     *     <th>Name</th><th>Type</th><th>Use</th>
041     *   </tr>
042     *   <tr>
043     *     <td>executable</td><td>File</td><td>The EOGenerator executable.</td>
044     *   </tr>
045     *   <tr>
046     *     <td>model</td><td>Path</td><td>The model to generate classes for.</td>
047     *   </tr>
048     *   <tr>
049     *     <td>baseClassDestination</td><td>Path</td><td>The location to generate base-classes at, not including package names.</td>
050     *   </tr>
051     *   <tr>
052     *     <td>subClassDestination</td><td>Path</td><td>The location to generate sub-classes at, not including package names.</td>
053     *   </tr>
054     *   <tr>
055     *     <td>templateDirectory</td><td>Path</td><td>The directory containing the templates for EOGenerator.</td>
056     *   </tr>
057     *   <tr>
058     *     <td>baseClassTemplate</td><td>String</td><td>Name of the template file for base classes.</td>
059     *   </tr>
060     *   <tr>
061     *     <td>subClassTemplate</td><td>String</td><td>Name of the template file for sub-classes.</td>
062     *   </tr>
063     *   <tr>
064     *     <td>entities</td><td>String</td><td>Optional, space seperated, list of entities to generate classes.  If ommitted, all entities are generated</td>
065     *   </tr>
066     *   <tr>
067     *     <td>isDebugging</td><td>boolean</td><td>Dumps out details of EOGenerator command useful when things go wrong.</td>
068     *   </tr>
069     *   <tr>
070     *     <td>referencedModels</td><td>ReferencedModels(path structure)</td><td>Paths to models referenced by model that classes are being generated for.</td>
071     *   </tr>
072     *   <tr>
073     *     <td><i>&lt;anything else&gt;<i></td><td>String</td><td>Define passed to EOGenerator.</td>
074     *   </tr>
075     * </table>
076     * </p>
077     * 
078     * <p>Additionally, some special property names can be used in place of direct specification
079     * in the task.  This is useful for general settings in the the <code>wobuild.properties</code> file
080     * and overrides in the project's <code>build.properties</code> file.  These are:
081     * <table border="1">
082     *   <tr align="left">
083     *     <th>eogenerator Task Property</th><th>wobuild.properties / build.properties name</th>
084     *   </tr>
085     *   <tr>
086     *     <td>executable</td><td>eogenerator.executable</td>
087     *   </tr>
088     *   <tr>
089     *     <td>templateDirectory</td><td>eogenerator.template.dir</td>
090     *   </tr>
091     *   <tr>
092     *     <td>baseClassTemplate</td><td>eogenerator.baseClassTemplate</td>
093     *   </tr>
094     *   <tr>
095     *     <td>subClassTemplate</td><td>eogenerator.subClassTemplate</td>
096     *   </tr>
097     * </table>
098     * </p>
099     * <p>Three special property names are also associated with defines.  These are useful 
100     * for general settings in the the <code>wobuild.properties</code> file and overrides in the 
101     * project's <code>build.properties</code> file. These are:
102     * <table border="1">
103     *   <tr align="left">
104     *     <th>eogenerator Task Property</th><th>wobuild.properties / build.properties name</th>
105     *   </tr>
106     *   <tr>
107     *     <td>EOSuperClass</td><td>eogenerator.superclass</td>
108     *   </tr>
109     *   <tr>
110     *     <td>copyrightYear</td><td>eogenerator.copyrightYear</td>
111     *   </tr>
112     *   <tr>
113     *     <td>copyrightBy</td><td>eogenerator.eogenerator.copyrightBy</td>
114     *   </tr>
115     * </table>
116     * </p>

151         * Runs EOGenerator.
152         * 
153         * @see org.apache.tools.ant.Task#execute()

155        public void execute() throws BuildException
156        {
157            setValuesFromPropertyFiles();
158            logDebugInfo();
159            checkErrorConditions();

161        }

165        /**
166         * Uses the values from the project's properties for executable, 
167         * templateDirectory, baseClassTemplate, subClassTemplate, and the EOSuperClass, 
168         * copyrightYear, and copyrightBy defines if they were not set in the xml file.
169         */
170        public void setValuesFromPropertyFiles()
171        {
172            if ((executable() == null) && isPropertyDefined(EXECUTABLE_PROPERTY))
173            {
174                setExecutable(new File(propertyNamed(EXECUTABLE_PROPERTY)));
175            }
176            
177            if ((templateDirectory() == null) && isPropertyDefined(TEMPLATE_DIRECTORY_PROPERTY))
178            { 
179                setTemplateDirectory(new Path(getProject(), propertyNamed(TEMPLATE_DIRECTORY_PROPERTY)));
180            }
181            
182            if ((baseClassTemplate() == null) && isPropertyDefined(BASE_CLASS_TEMPLATE_PROPERTY))
183            { 
184                setBaseClassTemplate(propertyNamed(BASE_CLASS_TEMPLATE_PROPERTY));
185            }
186    
187            
188            if ((subClassTemplate() == null) && isPropertyDefined(SUB_CLASS_TEMPLATE_PROPERTY))
189            { 
190                setSubClassTemplate(propertyNamed(SUB_CLASS_TEMPLATE_PROPERTY));
191            }
192            
193            if ( ( ! isDefined(COPYRIGHT_YEAR_DEFINE)) && (isPropertyDefined(COPYRIGHT_YEAR_PROPERTY)) )
194            {
195                setDynamicAttribute(COPYRIGHT_YEAR_DEFINE, propertyNamed(COPYRIGHT_YEAR_PROPERTY));
196            }
197            
198            if ( ( ! isDefined(COPYRIGHT_BY_DEFINE)) && (isPropertyDefined(COPYRIGHT_BY_PROPERTY)) )
199            {
200                setDynamicAttribute(COPYRIGHT_BY_DEFINE, propertyNamed(COPYRIGHT_BY_PROPERTY));
201            }
202    
203            if ( ( ! isDefined(EOSUPER_CLASS_DEFINE)) && (isPropertyDefined(EOSUPER_CLASS_PROPERTY)) )
204            {
205                setDynamicAttribute(EOSUPER_CLASS_DEFINE, propertyNamed(EOSUPER_CLASS_PROPERTY));
206            }
207        }

211        /**
212         * Checks for error conditions.  At present, only that the EOGenerator executable
213         * can be found. 
214         */
215        public void checkErrorConditions()
216        {
217            if ((executable() == null) || (! executable().exists()))
218            {
219                throw new BuildException("eoGenerator executable not set or not found.");
220            }
221        }

225        /**
226         * Logs out useful debug information. 
227         */
228        public void logDebugInfo()
229        {
230            if (isDebugging())
231            {
232                log("Starting eoGenerator task", Project.MSG_INFO);
233                log("Executable is " + executable(), Project.MSG_INFO);
234                log("Model is " + model(), Project.MSG_INFO);
235                log("Generating entities: " + (entities() == null ? "All" : entities()), Project.MSG_INFO);
236                
237                log("Referenced Models: ", Project.MSG_INFO);
238                String[] paths = referencedModels.list();
239                for (int i = 0; i < paths.length; i++) {
240                    log("  " + paths[i], Project.MSG_INFO);
241                }
242                log("eoSuperClass is " + defines().get(EOSUPER_CLASS_DEFINE), Project.MSG_INFO);
243                log("Base-classes generated to " + baseClassDestination(), Project.MSG_INFO);
244                log("Sub-classes generated to " + subClassDestination(), Project.MSG_INFO);
245                log("Using templates from " + templateDirectory(), Project.MSG_INFO);
246                
247                log("Base-class template " + baseClassTemplate(), Project.MSG_INFO);
248                log("Sub-class template " + subClassTemplate(), Project.MSG_INFO);
249                
250                log("Defined Values: ", Project.MSG_INFO);
251                for (Enumeration definedKeys = defines().keys() ; definedKeys.hasMoreElements() ;) 
252                {
253                    String aKey = (String) definedKeys.nextElement();
254                    log("  " + aKey + "=" + definitionFor(aKey), Project.MSG_INFO);
255                }
256            }
257        }

261        /**
262         * Creates a command line and launches EOGenerator. 
263         */
264        public void launchEOGenerator()
265        {
266            Commandline cmd = new Commandline();
267            cmd.setExecutable(executable().toString());
268            
269            cmd.createArgument().setValue("-model");
270            cmd.createArgument().setPath(model());
271            
272            String[] paths = referencedModels.list();
273            for (int i = 0; i < paths.length; i++) {
274                cmd.createArgument().setValue("-refmodel");
275                cmd.createArgument().setValue(paths[i]);
276            }
277                     
278            cmd.createArgument().setValue("-destination");
279            cmd.createArgument().setPath(baseClassDestination());
280    
281            cmd.createArgument().setValue("-subclassDestination");
282            cmd.createArgument().setPath(subClassDestination());
283            
284            cmd.createArgument().setValue("-templatedir");
285            cmd.createArgument().setPath(templateDirectory());
286            
287            cmd.createArgument().setValue("-javaTemplate");
288            cmd.createArgument().setValue(baseClassTemplate());
289            
290            cmd.createArgument().setValue("-subclassJavaTemplate");
291            cmd.createArgument().setValue(subClassTemplate());
292    
293            cmd.createArgument().setValue("-java");
294            cmd.createArgument().setValue("-packagedirs");
295            
296            if (isDebugging())
297            {
298                cmd.createArgument().setValue("-verbose");
299            }
300    
301            for (Enumeration definedKeys = defines().keys() ; definedKeys.hasMoreElements() ;) 
302            {
303                String aKey = (String) definedKeys.nextElement();
304                cmd.createArgument().setValue("-define-" + aKey);
305                cmd.createArgument().setValue(definitionFor(aKey));
306            }
307    
308            PumpStreamHandler streamHandler = new PumpStreamHandler();
309            Execute runner = new Execute(streamHandler, null);
310            runner.setAntRun(getProject());
311            runner.setCommandline(cmd.getCommandline());
312            
313            int returnValue = 0;
314            try
315            {
316                returnValue = runner.execute();
317            }
318            catch (IOException e)
319            {
320                log(e.getMessage(), Project.MSG_INFO);
321            }
322        }

351         * Returns the EOGenerator defintion for <code>key</code> if it is defined.
352         * 
353         *  @return the EOGenerator defintion for <code>key</code> if it is defined, 
354         * null otherwise.
355         */
356        public String definitionFor(String key)
357        {
358            return (String) defines().get(key);
359        }

361        
362    
363        /**
364         * Returns <code>true</code> if there is a EOGenerator defintion for <code>key</code>.
365         * 
366         *  @return <code>true</code> if there is a EOGenerator defintion for <code>key</code>, 
367         * <code>false</code> otherwise.
368         */
369        public boolean isDefined(String key)
370        {
371            return definitionFor(key) != null;
372        }
