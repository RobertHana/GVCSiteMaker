514        /**
515         * Examines the groups the <code>User</code> is a member of to determine which privileges they have.  Caches the privileges to speed subsequent queries.
516         */
517        public synchronized void determineCurrentPrivileges()
518        {
519            allPrivilegesAndIncludedPrivileges = new NSMutableSet();
520    
521            //Get all the privileges of this user's groups
522            Enumeration eachGroup = groups().objectEnumerator();
523            while(eachGroup.hasMoreElements())
524            {
525                UserGroup thisGroup = (UserGroup) eachGroup.nextElement();
526                allPrivilegesAndIncludedPrivileges.addObjectsFromArray(thisGroup.privileges());
527    
528                //Get all the privileges if this users' groups's privileges
529                Enumeration eachPrivilege = thisGroup.privileges().objectEnumerator();
530                while(eachPrivilege.hasMoreElements())
531                {
532                    Privilege thisPrivilege = (Privilege) eachPrivilege.nextElement();
533                    allPrivilegesAndIncludedPrivileges.addObjectsFromArray(thisPrivilege.includedPrivileges());
534                }
535            }
536    
537            allPrivilegesAndIncludedPrivilegesNames = (NSArray)allPrivilegesAndIncludedPrivileges.allObjects().valueForKey("name");
538            /** ensure
539            [privilege_cache_created] allPrivilegesAndIncludedPrivileges != null;
540            [privilege_name_cache_created] allPrivilegesAndIncludedPrivilegesNames != null; **/
541        }
542    
543    
544    
545        /**
546         * Protected method.  Supports lazy creation of privilege cache.
547         */
548        protected synchronized void determinePrivilegesIfNeeded()
549        {
550            if (allPrivilegesAndIncludedPrivileges == null)
551            {
552                determineCurrentPrivileges();
553            }
554            /** ensure
555            [privilege_cache_created] allPrivilegesAndIncludedPrivileges != null;
556            [privilege_name_cache_created] allPrivilegesAndIncludedPrivilegesNames != null; **/
557        }

574        /**
575         * Validate the user.
576         */
577        public void validateForSave() throws com.webobjects.foundation.NSValidation.ValidationException
578        {
579            NSMutableArray exceptions = new NSMutableArray();
580    
581            try
582            {
583                super.validateForSave();
584            }
585            catch ( com.webobjects.foundation.NSValidation.ValidationException e)
586            {
587                exceptions.addObjectsFromArray(ValidationExceptionAdditions.unaggregateExceptions(e));
588            }
589    
590            // Check for userName uniqueness, this validation should be done here instead in validateForKey so that it will only be called everytime a user tries to save and not during takeValuesFromRequest (in the case where userName is directly bound to a WOComponent)
591            if (userName() != null)
592            {
593                NSDictionary snapshot = editingContext().committedSnapshotForObject(this);
594                Object committedUserName = snapshot.objectForKey("userName");
595    
596                // Do not validate if it did not change
597                if ( ! userName().equals(committedUserName))
598                {
599                    if (userExists(userName(), editingContext()))
600                    {
601                        // Get the corresponding user
602                        User namedUser = User.user(userName(), editingContext());
603    
604                        // If we find a user and they are not the same, we have a problem
605                        if (namedUser != this)
606                        {
607                            exceptions.addObject(new com.webobjects.foundation.NSValidation.ValidationException("The user name must be unique."));
608                        }
609                    }
610                }
611            }
612    
613            // Conflict in changing password options
614            if (mustChangePassword().isTrue() && canChangePassword().isFalse())
615            {
616                exceptions.addObject(new com.webobjects.foundation.NSValidation.ValidationException("Must change password and can change password options are conflicting."));
617            }
618    
619    
620            if (exceptions.count() > 0)
621            {
622                throw EOFValidationException.aggregateExceptionWithExceptions(exceptions);
623            }
624        }
