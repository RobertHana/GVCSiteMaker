152        /**
153         * Tests computeAggregateFunction
154         */
155        public void testComputeAggregateFunction()
156        {
157            NSArray allEntitiesWithContraints = EOUtilities.objectsForEntityNamed(editingContext(), "EntityWithConstraints");
158    
159            BigDecimal nine = new BigDecimal(9.00);
160            NSArray argsWithNine = new NSArray (nine);
161            NSArray argsWithTwoNine = new NSArray (new Object[] {nine, nine});
162    
163            EOQualifier qual = EOQualifier.qualifierWithQualifierFormat("quantity < %f", argsWithNine);
164    
165            EOQualifier noMatchQualifier = EOQualifier.qualifierWithQualifierFormat("(quantity < %f) AND (quantity > %f)", argsWithTwoNine);
166            NSArray someEntitiesWithContraints = EOUtilities.objectsWithQualifierFormat(editingContext(), "EntityWithConstraints", "quantity < %f", argsWithNine);
167    
168            Object sumOfAllObjects = EOEditingContextAdditions.computeAggregateFunction(editingContext(), EOEditingContextAdditions.Sum, "quantity", "EntityWithConstraints", null);
169    
170            Object sumOfSomeObjects = EOEditingContextAdditions.computeAggregateFunction(editingContext(), EOEditingContextAdditions.Sum, "quantity", "EntityWithConstraints", qual);
171    
172            Object minimumOfAllObjects = EOEditingContextAdditions.computeAggregateFunction(editingContext(), EOEditingContextAdditions.Min, "quantity", "EntityWithConstraints", null);
173    
174            Object minimumOfSomeObjects = EOEditingContextAdditions.computeAggregateFunction(editingContext(), EOEditingContextAdditions.Min, "quantity", "EntityWithConstraints", qual);
175    
176            Object maximumOfAllObjects = EOEditingContextAdditions.computeAggregateFunction(editingContext(), EOEditingContextAdditions.Max, "quantity", "EntityWithConstraints", null);
177    
178            Object maximumOfSomeObjects = EOEditingContextAdditions.computeAggregateFunction(editingContext(), EOEditingContextAdditions.Max, "quantity", "EntityWithConstraints", qual);
179    
180    
181            // #warning Test breaks.  Oracle Adaptor bug?  This *used* to work, no code changes...
182            Object averageOfAllObjects  = EOEditingContextAdditions.computeAggregateFunction(editingContext(), EOEditingContextAdditions.Avg, "quantity", "EntityWithConstraints", null);
183    
184            Object averageOfSomeObjects = EOEditingContextAdditions.computeAggregateFunction(editingContext(), EOEditingContextAdditions.Avg, "quantity", "EntityWithConstraints", qual);
185    
186            // Previously this test checked for a BigDecimal(0) result.  That is just wrong.  No matches should
187            // result in null not zero.  The docs for EOEditingContextAdditions.computeAggregateFunction correctly
188            // state this.
189            Object realResult = EOEditingContextAdditions.computeAggregateFunction(editingContext(), EOEditingContextAdditions.Avg, "quantity", "EntityWithConstraints", noMatchQualifier);
190            assertEquals("no matching records returns null not zero", null, realResult);
191    
192            assertEquals("sumOfAllObjects.getClass() equals BigDecimal.class", sumOfAllObjects.getClass(), BigDecimal.class);
193            assertEquals("sumOfSomeObjects.getClass() equals BigDecimal.class", sumOfSomeObjects.getClass(), BigDecimal.class);
194            assertEquals("minimumOfAllObjects.getClass() equals BigDecimal.class", minimumOfAllObjects.getClass(), BigDecimal.class);
195            assertEquals("minimumOfSomeObjects.getClass() equals BigDecimal.class", minimumOfSomeObjects.getClass(), BigDecimal.class);
196            assertEquals("maximumOfAllObjects.getClass() equals BigDecimal.class", maximumOfAllObjects.getClass(), BigDecimal.class);
197            assertEquals("maximumOfSomeObjects.getClass() equals BigDecimal.class", maximumOfSomeObjects.getClass(), BigDecimal.class);
198            assertEquals("averageOfAllObjects.getClass() equals BigDecimal.class", averageOfAllObjects.getClass(), BigDecimal.class);
199            assertEquals("averageOfSomeObjects.getClass() equals BigDecimal.class", averageOfSomeObjects.getClass(), BigDecimal.class);
200    
201            assertEquals("sumOfAllObjects equals sum.quantity", sumOfAllObjects, allEntitiesWithContraints.valueForKeyPath("@sum.quantity"));
202            assertEquals("sumOfSomeObjects equals sum.quantity", sumOfSomeObjects, someEntitiesWithContraints.valueForKeyPath("@sum.quantity"));
203    
204            // equals() returns false if the scales are different...
205            assertTrue("minimumOfAllObjects equals min.quantity", ((BigDecimal)minimumOfAllObjects).compareTo((BigDecimal)allEntitiesWithContraints.valueForKey("@min.quantity")) == 0);
206            assertTrue("someEntitiesWithContraints equals min.quantity", ((BigDecimal)minimumOfSomeObjects).compareTo((BigDecimal)someEntitiesWithContraints.valueForKeyPath("@min.quantity")) == 0);
207            assertTrue("allEntitiesWithContraints equals min.quantity", ((BigDecimal)maximumOfAllObjects).compareTo((BigDecimal)allEntitiesWithContraints.valueForKeyPath("@max.quantity")) == 0);
208            assertTrue("maximumOfSomeObjects equals min.quantity",((BigDecimal)maximumOfSomeObjects).compareTo((BigDecimal)someEntitiesWithContraints.valueForKeyPath("@max.quantity")) == 0);
209    
210            assertTrue("averageOfAllObjects equals avg.quantity", BigDecimalAdditions.equalsWithScale((BigDecimal)averageOfAllObjects, (BigDecimal)allEntitiesWithContraints.valueForKeyPath("@avg.quantity")));
211            assertTrue("averageOfSomeObjects equals avg.quantity", BigDecimalAdditions.equalsWithScale((BigDecimal)averageOfSomeObjects, (BigDecimal)someEntitiesWithContraints.valueForKeyPath("@avg.quantity")));
212    
213            try
214            {
215                EOEditingContextAdditions.computeAggregateFunction(editingContext(),
216                                                                   "bad",
217                                                                   "quantity",
218                                                                   "EntityWithConstraints",
219                                                                   qual);
220                fail("Expected exception did not occur");
221            }
222            catch (Exception e) {}
223        }

388         * test objectForGlobalIDs
389         */
390        public void testObjectsForGlobalIDs()
391        {
392            NSArray emptyArray = new NSArray();
393            NSArray arrayWithEntityOne = new NSArray(entityOne);
394            NSArray arrayWithEntityOnesID = new NSArray(EOObject.globalID(entityOne));
395            NSArray arrayWithFourEntityIDs = new NSArray(new Object[] {EOObject.globalID(entityOne),
396                EOObject.globalID(entityTwo),
397                EOObject.globalID(entityThree),
398                EOObject.globalID(entityFour)});
399            EOEditingContext newEditingContext = new EOEditingContext();
400    
401            assertEquals(EOEditingContextAdditions.objectsForGlobalIDs(editingContext(), emptyArray), emptyArray);
402            assertEquals(EOEditingContextAdditions.objectsForGlobalIDs(editingContext(), arrayWithEntityOnesID), arrayWithEntityOne);
403    
404            try
405            {
406                EOEditingContextAdditions.objectsForGlobalIDs(newEditingContext, arrayWithEntityOnesID);
407                fail("Expected exception did not occur");
408            }
409            catch (RuntimeException e) { }
410    
411            EOUtilities.localInstanceOfObject(newEditingContext, entityOne);
412            try
413            {
414                EOEditingContextAdditions.objectsForGlobalIDs(newEditingContext, arrayWithEntityOnesID);
415            }
416            catch (RuntimeException e) { fail("Unexpected exception occured"); }
417    
418            NSSet setWithFourEntities = new NSSet (new Object[] {entityOne, entityTwo, entityThree, entityFour});
419            NSMutableSet resultSet = new NSMutableSet();
420            resultSet.addObjectsFromArray(EOEditingContextAdditions.objectsForGlobalIDs(editingContext(), arrayWithFourEntityIDs));
421            assertTrue(resultSet.isEqualToSet(setWithFourEntities));
422        }

426        /**
427         * Test valueForAggregateSQL
428         */
429        public void testValueForAggregateSQL()
430        {
431            String sql = "SELECT sum(quantity) FROM \"Entity_With_Constraints\"";
432            BigDecimal sumOfQuantity = EOEditingContextAdditions.valueForAggregateSQL(editingContext(), "EntityWithConstraints", sql);
433            assertTrue("sumOfQuantity has correct sum", sumOfQuantity.setScale(4, BigDecimal.ROUND_HALF_UP).equals(new BigDecimal("-317.0618")));
434    
435            //Test with filter
436            String sqlWithFilter = "SELECT avg(quantity) FROM \"Entity_With_Constraints\" where quantity > 0";
437            BigDecimal average = EOEditingContextAdditions.valueForAggregateSQL(editingContext(), "EntityWithConstraints", sqlWithFilter);
438            assertTrue("valueForAggregateSQL has correct average", average.setScale(4, BigDecimal.ROUND_HALF_UP).equals(new BigDecimal("3.9770")));
439    
440            //Test without result
441            String sqlNoResults = "SELECT sum(quantity) FROM \"Entity_With_Constraints\" where quantity > 0 and quantity <= 0";
442            BigDecimal sum = EOEditingContextAdditions.valueForAggregateSQL(editingContext(), "EntityWithConstraints", sqlNoResults);
443            assertTrue("valueForAggregateSQL has correct sum if no results", sum.equals(new BigDecimal("0")));
444        }

448        /**
449         * Test primaryKey
450         */
451        public void testPrimaryKey()
452        {
453            Integer primaryKeyOne = (Integer) EOEditingContextAdditions.primaryKey(editingContext(), entityOne, "theID");
454            SimpleTestEntity equivalentObject = (SimpleTestEntity) EOUtilities.objectWithPrimaryKey(editingContext(),
455                                                                                                    "SimpleTestEntity",
456                                                                                                    new NSDictionary(primaryKeyOne, "theID"));
457            assertTrue("correct primary key for entityOne", entityOne.equals(equivalentObject));
458    
459            //Test returns null for uncommited objects
460            SimpleTestEntity newObjectInsertedToEC = new SimpleTestEntity();
461            editingContext().insertObject(newObjectInsertedToEC);
462            Integer primaryKeyforNewObjectInsertedToEC = (Integer) EOEditingContextAdditions.primaryKey(editingContext(), newObjectInsertedToEC, "theID");
463            editingContext().deleteObject(newObjectInsertedToEC);
464            assertTrue("correct primary key for newObject", primaryKeyforNewObjectInsertedToEC == null);
465        }
