001    package net.global_village.foundation.tests;
002    
003    import java.util.Enumeration;
004    
005    import junit.framework.TestCase;
006    import net.global_village.foundation.FixedWidthStringTokenizer;
007    
008    import com.webobjects.foundation.NSArray;
009    import com.webobjects.foundation.NSDictionary;
010    
011    
012    /**
013     * Test the fixed width string tokenizer functionality.
014     *
015     * @author Copyright (c) 2001-2005  Global Village Consulting, Inc.  All rights reserved.
016     * This software is published under the terms of the Educational Community License (ECL) version 1.0,
017     * a copy of which has been included with this distribution in the LICENSE.TXT file.
018     */
019    public class FixedWidthStringTokenizerTest extends TestCase
020    {
021        protected String testString;
022        protected NSArray widths, names;
023        protected FixedWidthStringTokenizer tokenizer;
024    
025    
026        public FixedWidthStringTokenizerTest(String name)
027        {
028            super(name);
029        }
030    
031    
032    
033        /**
034         * Sets up the fixtures.
035         * @exception Exception an exception that the setup may throw
036         */
037        public void setUp() throws java.lang.Exception
038        {
039            super.setUp();
040    
041            testString = "1234567890";
042            widths = new NSArray(new Integer[] {new Integer(1), new Integer(4), new Integer(5)});
043            names = new NSArray(new String[] {"first", "second", "third"});
044    
045            tokenizer = new FixedWidthStringTokenizer(testString, widths, names);
046        }
047    
048    
049    
050        /**
051         * Test basic functioanlity.
052         */
053        public void testBasic()
054        {
055            assertEquals("1", tokenizer.nextToken());
056            assertEquals("2345", tokenizer.nextToken());
057            assertEquals("67890", tokenizer.nextToken());
058        }
059    
060    
061    
062        /**
063         * Test some contracts.
064         */
065        public void testContracts()
066        {
067            NSArray longWidths = new NSArray(new Integer[] {new Integer(1), new Integer(4), new Integer(6)});
068            NSArray shortWidths = new NSArray(new Integer[] {new Integer(1), new Integer(4), new Integer(4)});
069    
070            try {
071                new FixedWidthStringTokenizer(testString, longWidths);
072                fail("Widths are too long");
073            } catch (RuntimeException e) {}
074    
075            try {
076                new FixedWidthStringTokenizer(testString, shortWidths);
077                fail("Widths are too short");
078            } catch (RuntimeException e) {}
079        }
080    
081    
082    
083        /**
084         * Test allRemainingTokens().
085         */
086        public void testAllRemainingTokens()
087        {
088            NSArray allTokens = tokenizer.allRemainingTokens();
089            Enumeration tokenEnum = allTokens.objectEnumerator();
090    
091            assertEquals(tokenEnum.nextElement(), "1");
092            assertEquals(tokenEnum.nextElement(), "2345");
093            assertEquals(tokenEnum.nextElement(), "67890");
094        }
095    
096    
097    
098        /**
099         * Test allRemainingTokensWithNames().
100         */
101        public void testAllRemainingTokensWithNames()
102        {
103            NSDictionary allTokens = tokenizer.allRemainingTokensWithNames();
104            Enumeration tokenEnum = allTokens.objectEnumerator();
105            Enumeration nameEnum = allTokens.keyEnumerator();
106    
107            assertEquals(tokenEnum.nextElement(), "1");
108            assertEquals(tokenEnum.nextElement(), "2345");
109            assertEquals(tokenEnum.nextElement(), "67890");
110    
111            assertEquals(nameEnum.nextElement(), "first");
112            assertEquals(nameEnum.nextElement(), "second");
113            assertEquals(nameEnum.nextElement(), "third");
114        }
115    
116    
117    
118    }
