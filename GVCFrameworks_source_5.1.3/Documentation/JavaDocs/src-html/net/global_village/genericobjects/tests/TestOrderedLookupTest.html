020    public class TestOrderedLookupTest extends GVCJUnitEOTestCase
021    {
022        protected TestOrderedLookup instanceOne;
023        protected TestOrderedLookup instanceTwo;
024        protected TestOrderedLookup instanceThree;
025    
026        protected Lookup defaultTestObject;
027        protected String defaultTestObjectName;
028    
029    
030        /**
031         * Designated constructor.
032         *
033         * @param name The name of the test to be initialized
034         */
035        public TestOrderedLookupTest(String name)
036        {
037            super(name);
038        }
039        
040    
041    
042        public void setUp() throws java.lang.Exception
043        {
044            super.setUp();
045    
046            instanceOne   = new TestOrderedLookup();
047            instanceTwo   = new TestOrderedLookup();
048            instanceThree = new TestOrderedLookup();
049    
050            instanceOne.setName("One");
051            instanceOne.setOrdering(new Integer(1));
052            instanceTwo.setName("Two");
053            instanceTwo.setOrdering(new Integer(2));
054            instanceThree.setName("Three");
055            instanceThree.setOrdering(new Integer(3));
056    
057            editingContext.insertObject( instanceOne );
058            editingContext.insertObject( instanceTwo );
059            editingContext.insertObject( instanceThree );
060    
061            defaultTestObjectName = DefaultValueRetrieval.defaultString(instanceOne.getClass(), "TestOrderedLookup.defaultName");
062            defaultTestObject = new TestOrderedLookup();
063            ((TestOrderedLookup) defaultTestObject).setOrdering(new Integer(4));
064            defaultTestObject.setName(defaultTestObjectName);
065            editingContext.insertObject(defaultTestObject);
066    
067            saveChangesShouldntThrow("after creating Objects");
068        }
069    
070    
071    
072        public void tearDown() throws java.lang.Exception
073        {
074            editingContext.deleteObject(instanceOne);
075            editingContext.deleteObject(instanceTwo);
076            editingContext.deleteObject(instanceThree);
077    
078            editingContext.deleteObject(defaultTestObject);
079    
080            saveChangesShouldntThrow("destroying Objects");
081    
082            super.tearDown();
083        }
084    
085    
086    
087    
088        /**
089         * Array of Ordered objects for the Entity being tested
090         */
091        protected NSArray orderedObjects()
092        {
093            return TestOrderedLookup.orderedObjects(editingContext());
094        }
095    
096    
097    
098        public void testObjectOrder()
099        {
100            NSArray resultArray = orderedObjects();
101    
102            for (int i = resultArray.count() - 1; i > 0; i-- )
103            {
104                OrderedLookup firstObject = (OrderedLookup) resultArray.objectAtIndex( i - 1 );
105                OrderedLookup secondObject = (OrderedLookup) resultArray.objectAtIndex( i );
106                Number firstOrdering =  firstObject.ordering();
107                Number secondOrdering =  secondObject.ordering();
108    
109                assertTrue("firstObject compared to secondObject is before", firstOrdering.intValue() < secondOrdering.intValue());
110            }
111        }
112    
113        
114    
115        /**
116         * Test OrderingComparator
117         */
118        public void testOrderingComparator()
119        {
120            NSArray orderedTestObjects = OrderedLookup.orderedObjects(editingContext, "TestOrderedLookup", OrderedLookup.OrderingComparator);
121    
122            TestOrderedLookup firstObject = (TestOrderedLookup) EOUtilities.objectMatchingKeyAndValue(editingContext,
123                                                                                            "TestOrderedLookup",
124                                                                                            "name",
125                                                                                            "One");
126            TestOrderedLookup middleObject = (TestOrderedLookup) EOUtilities.objectMatchingKeyAndValue(editingContext,
127                                                                                             "TestOrderedLookup",
128                                                                                             "name",
129                                                                                             "Two");
130            TestOrderedLookup lastObject = (TestOrderedLookup) EOUtilities.objectMatchingKeyAndValue(editingContext,
131                                                                                           "TestOrderedLookup",
132                                                                                           "name",
133                                                                                           "Three");
134    
135            assertTrue(orderedTestObjects.indexOfObject(firstObject) < orderedTestObjects.indexOfObject(middleObject));
136            assertTrue(orderedTestObjects.indexOfObject(middleObject) < orderedTestObjects.indexOfObject(lastObject));
137            assertTrue(orderedTestObjects.indexOfObject(lastObject) < orderedTestObjects.indexOfObject(defaultTestObject));
138        }
