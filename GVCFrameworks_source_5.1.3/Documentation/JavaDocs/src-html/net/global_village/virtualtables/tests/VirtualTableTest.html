226        /**
227         * Test importing and exporting data.
228         */
229        public void testImportExportTable() throws IOException, ParseException
230        {
231            VirtualTable lookupTable = null;
232            try
233            {
234                // A lookup table
235                lookupTable = VirtualTable.createVirtualTable(editingContext(), "Lookup Table");
236                Column columnToLookup = lookupTable.newColumn("lookupString", ColumnType.StringColumnType);
237                VirtualRow row = lookupTable.newRow();
238                row.takeValueForKey("lookup value 1", "lookupString");
239                row = lookupTable.newRow();
240                row.takeValueForKey("lookup value 2", "lookupString");
241    
242                // Add a lookup column to the test table
243                VirtualLookupColumn lookupColumn = (VirtualLookupColumn)testTable.newColumn("lookup column", ColumnType.LookupColumnType);
244                lookupColumn.setLookupColumn(columnToLookup);
245    
246                // Have to save so that the new lookup column gets a permanent global ID
247                editingContext().saveChanges();
248    
249                NSBundle bundle = NSBundle.bundleForClass(VirtualTableTest.class);
250                String tablePath = bundle.resourcePathForLocalizedResourceNamed("TestImportData.txt", null);
251                byte[] tableBytes = bundle.bytesForResourcePath(tablePath);
252    
253                NSArray columnsToImport = new NSArray(new Column[] {testTable.columnNamed(stringColumnName),
254                                                                    testTable.columnNamed("numbercolumn"),
255                                                                    testTable.columnNamed("date column"),
256                                                                    testTable.columnNamed("lookup column"),
257                                                                    testTable.columnNamed("boolean column"),
258                                                                    testTable.columnNamed("expression")});
259                NSDictionary formatters = new NSDictionary(new NSTimestampFormatter("%Y-%m-%d"), "date column");
260    
261                InputStream stream = new ByteArrayInputStream(tableBytes);
262                testTable.importTable(stream, null, columnsToImport, formatters, "\t", true, true);
263                editingContext().saveChanges();
264                assertEquals(3, testTable.rows().count());
265    
266                stream = new ByteArrayInputStream(tableBytes);
267                testTable.importTable(stream, null, columnsToImport, formatters, "\t", true, true);
268                editingContext().saveChanges();
269                assertEquals(3, testTable.rows().count());
270    
271                stream = new ByteArrayInputStream(tableBytes);
272                testTable.importTable(stream, null, columnsToImport, formatters, "\t", true, false);
273                editingContext().saveChanges();
274                assertEquals(6, testTable.rows().count());
275    
276    
277                // Test for invalid type
278                /*columnsToImport = new NSArray(new Column[] {testTable.columnNamed("numbercolumn"), testTable.columnNamed(stringColumnName)});
279                stream = new ByteArrayInputStream(tableBytes);
280                try
281                {
282                    testTable.importTable(stream, null, columnsToImport, new NSDictionary(), "\t", true, true);
283                }
284                catch (Exception e)
285                {
286                    throw new ExceptionConverter(e);
287                } */
288    
289    
290                // Now export some of that data... first, munge some data so that it has cr/lf to test the munging facility
291                row = (VirtualRow)testTable.rows().objectAtIndex(0);
292                row.takeValueForKey("test\nof\rEOL\r\ncharacters", stringColumnName);
293                NSArray columnsToExport = new NSArray(new Column[] {testTable.columnNamed(stringColumnName),
294                                                                    testTable.columnNamed("numbercolumn"),
295                                                                    testTable.columnNamed("date column"),
296                                                                    testTable.columnNamed("lookup column"),
297                                                                    testTable.columnNamed("boolean column"),
298                                                                    testTable.columnNamed("expression")});
299                // Test without header
300                File exportTableData = File.createTempFile("ExportedTableData", ".txt");
301                FileOutputStream fos = new FileOutputStream(exportTableData);
302                testTable.exportTable(null, null, fos, null, columnsToExport, "\t", false);
303                fos.close();
304                exportTableData.deleteOnExit();
305    
306                FileInputStream fileInputStream = new FileInputStream(exportTableData);
307                NSData contentsOfFile = new NSData(fileInputStream, fileInputStream.available());
308                fileInputStream.close();
309                String exportTableString = new String(contentsOfFile.bytes());
310    
311                String cannonicalTestTablePath = bundle.resourcePathForLocalizedResourceNamed("ExportTestDataWithoutHeader.txt", null);
312                byte[] cannonicalTableBytes = bundle.bytesForResourcePath(cannonicalTestTablePath);
313                String cannonicalTestString = new String(cannonicalTableBytes);
314    
315                assertEquals(cannonicalTestString, exportTableString);
316    
317                // Test with header
318                ByteArrayOutputStream exportedTableDataWithHeader = new ByteArrayOutputStream();
319                testTable.exportTable(null, null, exportedTableDataWithHeader, null, columnsToExport, ",", true);
320                exportedTableDataWithHeader.close();
321    
322                cannonicalTestTablePath = bundle.resourcePathForLocalizedResourceNamed("ExportTestDataWithHeader.txt", null);
323                cannonicalTableBytes = bundle.bytesForResourcePath(cannonicalTestTablePath);
324                cannonicalTestString = new String(cannonicalTableBytes);
325    
326                assertEquals(cannonicalTestString, exportedTableDataWithHeader.toString());
327            }
328            finally
329            {
330                // Cleanup after this test
331                editingContext().deleteObject(lookupTable);
332            }
333        }

337        /**
338         * Test importing quoted data.
339         */
340        public void testImportQuotedTable() throws IOException, ParseException
341        {
342            VirtualTable importTable = null;
343            try
344            {
345                NSMutableArray columnsToImport = new NSMutableArray();
346                importTable = VirtualTable.createVirtualTable(editingContext(), "Import Table");
347                columnsToImport.addObject(importTable.newColumn("Case Name", ColumnType.StringColumnType));
348                columnsToImport.addObject(importTable.newColumn("Date of Decision", ColumnType.TimestampColumnType));
349                columnsToImport.addObject(importTable.newColumn("Court", ColumnType.StringColumnType));
350                columnsToImport.addObject(importTable.newColumn("Country", ColumnType.StringColumnType));
351                columnsToImport.addObject(importTable.newColumn("Reported Citation", ColumnType.StringColumnType));
352                columnsToImport.addObject(importTable.newColumn("Electronic Citation", ColumnType.StringColumnType));
353                columnsToImport.addObject(importTable.newColumn("Web Citation", ColumnType.StringColumnType));
354                columnsToImport.addObject(importTable.newColumn("Other Locator", ColumnType.StringColumnType));
355                columnsToImport.addObject(importTable.newColumn("Holdings re subjective element", ColumnType.StringColumnType));
356                columnsToImport.addObject(importTable.newColumn("Ratio or Obiter Dictum", ColumnType.StringColumnType));
357                columnsToImport.addObject(importTable.newColumn("Relevant Quotations", ColumnType.StringColumnType));
358                columnsToImport.addObject(importTable.newColumn("Comments or Observations", ColumnType.StringColumnType));
359                columnsToImport.addObject(importTable.newColumn("References for Follow Up", ColumnType.StringColumnType));
360                columnsToImport.addObject(importTable.newColumn("Ideas", ColumnType.StringColumnType));
361                columnsToImport.addObject(importTable.newColumn("Author or Authors", ColumnType.StringColumnType));
362                columnsToImport.addObject(importTable.newColumn("Published", ColumnType.BooleanColumnType));
363                assertEquals(18, importTable.columns().count());
364    
365                editingContext().saveChanges();
366    
367                NSBundle bundle = NSBundle.bundleForClass(VirtualTableTest.class);
368                String tablePath = bundle.resourcePathForLocalizedResourceNamed("TestImportQuotingData.txt", null);
369                byte[] tableBytes = bundle.bytesForResourcePath(tablePath);
370    
371                NSDictionary formatters = new NSDictionary(new NSTimestampFormatter("%Y-%m-%d %H:%M:%S"), "Date of Decision");
372    
373                InputStream stream = new ByteArrayInputStream(tableBytes);
374                importTable.importTable(stream, null, columnsToImport, formatters, ",", true, true);
375                editingContext().saveChanges();
376                assertEquals(2, importTable.rows().count());
377                assertEquals(18, ((VirtualRow)importTable.rows().objectAtIndex(0)).fields().count());
378                assertEquals(18, ((VirtualRow)importTable.rows().objectAtIndex(1)).fields().count());
379            }
380            finally
381            {
382                // Cleanup after this test
383                editingContext().deleteObject(importTable);
384            }
385        }
