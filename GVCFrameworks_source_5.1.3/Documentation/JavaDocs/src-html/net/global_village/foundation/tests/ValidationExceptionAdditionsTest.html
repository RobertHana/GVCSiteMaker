016    public class ValidationExceptionAdditionsTest extends TestCase
017    {
018    
019    
020        public ValidationExceptionAdditionsTest(String name)
021        {
022            super(name);
023        }
024    
025    
026    
027        /**
028         * Tests for unaggregateExceptions.
029         */
030        public void testUnaggregateExceptions()
031        {
032            com.webobjects.foundation.NSValidation.ValidationException parentException = new com.webobjects.foundation.NSValidation.ValidationException("Test parent");
033            com.webobjects.foundation.NSValidation.ValidationException childException = new com.webobjects.foundation.NSValidation.ValidationException("Test child");
034    
035            com.webobjects.foundation.NSValidation.ValidationException aggregateException = com.webobjects.foundation.NSValidation.ValidationException.aggregateExceptionWithExceptions(new NSArray(new Object[] {parentException, childException}));
036    
037            assertEquals("Number of exceptions not what was expected", ValidationExceptionAdditions.unaggregateExceptions(aggregateException).count(), 2);
038            assertEquals("Number of exceptions not what was expected", ValidationExceptionAdditions.unaggregateExceptions(childException).count(), 1);
039        }
040    
041        
042        
043        /**
044         * Tests for unaggregateExceptions.
045         */
046        public void testLocalizedMessagesForUnaggregatedExceptions()
047        {
048            com.webobjects.foundation.NSValidation.ValidationException parentException = new com.webobjects.foundation.NSValidation.ValidationException("Test parent");
049            com.webobjects.foundation.NSValidation.ValidationException childException = new com.webobjects.foundation.NSValidation.ValidationException("Test child");
050    
051            com.webobjects.foundation.NSValidation.ValidationException aggregateException = com.webobjects.foundation.NSValidation.ValidationException.aggregateExceptionWithExceptions(new NSArray(new Object[] {parentException, childException}));
052    
053            assertEquals("Number of exceptions not what was expected", ValidationExceptionAdditions.localizedMessagesForUnaggregatedExceptions(aggregateException).count(), 2);
054            assertEquals("Number of exceptions not what was expected", ValidationExceptionAdditions.localizedMessagesForUnaggregatedExceptions(aggregateException), new NSArray(new String[] {"Test parent", "Test child"}));
055        
056            assertEquals("Number of exceptions not what was expected", ValidationExceptionAdditions.localizedMessagesForUnaggregatedExceptions(childException).count(), 1);
057    
058        }
059
