017    /**
018     * This class is used to send mails in a threaded way.  This is useful as sending multiple messages
019     * can be time consuming.
020     *
021     * <p>Notifications are broadcast when messages are accepted by the SMTP server or when they fail to send.
022     * </p>
023     *
024     * <p>These properties should be set for better recovery and notification.  The default timeouts are <b>forever</b>.</p>
025     * <pre>
026     * mail.smtp.connectiontimeout=15000
027     * mail.smtp.timeout=15000
028     * </pre>
029     *
030     * This code is adapted from ERMailSender from the ERJavaMail framework of Project Wonder.  This file
031     * is protected under the GNU Lesser Public License.
032     *
033     * @see <a href="http://java.sun.com/products/javamail/javadocs/com/sun/mail/smtp/package-summary.html">SMTP Package Summary</a>
034     * @author Camille Troillard <tuscland@mac.com>
035     * @author Tatsuya Kawano <tatsuyak@mac.com>
036     * @author Max Muller <maxmuller@mac.com>
037     * @author Sacha Mallais <sacha@global-village.net> DBC and clean up
038     * @author Chuck Hill <chill@global-village.net> added failure recovery, notifications
039     */
040    public class ThreadedMailAgent extends Thread

319        /**
320         * Sends a message in a non-blocking way.<br>
321         * This means that the thread won't be blocked.  Instead the message will be queued for later delivery.
322         *
323         * <p>When the message is accepted by the transport without exception, the SentMessageNotification will be
324         * broadcast for message.</p>
325         *
326         * @param message the message to send
327         */
328        public void sendMessageDeferred(com.sun.mail.smtp.SMTPMessage message) throws ThreadedMailAgent.SizeOverflowException
329        {
330            /** require [valid_message] message != null;  **/
331            queueMessage(message);
332            setThreadSuspended(false);
333    
334            // If we have not started to send mails, start the thread
335            if ( ! isAlive())
336            {
337                start();
338                return;
339            }
340        }

344        /**
345         * Queues the message for later delivery.
346         *
347         * @param message the message to queue
348         * @throws ThreadedMailAgent.SizeOverflowException if the mail queue size is exceeded
349         */
350        protected void queueMessage(com.sun.mail.smtp.SMTPMessage message) throws ThreadedMailAgent.SizeOverflowException
351        {
352            /** require [valid_message] message != null;  **/
353            try
354            {
355                messages().push(new MessageWithRetries(message));
356                if (logger.isDebugEnabled())
357                {
358                    try
359                    {
360                       logger.debug("Added message titled " + message.getSubject() + ", to: " + prettyPrintAddresses(message.getAllRecipients()) + " to the mail queue.");
361                    }
362                    catch (MessagingException e)
363                    {
364                        logger.debug("Failed to log adding message to queue", e);
365                    }
366                }
367            }
368            catch(FIFOQueue.SizeOverflowException e)
369            {
370                logger.error("Mail queue is full, failed to add message to queue", e);
371                throw new ThreadedMailAgent.SizeOverflowException();
372            }
373        }

377        /**
378         * Sends a message immediately.<br> This means that the thread could be blocked if the message
379         * takes time to be delivered.  If the mail server can't be contacted the message is queued for
380         * future delivery attempts.
381         *
382         * <p>When the message is accepted by the transport without exception, the SentMessageNotification will be
383         * broadcast for message.</p>
384         *
385         * @param message the message to send
386         * @throws ThreadedMailAgent.SizeOverflowException if the message can't be sent immediately and the mail queue size is exceeded
387         */
388        public void sendMessageNow(com.sun.mail.smtp.SMTPMessage message) throws ThreadedMailAgent.SizeOverflowException
389        {
390            Transport transport = null;
391            try
392            {
393               transport = connectedTransport();
394               if (transport != null)
395               {
396                   _sendMessageNow(message, transport);
397               }
398               else
399               {
400                   // If we cannot connect then queue the message and try again later.
401                   sendMessageDeferred(message);
402               }
403            }
404            catch (MessagingException e)
405            {
406                logger.error("Caught exception when sending mail in a non-blocking manner.", e);
407                throw new NSForwardException(e, "Caught exception when sending mail in a non-blocking manner.");
408            }
409            finally
410            {
411                if (transport != null)
412                {
413                    try
414                    {
415                        transport.close();
416                    }
417                    catch (MessagingException e)
418                    {
419                        // Fatal exception ... we must at least notify the user
420                        logger.error("Caught exception when closing transport.", e);
421                        throw new NSForwardException(e, "Unable to open nor close the messaging transport channel.");
422                    }
423                }
424            }
425        }

429        /**
430         * All messages are sent from this method.  It is used by 'sendMessageNow' and 'sendMessageDeffered' and
431         * is the meat of the 'run' method when the thread is running.<br> If an exception occurs while
432         * sending the mail, it is logged and thrown.
433         *
434         * <p>If the message is accepted by the transport without exception, the SentMessageNotification will be
435         * broadcast for message.</p>
436         * <p>If the message is rejected by the transport or sending fails for any reason, the FailedToSendMessageNotification will be
437         * broadcast for message.</p>
438         */
439        protected void _sendMessageNow(com.sun.mail.smtp.SMTPMessage message, Transport transport) throws MessagingException
440        {
441            /** require [valid_message] message != null;
442                        [valid_transport] transport != null;   **/
443    
444            try
445            {
446                transport.sendMessage(message, message.getAllRecipients());
447                NSNotificationCenter.defaultCenter().postNotification(SentMessageNotification, message);
448    
449                if (logger.isDebugEnabled())
450                {
451                    logger.debug("Message titled " + message.getSubject() + " sent to " + prettyPrintAddresses(message.getAllRecipients()));
452                }
453            }
454            catch (SendFailedException e)
455            {
456                // BUG NOTE: this will not attempt to resend this message to e.getValidUnsentAddresses()
457                NSNotificationCenter.defaultCenter().postNotification(FailedToSendMessageNotification, message, new NSDictionary(e, ExceptionKey));
458                logger.error("Failed to send message titled " + message.getSubject() + " to " + prettyPrintAddresses(e.getInvalidAddresses()));
459                logger.error("  - successfully sent message to " + prettyPrintAddresses(e.getValidSentAddresses()));
460                logger.error("  - stopped before sending message to " + prettyPrintAddresses(e.getValidUnsentAddresses()), e);
461            }
462            catch (MessagingException e)
463            {
464                logger.error("Error while sending message titled " + message.getSubject() + " to " + prettyPrintAddresses(message.getAllRecipients()), e);
465                logger.error("Cause: ", e.getCause());
466                logger.error("Next: ", e.getNextException());
467                throw e;
468            }
469            catch (Throwable t)
470            {
471                NSNotificationCenter.defaultCenter().postNotification(FailedToSendMessageNotification, message, new NSDictionary(t, ExceptionKey));
472                logger.error("Unexpected error - failed to send message titled " + message.getSubject() + " to " + prettyPrintAddresses(message.getAllRecipients()), t);
473                throw new ExceptionConverter(t);
474            }
475        }

479        /**
480         * Utility method that gets the SMTP Transport method for a session and connects the Transport
481         * before returning it. Returns null if the transport can NOT be connected!  Can be retried later
482         * if null is returned.
483         *
484         * <p>If the transport for SMTP is not found, the SendingMessagesNotification will be  broadcast for this ThreadedMailAgent.</p>
485         * <p>If the transport can't be connected to, the FailedToContactMailServer will be  broadcast for this ThreadedMailAgent.</p>
486         *
487         * @return null or connected mail transport.
488         */
489        protected Transport connectedTransport()
490        {
491            Transport transport = null;
492            try
493            {
494                transport = mailSession().getTransport("smtp");
495            }
496            catch (NoSuchProviderException e)
497            {
498                NSNotificationCenter.defaultCenter().postNotification(FailedToGetTransportNotification, this);
499    
500                logger.error("Unable to get the mail transport from the session.", e);
501                throw new NSForwardException(e, "Unable to get the mail transport from the session.");
502            }
503    
504            if ( ! transport.isConnected())
505            {
506                try
507                {
508                    transport.connect();
509                }
510                catch (MessagingException e)
511                {
512                    NSNotificationCenter.defaultCenter().postNotification(FailedToContactMailServerNotification, this);
513                    logger.error("Unable to connect transport due to", e);
514                    transport = null;
515                }
516            }
517    
518            return transport;
519        }

523        /**
524         * Don't call this method, this is the thread run loop and is automatically called.
525         *
526         * <p>When transport is established and a cycle of sending queued messages is started, the SendingMessagesNotification will be
527         * broadcast for this ThreadedMailAgent.</p>
528         * <p>If a SocketException occurs while sending messages, the FailedToContactMailServer will be broadcast for this ThreadedMailAgent.</p>
529         */
530        public void run()
531        {
532            while( ! (messages().empty() && shouldEnd()))
533            {
534                if (shouldEnd())
535                {
536                    logger.info("ThreadedMailAgent signaled to end, but still has to process " + messages().size() + " messages in queue.");
537                }
538    
539                // The thread sleeps until sendMessageDeferred wakes it up.
540                try
541                {
542                    if(threadSuspended())
543                    {
544                        synchronized(this)
545                        {
546                            while (threadSuspended())
547                            {
548                                sleep(milliSecondsWaitRunLoop());
549                            }
550                        }
551                    }
552                }
553                catch(InterruptedException e)
554                {
555                    setThreadSuspended(true);  // Does this serve any useful purpose?
556                }
557    
558                // If there are still messages pending ...
559                if( ! messages().empty())
560                {
561                    // This is a try block of last resort.  We must not allow run() to throw.
562                    try
563                    {
564                        Transport transport = connectedTransport();
565    
566                        // If the transport is not available, sleep a bit and we will try again later.
567                        if (transport == null)
568                        {
569                            logger.error("ThreadedMailAgent can't connect to SMTP server, sleeping...");
570                            try
571                            {
572                                sleep(TRANSPORT_UNAVAILABLE_DELAY);
573                                continue;
574                            }
575                            catch(InterruptedException e)
576                            {
577                                setThreadSuspended(true); // Does this serve any useful purpose?
578                                continue;
579                            }
580                        }
581    
582                        NSNotificationCenter.defaultCenter().postNotification(SendingMessagesNotification, this);
583    
584                        while( ! messages().empty())
585                        {
586                            MessageWithRetries message = (MessageWithRetries)messages().pop();
587                            try
588                            {
589                                _sendMessageNow(message.message(), transport);
590                            }
591                            catch(MessagingException e)
592                            {
593                                // Here we get all the exceptions that are not 'SendFailedException's.
594                                // This does not seem to happen any more.  - CH 2008-4-3
595                                if (e.getNextException() instanceof java.net.SocketException)
596                                {
597                                    logger.error("Lost connection to mail server, non-fatal Socket Exception.");
598                                    // Hey!  Someone pulled the plug on us!  Bail out of the loop so a new connection will be made
599                                    // Re-queue the message, will try it again later.
600                                    messages().push(message);
601                                    // Discard the transport so that we don't attempt to close it.
602                                    transport = null;
603                                    break;
604                                }
605    
606                                // Only record the attempt if it was not a socket exception.  It might still be a transient (network) failure,
607                                // but we will assume it is not.
608                                message.recordAttempt();
609    
610                                if (message.hasExhaustedAttempts())
611                                {
612                                    NSNotificationCenter.defaultCenter().postNotification(FailedToContactMailServerNotification, this);
613                                    logger.error("Fatal Messaging Exception. Can't send the message after " + message.attemptsMade() + " attempts, abandoning.");
614                                    NSNotificationCenter.defaultCenter().postNotification(AbandonedMessageNotification, message.message(), new NSDictionary(e, ExceptionKey));
615                                }
616                                else
617                                {
618                                    logger.error("Non-fatal Messaging Exception. Failed to send message on attempt " + message.attemptsMade());
619                                    // Re-queue the message, will try it again later.
620                                    messages().push(message);
621                                }
622    
623                                // Get the transport to close and re-open so that we avoid sending nested messages
624                                break;
625                            }
626                        }
627    
628                        try
629                        {
630                            if (transport != null)
631                            {
632                                transport.close();
633                            }
634                        }
635                        catch(MessagingException e)
636                        {
637                            logger.error("Unable to close transport.  Perhaps it has already been closed? ", e);
638                        }
639                    }
640                    // Can't allow this to happen, the thread will die.
641                    catch (Throwable t)
642                    {
643                        logger.error("ThreadedMailAgent run() loop trying to throw unhandled exception: ", t);
644                    }
645                }
646                setThreadSuspended(messages().empty());
647            }
648        }

653         * Call this to signal this tread to end as soon as it sends all of the messages in the queue.
654         * Depending on what is in the queue, the thread might run for while
655         */
656        public void end()
657        {
658            logger.info("ThreadedMailAgent told to end, messages in queue: " + messages().size());
659            shouldEnd = true;
660            interrupt();
661            /** ensure [shouldEnd] shouldEnd();  **/
662        }
