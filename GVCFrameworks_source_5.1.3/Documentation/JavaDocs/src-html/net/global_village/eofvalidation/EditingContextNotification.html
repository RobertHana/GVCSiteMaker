004    /**
005     * The interface used by the NotifyingEditingContext to determine which EOs should receive notifications.
006     *
007     * @author Copyright (c) 2001-2005  Global Village Consulting, Inc.  All rights reserved.
008     * This software is published under the terms of the Educational Community License (ECL) version 1.0,
009     * a copy of which has been included with this distribution in the LICENSE.TXT file.
010     * @version $Revision: 4$
011     */
012    public interface EditingContextNotification
013    {
014    
015        /**
016         * Called after a newly inserted object has been saved to the persistent object store (a.k.a. database).
017         */
018        public void hasInserted();
019    
020        /**
021         * Called before deleteObject is invoked on our EOEditingContext.
022         */
023        public void willDelete();
024    
025        /**
026         * Called after a deleted object has been removed from the persistent object store.  This would handle the need to clean up related, non-EO, resources when the EO was deleted.
027         */
028        public void hasDeleted();
029    
030        /**
031         * Called before saveChanges is invoked on our EOEditingContext.  This would allow things such as setting the dateModified before saving an EO.  Note that changes in this method may result in more updated (or inserted or deleted!) objects, so it should iterate until there are no more changes.
032         */
033        public void willUpdate();
034    
035        /**
036         * Called after saveChanges is invoked on our EOEditingContext.
037         */
038        public void hasUpdated();
039    
040        /**
041         * Called after revert is invoked on our EOEditingContext.
042         */
043        public void hasReverted();
044    
045    }
