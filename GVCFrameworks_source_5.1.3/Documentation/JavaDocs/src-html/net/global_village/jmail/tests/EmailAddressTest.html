018    public class EmailAddressTest extends TestCase
019    {
020        protected String emailAddress1 = "someone.1@global-village.net";
021        protected String emailAddress2 = "someone.2@global-village.net";
022        protected String emailAddress3 = "someone.3@global-village.net";
023        protected String emailAddress4 = "someone.4@global-village.net";
024        protected NSArray oneAddressArray = new NSArray(emailAddress1);
025        protected NSArray threeAddressArray = new NSArray(new Object[] {emailAddress1, emailAddress2, emailAddress3});
026    
027    
028        /**
029         * Constructor for EmailAddressTest.
030         * @param name
031         */
032        public EmailAddressTest(String name)
033        {
034            super(name);
035        }
036    
037    
038    
039        /**
040         * Test isValidEmailAddressFormat().
041         */
042        public void testIsValidEmailAddressFormat()
043        {
044            assertTrue(EmailAddress.isValidAddressFormat("john@smith.com"));
045            assertTrue(EmailAddress.isValidAddressFormat("john.smith@john.smith.com"));
046            assertTrue( ! EmailAddress.isValidAddressFormat("@smith.com"));
047            assertTrue( ! EmailAddress.isValidAddressFormat("johnsmith.com"));
048            assertTrue( ! EmailAddress.isValidAddressFormat("john@"));
049            assertTrue( ! EmailAddress.isValidAddressFormat("john@smith@com"));
050            assertTrue( ! EmailAddress.isValidAddressFormat("john@smith"));
051            assertTrue( ! EmailAddress.isValidAddressFormat("john@smith."));
052            assertTrue( ! EmailAddress.isValidAddressFormat("john@.smith"));
053            assertTrue( ! EmailAddress.isValidAddressFormat("john@.smith."));
054            assertTrue( ! EmailAddress.isValidAddressFormat("http://www.gvcsitemaker.com/me@home.com"));
055        }
056        
057        
058        
059        /**
060         * Test internetAddresses and emailAddresses to ensure that all elements are maintained and the process is reversible..
061         */
062        public void testInternetAddresses()
063        {
064            addressReversabilityTest(NSArray.EmptyArray);
065            addressReversabilityTest(oneAddressArray);
066            addressReversabilityTest(threeAddressArray);
067        }
068    
069    
070    
071        /**
072         * Helper method for testInternetAddresses
073         */
074        protected void addressReversabilityTest(NSArray arrayToTest)
075        {
076            try
077            {
078                InternetAddress[] resultArray = EmailAddress.internetAddresses(arrayToTest);
079                assertTrue(arrayToTest.count() == resultArray.length);
080                assertTrue(arrayToTest.equals(EmailAddress.emailAddresses(resultArray)));
081            }
082            catch (AddressException e)
083            {
084                fail("Malformed e-mail address: " + e.getMessage());
085            }
086        }
087    
088    
089    }
