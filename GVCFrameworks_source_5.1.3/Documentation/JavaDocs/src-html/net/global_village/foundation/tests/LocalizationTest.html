019    public class LocalizationTest extends TestCase

021        private static final String GVCNonExistantString = "Not in any of the string files.";
022        private static final String GVCTestString = "This string is in the Test table";
023        private static final String GVCTestStringResult = "YES!";
024        private static final String GVCTestGVCLocalizationString = "This string is in the TestGVCLocalization table";
025        private static final String GVCTestGVCLocalizationStringResult = "Yay Yay Yay";
026        private static final String GVCGVCJFoundationString = "This string is in the GVCFoundation table";
027        private static final String GVCGVCJFoundationStringResult = "Yay Yay";
028        private static final String GVCLocalizedStringsString = "This string is in the LocalizedStrings table";
029        private static final String GVCLocalizedStringsStringResult = "Yay";

032        public LocalizationTest(String name)
033        {
034            super(name);
035        }

039        /**
040         * Tests localizedStringExistsForString.  Makes sure that the various files are checked properly.
041         */
042        public void testLocalizedStringExistsForString()
043        {
044            assertTrue( ! Localization.localizedStringExistsForString(this, GVCNonExistantString));
045            assertTrue(Localization.localizedStringExistsForString(this, GVCTestGVCLocalizationString));
046            assertTrue(Localization.localizedStringExistsForString(this, GVCGVCJFoundationString));
047            assertTrue(Localization.localizedStringExistsForString(this, GVCLocalizedStringsString));
048            assertTrue( ! Localization.localizedStringExistsForString(this, GVCTestString));
049    
050            DynamicInstanceVariable.setInstanceVariable(this, Localization.GVCLocalizedFileNameVariable, "Test");
051            // Test same four again after setting the file name DIV.
052            assertTrue( ! Localization.localizedStringExistsForString(this, GVCNonExistantString));
053            assertTrue(Localization.localizedStringExistsForString(this, GVCTestGVCLocalizationString));
054            assertTrue(Localization.localizedStringExistsForString(this, GVCGVCJFoundationString));
055            assertTrue(Localization.localizedStringExistsForString(this, GVCLocalizedStringsString));
056            assertTrue(Localization.localizedStringExistsForString(this, GVCTestString));
057    
058            DynamicInstanceVariable.setInstanceVariable(this, Localization.GVCLocalizedBundleVariable, 
059                NSBundleAdditions.anyBundle(NSBundle.mainBundle()));
060            assertTrue( ! Localization.localizedStringExistsForString(this, "Test file is not in the main bundle"));
061    
062            DynamicInstanceVariable.removeInstanceVariable(this, Localization.GVCLocalizedBundleVariable);
063            DynamicInstanceVariable.setInstanceVariable(this, Localization.GVCLocalizedBundleVariable, NSBundle.bundleForClass(getClass()));
064            assertTrue(Localization.localizedStringExistsForString(this, "This is a test string"));
065    
066            assertTrue( ! Localization.localizedStringExistsForString(this, "This is NOT a test string"));
067        }
068

071        /**
072         * Tests localizedStringForString.  Makes sure that the various files are checked properly.
073         */
074        public void testLocalizedStringForString()
075        {
076            String aLocalizedString;
077    
078            try
079            {
080                Localization.localizedStringForString(this, null);
081                fail("Expected exception did not occur");
082            }
083            catch (RuntimeException e) { }
084    
085            try
086            {
087                Localization.localizedStringForString(this, GVCNonExistantString);
088                fail("Expected exception did not occur");
089            }
090            catch (RuntimeException e) { }
091    
092            try
093            {
094                Localization.localizedStringForString(this, GVCTestString);
095                fail("Expected exception did not occur");
096            }
097            catch (RuntimeException e) { }
098    
099            aLocalizedString = Localization.localizedStringForString(this, GVCTestGVCLocalizationString);
100            assertEquals(aLocalizedString, GVCTestGVCLocalizationStringResult);
101    
102            aLocalizedString = Localization.localizedStringForString(this, GVCGVCJFoundationString);
103            assertEquals(aLocalizedString, GVCGVCJFoundationStringResult);
104    
105            aLocalizedString = Localization.localizedStringForString(this, GVCLocalizedStringsString);
106            assertEquals(aLocalizedString, GVCLocalizedStringsStringResult);
107    
108    
109            // Test again once the file name DIV is set.
110            DynamicInstanceVariable.setInstanceVariable(this,
111                                                        Localization.GVCLocalizedFileNameVariable,
112                                                        "Test");
113    
114            try
115            {
116                Localization.localizedStringForString(this, GVCNonExistantString);
117                fail("Expected exception did not occur");
118            }
119            catch (RuntimeException e) { }
120    
121            aLocalizedString = Localization.localizedStringForString(this, GVCTestString);
122            assertEquals(aLocalizedString, GVCTestStringResult);
123    
124            aLocalizedString = Localization.localizedStringForString(this, GVCTestGVCLocalizationString);
125            assertEquals(aLocalizedString, GVCTestGVCLocalizationStringResult);
126    
127            aLocalizedString = Localization.localizedStringForString(this, GVCGVCJFoundationString);
128            assertEquals(aLocalizedString, GVCGVCJFoundationStringResult);
129    
130            aLocalizedString = Localization.localizedStringForString(this, GVCLocalizedStringsString);
131            assertEquals(aLocalizedString, GVCLocalizedStringsStringResult);
132    
133    
134            DynamicInstanceVariable.setInstanceVariable(this,
135                                                        Localization.GVCLocalizedBundleVariable,
136                                                        NSBundleAdditions.anyBundle(NSBundle.mainBundle()));
137            try
138            {
139                Localization.localizedStringForString(this, "Test file is not in the main bundle");
140                fail("Expected exception did not occur");
141            }
142            catch (RuntimeException e) { }
143    
144            DynamicInstanceVariable.removeInstanceVariable(this, Localization.GVCLocalizedBundleVariable);
145            DynamicInstanceVariable.setInstanceVariable(this,
146                                                        Localization.GVCLocalizedBundleVariable,
147                                                        NSBundle.bundleForClass(getClass()));
148            aLocalizedString = Localization.localizedStringForString(this, "This is a test string");
149            assertEquals(aLocalizedString, "Hello!");
150    
151            try
152            {
153                Localization.localizedStringForString(this, "This is NOT a test string");
154                fail("Expected exception did not occur");
155            }
156            catch (RuntimeException e) { }
157        }
