014    public class ClassAdditions
015    {
016    
017    
018        /**
019         * Static methods only.  You'll never need to instantiate this class.
020         */
021        private ClassAdditions()
022        {
023            super();
024        }
025    
026    
027    
028        /**
029         * Returns true if the params represent the same class, irrespective of class loaders.
030         *
031         * @param class1 one of the classes to compare
032         * @param class2 one of the classes to compare
033         * @return <code>true</code> if the two classes represent the same classes (even if they were loaded by different ClassLoaders), <code>false</code> otherwise
034         */
035        public static boolean classesAreEqual(java.lang.Class class1, java.lang.Class class2)
036        {
037            /** require [valid_class1_param] class1 != null; [valid_class2_param] class2 != null; **/
038    
039            // WOW, lame as this seems, its the only thing that seems to work...
040            return class1.toString().equals(class2.toString());
041        }
042    
043    
044    
045        /**
046         * Returns true if the params represent the same class.
047         *
048         * @param classString1 one of the classes to compare
049         * @param classString2 one of the classes to compare
050         * @return <code>true</code> if the two classes represent the same classes (even if they were loaded by different ClassLoaders), <code>false</code> otherwise
051         */
052        public static boolean classesAreEqual(String classString1, String classString2)
053        {
054            /** require
055            [valid_classString1_param] classString1 != null;
056            [valid_classString2_param] classString2 != null; **/
057    
058            return classString1.equals(classString2);
059        }
060    
061    
062    
063        /**
064         * Convenience method for getting just the classname (not fully qualified).
065         *
066         * @param aClass the Class whose unqualified name we are getting
067         * @return <code>aClass</code>'s unqualified name (e.g. String vs java.lang.String)
068         */
069        public static String unqualifiedClassName(java.lang.Class aClass)
070        {
071            /** require [valid_param] aClass != null; **/
072    
073            NSArray classNameComponents = NSArray.componentsSeparatedByString(aClass.getName(), ".");
074            return (String)classNameComponents.lastObject();
075    
076            /** ensure [valid_result] Result != null; **/
077        }
078    
079    
080    
081        /**
082         * A hack to get around a limitation in Jass that doesn't allow the NSArray.class construction.  This method uses <code>Class.forName()</code> and assumes that the class name passed in exists.  A <code>RuntimeException</code> is thrown if it doesn't.
083         *
084         * @param className the class name to return
085         * @return the class represented by the param
086         * @exception RuntimeException if the class doesn't exist
087         */
088        public static Class unsafeClassForName(String className)
089        {
090            /** require [valid_param] className != null; **/
091    
092            try
093            {
094                return Class.forName(className);
095            }
096            catch (ClassNotFoundException e)
097            {
098                throw new ExceptionConverter(e);
099            }
100    
101            /** ensure [valid_result] Result != null; **/
102        }
103    
104    
105    
106    }
