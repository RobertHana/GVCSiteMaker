008    /** 
009     * <p>Qualifier to find an object with a given attribute as close as possible to a specified value.</p>
010     * 
011     * <p>The qualifier is initialized with the key path to the value to approximate, the target value and a
012     * qualifierOperator specifying how to approximate the limit. Usually this qualifier will match at most one
013     * object. Thus, in most cases, sensible results are obtained only if the key path represents an attribute
014     * or a to-one relationship. In cases where a to-many relationship is involved, the BestRelationshipMatchesQualifier
015     * is likely to be most appropriate for the task.</p>
016     * 
017     * <p>An alternate way of using the qualifier is to specify an optional array of valuing key paths. This allows for
018     * handling cases where the objects to be fetched are valued by relationships in a way that the best match is
019     * different for each value in the relationship. The qualifier will thus match as many objects as there are
020     * distinct value combinations reachable by theses key paths.</p>
021     * 
022     * <p>For database qualification the best matching entry is searched for among all commited instances of that
023     * entity. The list may actually be restricted by using a qualifier. For in-memory evaluation such a list must
024     * be made available to the qualifier. This is done by calling either prepareWithArray() or qualifierPreparedWithArray().</p>
025     * 
026     * <p>A null (or NSKeyValueCoding.NullValue) limit allows for matching of the object with largest or lowest value
027     * at keyPath.
028     * 
029     * <p>This sample code is provided for educational purposes. It is mainly to be considered a source of information 
030     * and a way to spark off discussion. I make no warranty, expressed or implied, about the quality of this code or 
031     * its usefulness in any particular situation. Use this code or any code based on it at your own risk. Basically, 
032     * enjoy the read, but don't blame me for anything.</p>
033     * 
034     * @author Pierre Bernard http://homepage.mac.com/i_love_my/webobjects.html
035     */

145             * @param qualifierOperator qualifierOperator used to approximate the limit
146             * @param subQualifier qualifier used to limit the list of accepatble values
147             * @param uniquingPaths list of key paths to the valuing attributes

149            public BestMatchQualifier(
150                    String keyPath,
151                    Object limit,
152                    NSSelector qualifierOperator,
153                    EOQualifier subQualifier,
154                    NSArray uniquingPaths)
155            {
156                    this.keyPath = keyPath;
157                    this.limit = limit;
158                    this.qualifierOperator = qualifierOperator;
159                    this.subQualifier = subQualifier;
160                    this.uniquingPaths = (uniquingPaths != null) ? uniquingPaths : NSArray.EmptyArray;
161    
162                    if (this.keyPath == null)
163                    {
164                            throw new IllegalArgumentException("The keyPath may not be null");
165                    }
166    
167                    if (!availableSelectors.containsObject(this.qualifierOperator))
168                    {
169                            throw new IllegalArgumentException("qualifierOperator not applicable for this qualifier");
170                    }
171    
172                    if (this.limit == null)
173                    {
174                            this.limit = NSKeyValueCoding.NullValue;
175                    }
176            }

205            /** Prepares the qualifier for in-memory qualifying.</p></p>
206             * 
207             * The best matching values (usually one) are picked from the array. Later evaluations
208             * need only to determine if the object to evaluate matches the found value.
209             * 
210             * @param array the array from which to pick values
211             */
212            public void prepareWithArray(NSArray array)
213            {
214                    resetBestValues();
215    
216                    if (array != null)
217                    {
218                            Enumeration en = array.objectEnumerator();
219    
220                            while (en.hasMoreElements())
221                            {
222                                    Object object = en.nextElement();
223                                    ValueHolder bestValueHolder = bestValueHolder(object, true);
224                                    Object bestValue = bestValueHolder.value();
225    
226                                    if ((subQualifier() == null) || (subQualifier().evaluateWithObject(object)))
227                                    {
228                                            Object value =
229                                                    NSKeyValueCodingAdditions.Utility.valueForKeyPath(object, keyPath());
230                                            Object limit = limit();
231                                            boolean resultOne =
232                                                    (limit == NSKeyValueCoding.NullValue)
233                                                            || EOQualifier.ComparisonSupport.compareValues(
234                                                                    value,
235                                                                    limit,
236                                                                    qualifierOperator());
237    
238                                            if (resultOne)
239                                            {
240                                                    boolean resultTwo =
241                                                            (bestValue == null)
242                                                                    || EOQualifier.ComparisonSupport.compareValues(
243                                                                            bestValue,
244                                                                            value,
245                                                                            qualifierOperator());
246    
247                                                    if (resultTwo)
248                                                    {
249                                                            bestValue = value;
250                                                    }
251                                            }
252                                    }
253    
254                                    bestValueHolder.setValue(bestValue);
255                            }
256                    }
257            }

259            public BestMatchQualifier qualifierPreparedWithArray(NSArray array)
260            {
261                    BestMatchQualifier clone = (BestMatchQualifier) clone();
262    
263                    clone.prepareWithArray(array);
264    
265                    return clone;

268            public boolean evaluateWithObject(Object object)
269            {
270                    if (this.bestValues != null)
271                    {
272                            ValueHolder bestValueHolder = bestValueHolder(object, false);
273                            Object bestValue = (bestValueHolder != null) ? bestValueHolder.value() : null;
274                            Object value = NSKeyValueCodingAdditions.Utility.valueForKeyPath(object, keyPath());
275    
276                            return (
277                                    (bestValue != null)
278                                            && (bestValue.equals(value))
279                                            && ((subQualifier() == null) || (subQualifier().evaluateWithObject(object))));
280                    }
281                    else
282                    {
283                            throw new IllegalStateException("The qualifier needs to be prepared first by passing it a value array");
284                    }
285            }

287            public EOQualifier qualifierWithBindings(NSDictionary bindings, boolean requiresAll)
288            {
289                    if (this.limit instanceof EOQualifierVariable)
290                    {
291                            Object value = null;
292    
293                            if (bindings != null)
294                            {
295                                    value = bindings.valueForKeyPath(((EOQualifierVariable) limit).key());
296                            }
297    
298                            if (value != null)
299                            {
300                                    BestMatchQualifier clone = (BestMatchQualifier) clone();
301    
302                                    clone.setLimit(value);
303    
304                                    return clone;
305                            }
306    
307                            if (requiresAll)
308                            {
309                                    throw new EOQualifier.QualifierVariableSubstitutionException(
310                                            "Error in variable substitution: value for variable "
311                                                    + this.limit
312                                                    + " not found");
313                            }
314                            else
315                            {
316                                    return null;
317                            }
318                    }
319                    else
320                    {
321                            return this;
322                    }
323            }

338            public void addQualifierKeysToSet(NSMutableSet keySet)
339            {
340                    keySet.addObject(keyPath());
341    
342                    if (subQualifier() != null)
343                    {
344                            NSMutableSet subKeySet = new NSMutableSet();
345    
346                            subQualifier().addQualifierKeysToSet(subKeySet);
347    
348                            Enumeration subKeys = subKeySet.objectEnumerator();
349                            String prefix = keyPath() + NSKeyValueCodingAdditions.KeyPathSeparator;
350    
351                            while (subKeys.hasMoreElements())
352                            {
353                                    keySet.addObject(prefix + subKeys.nextElement());
354                            }
355                    }
356    
357                    keySet.addObjectsFromArray(uniquingPaths());
358            }

360            public boolean equals(Object object)
361            {
362                    if (object instanceof BestMatchQualifier)
363                    {
364                            BestMatchQualifier other = (BestMatchQualifier) object;
365    
366                            return (
367                                    (this.keyPath.equals(other.keyPath))
368                                            && (this.limit.equals(other.limit))
369                                            && (this.qualifierOperator.equals(other.qualifierOperator))
370                                            && (((this.subQualifier == null) && (this.subQualifier == other.subQualifier))
371                                                    || (this.subQualifier.equals(other.subQualifier)))
372                                            && (this.uniquingPaths.equals(other.uniquingPaths)));
373                    }
374    
375                    return false;
376            }

391            public BestMatchQualifier clone(NSDictionary newValues)
392            {
393                    BestMatchQualifier clone = (BestMatchQualifier) clone();
394    
395                    if (newValues != null)
396                    {
397                            Enumeration en = newValues.keyEnumerator();
398    
399                            while (en.hasMoreElements())
400                            {
401                                    String key = (String) en.nextElement();
402                                    Object value = newValues.objectForKey(key);
403    
404                                    NSKeyValueCoding.Utility.takeValueForKey(clone, value, key);
405                            }
406                    }
407    
408                    return clone;
409            }

411            public String toString()
412            {
413                    StringBuffer buffer = new StringBuffer();
414    
415                    buffer.append("(");
416                    buffer.append(this.keyPath);
417                    buffer.append(" = ");
418                    buffer.append(
419                            ((this.qualifierOperator.equals(QualifierOperatorLessThan)
420                                    || (this.qualifierOperator.equals(QualifierOperatorLessThanOrEqualTo)))
421                                    ? "MAX("
422                                    : "MIN("));
423                    buffer.append(this.keyPath);
424                    buffer.append(")");
425    
426                    if (this.limit != NSKeyValueCoding.NullValue)
427                    {
428                            String value;
429    
430                            if (this.limit instanceof Number)
431                            {
432                                    value = this.limit.toString();
433                            }
434                            else if (this.limit instanceof EOQualifierVariable)
435                            {
436                                    value = "$" + ((EOQualifierVariable) this.limit).key();
437                            }
438                            else
439                            {
440                                    value = "'" + this.limit + "'";
441                            }
442                            buffer.append(" WHERE ");
443                            buffer.append(this.keyPath);
444                            buffer.append(
445                                    ((this.qualifierOperator.equals(QualifierOperatorLessThan)
446                                            || (this.qualifierOperator.equals(QualifierOperatorLessThanOrEqualTo)))
447                                            ? " <"
448                                            : " >"));
449                            buffer.append(
450                                    ((this.qualifierOperator.equals(QualifierOperatorLessThanOrEqualTo)
451                                            || (this.qualifierOperator.equals(QualifierOperatorGreaterThanOrEqualTo)))
452                                            ? "= "
453                                            : " "));
454                            buffer.append(value);
455                    }
456    
457                    if (this.subQualifier != null)
458                    {
459                            if (this.limit != NSKeyValueCoding.NullValue)
460                            {
461                                    buffer.append(" AND ");
462                            }
463                            else
464                            {
465                                    buffer.append(" WHERE ");
466                            }
467                    }
468    
469                    buffer.append(this.subQualifier.toString());
470    
471                    if (this.uniquingPaths != null)
472                    {
473                            buffer.append(") FOR EACH ");
474                            buffer.append(this.uniquingPaths);
475                    }
476                    else
477                    {
478                            buffer.append(")");
479                    }
480    
481                    return buffer.toString();
482            }

484            // Protected instance methods
485    
486            protected void resetBestValues()
487            {
488                    int count = uniquingPaths.count();
489    
490                    if (count > 0)
491                    {
492                            this.bestValues = new HashMap();
493                    }
494                    else
495                    {
496                            this.bestValues = new ValueHolder();
497                    }
498            }
499    
500            protected ValueHolder bestValueHolder(Object object, boolean mayExpand)
501            {
502                    int count = uniquingPaths.count();
503                    Object holder = this.bestValues;
504    
505                    for (int i = 0; i < count; i++)
506                    {
507                            String path = (String) uniquingPaths.objectAtIndex(i);
508                            Object key = NSKeyValueCodingAdditions.Utility.valueForKeyPath(object, path);
509                            HashMap hashMap = (HashMap) holder;
510    
511                            holder = hashMap.get(key);
512    
513                            if (holder == null)
514                            {
515                                    if (mayExpand)
516                                    {
517                                            if (i < (count - 1))
518                                            {
519                                                    holder = new HashMap();
520                                            }
521                                            else
522                                            {
523                                                    holder = new ValueHolder();
524                                            }
525    
526                                            hashMap.put(key, holder);
527                                    }
528                                    else
529                                    {
530                                            return null;
531                                    }
532                            }
533                    }
534    
535                    return (ValueHolder) holder;
536            }
