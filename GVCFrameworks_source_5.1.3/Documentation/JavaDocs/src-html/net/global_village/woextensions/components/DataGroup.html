012    /**
013     * A <code>DataGroup</code> organizes a sorted group of similar Enterprise Objects and is able to split it into sub-subgroups to model a hierarchy of EOs.  It is intended to be used as the model for drill down components.  Representative or aggregate values from the contained EOs of the <code>DataGroup</code> are accessed through Key-Value coding. Note that a <code>DataGroup</code> object only handles one level of hierarchy (other level are handled by other <code>DataGroups</code>).
014     *
015     * To create a <code>DataGroup</code>, you need to pass in the following:
016     * <ul>
017     * <li>data - an array of EOs to be managed by this <code>DataGroup</code>; all the <code>EOs</code> must be of the same type family, which means that all these EOs must respond to the same keypaths and compare methods passed during its creation)
018     * <li>groupAttributeKeyPath - the attribute keyPath used for grouping this data; if you want to perform calculations for a specific attribute, use this format "@<aggregate function>.attribute", ie "@sum.hours", "@avg.hours"
019     * <li>attributeKeyPaths - the ordered array of attribute keyPaths that will determine the groupAttributeKeyPath to be used to generate its subGroups; groupAttributeKeyPath of this DataGroup must also belong to attributeKeyPaths;
020     * <li>comparators - the ordered array of NSComparator instances that will be used when generating the subGroups; it is expected that the NSComparator at a particular index will result in objects which are ordered in a sensible way with respect to the attributeKeyPath at the matching index. (WHAT DOES THIS MEAN? - ch: If they do not match, more than one group with the same title will be created.)
021     * </ul>
022     *
023     * Below is an example that creates a <code>DataGroup</code> for the array of EOCustomObjects, the arrayOfWorkDetails().  It is created for the attribute key path "project.name" which is the first level of the hiearchy. From these param values, this <code>DataGroup's subGroup</code> will contain dataGroups for "taskArea.area.name", which is the next attribute key path from the <code>attributeKeyPaths</code> array.:<br>
024     * <code>
025     *     DataGroup newDataGroup = new DataGroup("project.name",
026     *                                             arrayOfWorkDetails(),
027     *                                             new Array(new Object[] {"project.name", "taskArea.area.name", "taskArea.task.name", "user.fullName", "@sum.hours"}),
028     *                                             new Array(new Object[] {"compareByProject", "compareByArea", "compareByTask", "compareByUser", "compareByHours"})),
029     * </code>
030     *
031     * Note that a DataGroup does not require valid attributeKeyPaths for it to function properly, you can pass in a dummy attribute key path like "$dummyAttribute" to initialize your DataGroups. For example, if you want to create the root DataGroup for an array of workDetails with the "project.name" as the first level, you can create it using this...
032     *
033     * <code>
034     *     DataGroup newDataGroup = new DataGroup("$dummyAttribute%",
035     *                                             arrayOfWorkDetails(),
036     *                                             new Array(new Object[] {"$dummyAttribute%", "project.name", "taskArea.area.name", "taskArea.task.name", "user.fullName", "@sum.hours"}),
037     *                                             new Array(new Object[] {"$dummyCompare%", "compareByProject", "compareByArea", "compareByTask", "compareByUser", "compareByHours"})),
038     * </code>
039     *
040     * @author Copyright (c) 2001-2005  Global Village Consulting, Inc.  All rights reserved.
041     * This software is published under the terms of the Educational Community License (ECL) version 1.0,
042     * a copy of which has been included with this distribution in the LICENSE.TXT file.
043     * @version $Revision: 6$
044     *
045     */
046    public class DataGroup extends Object

048        protected String groupAttributeKeyPath;
049        protected NSArray data;
050        protected NSArray attributeKeyPaths;
051        protected NSArray comparators;
052        protected NSMutableDictionary cachedValues;
053        protected NSMutableArray cachedSubGroups;
054        protected boolean isExpanded;

057        /**
058         * Designated constructor.
059         *
060         * @param theGroupAttributeKeyPath the attribute keyPath used for grouping this data
061         * @param theData the EOCustomObjects to be managed by this DataGroup
062         * @param theAttributeKeyPaths the ordered array of attribute keyPaths that will determine the groupAttributeKeyPath to be used to generate its subGroups
063         * @param theComparators the ordered array of NSComparators that will be used when generating the subGroups
064         */
065        public DataGroup(String theGroupAttributeKeyPath,
066                         NSArray theData,
067                         NSArray theAttributeKeyPaths,
068                         NSArray theComparators)
069        {
070            super();
071            /** require
072            [valid_theGroupAttributeKeyPath_param] theGroupAttributeKeyPath != null;
073            [valid_theData_param] theData != null;
074            [has_data] theData.count() > 0;
075            [all_data_are_EOCustomObject] forall i : {0 .. theData.count() - 1} # theData.objectAtIndex(i) instanceof EOCustomObject;
076            [valid_theAttributeKeyPaths_param] theAttributeKeyPaths != null;
077            [attribute_keypaths_contain_group_keypath] theAttributeKeyPaths.containsObject(theGroupAttributeKeyPath);
078            [valid_theComparators_param] theComparators != null;
079            [has_comparator_for_each_keypath] theAttributeKeyPaths.count() == theComparators.count(); **/
080    
081            groupAttributeKeyPath = theGroupAttributeKeyPath;
082            data = theData;
083            attributeKeyPaths = theAttributeKeyPaths;
084            comparators = theComparators;
085            
086            isExpanded = false;
087            cachedValues = new NSMutableDictionary();
088            cachedSubGroups = null;
089        }

105        /**
106         * Sets this group's data with the newData.
107         */
108        public synchronized void setData(NSArray newData)
109        {
110            /** require [valid_param] newData != null; **/
111    
112            if ((data == null) || ( ! data.isEqualToArray(newData)))
113            {
114                data = newData;
115    
116                // Discard old cached data.
117                setCachedValues(new NSMutableDictionary());
118                setCachedSubGroups(null);
119            }
120    
121            /** ensure [data_changed] data.isEqualToArray(newData); **/
122        }

169        /**
170         * Returns the object that corresponds to <code>aKeyPath</code> in the <code>data</code> array if aKeyPath is in attributeKeyPaths(), or from the <code>dataGroup</code> otherwise.
171         *
172         * @param aKeyPath the attribute key path of the value to look for
173         * @return the object that corresponds to <code>aKeyPath</code> in the <code>data</code> array.
174         */
175        public Object valueForKeyPath(String aKeyPath)
176        {
177            /** require [valid_param] aKeyPath != null; **/
178    
179            Object theObject = NSKeyValueCoding.NullValue;
180    
181            if (attributeKeyPaths().containsObject(aKeyPath))
182            {
183                theObject = cachedValues().objectForKey(aKeyPath);
184    
185                if (theObject == null)
186                {
187                   //try to get the value from data and add it to the dictionary...
188                    
189                    if (aKeyPath.indexOf("@") == -1)             //if aKeyPath does not contain an "@" sign
190                    {
191                        theObject = ((EOCustomObject)data.objectAtIndex(0)).valueForKeyPath(aKeyPath);
192                    }
193                    else
194                    {
195                        theObject = data.valueForKey(aKeyPath);                    
196                    }
197    
198                    cachedValues().setObjectForKey(theObject, aKeyPath);
199                }
200            }
201            // We need to defer to super so that we can use binding in WOComponents!
202            else
203            {
204                theObject = valueForKeyPath(aKeyPath);
205            }
206    
207            return theObject;
208    
209            /** ensure [valid_result] Result != null; **/
210        }

213        /**
214         * Returns <code>true</code> if this data group contains subGroups, <code>false</code> otherwise.
215         *
216         * @return <code>true</code> if this data group contains subGroups, <code>false</code> otherwise.

218        public boolean hasSubGroups()
219        {
220            return ( ! isLastGroup()) || (nextAttributeKeyPath().indexOf("@") != -1);
221        }

228         * @return <code>true</code> if this data group is the last group, <code>false</code> otherwise.
229         */
230        public boolean isLastGroup()
231        {
232            return attributeKeyPaths().lastObject().equals(groupAttributeKeyPath());
233        }

277        /**
278         * Toggles the value of <code>isExpanded</code>.  It generates the <code>cachedSubGroups</code> if necessary.
279         */
280        public void toggleExpansion()
281        {
282            /** require [has_subgroups] hasSubGroups(); **/
283    
284            isExpanded = ( ! isExpanded());
285    
286            if ((isExpanded()) && (cachedSubGroups() == null))
287            {
288                String attributeKeyPathForSubGroup = nextAttributeKeyPath();
289                NSComparator comparatorForSubGroup = (NSComparator)comparators.objectAtIndex(attributeKeyPaths.indexOfObject(attributeKeyPathForSubGroup));
290    
291                //Sort data() using comparator, converting any comparison exceptions to runtime ones.
292                NSArray sortedObjects = NSArrayAdditions.sortedArrayUsingComparator(data(), comparatorForSubGroup);    
293                
294                //Iterate through sortedObjects
295                String keyPathToBeUsedForGrouping = attributeKeyPathForSubGroup;
296                Object valueCommonToGroup = null;
297                NSMutableArray groupedData = new NSMutableArray();
298                cachedSubGroups = new NSMutableArray();
299                    
300                Enumeration objectEnum = sortedObjects.objectEnumerator();
301                while (objectEnum.hasMoreElements())
302                {
303                    EOCustomObject anObject = (EOCustomObject) objectEnum.nextElement();
304    
305                    if (valueCommonToGroup == null)
306                    {
307                        valueCommonToGroup = anObject.valueForKeyPath(keyPathToBeUsedForGrouping);
308                    }
309    
310                    if ( ! anObject.valueForKeyPath(keyPathToBeUsedForGrouping).equals(valueCommonToGroup))
311                    {
312    
313                        DataGroup newDataGroup = new DataGroup(keyPathToBeUsedForGrouping,
314                                                               groupedData,
315                                                               attributeKeyPaths(),
316                                                               comparators());
317                        cachedSubGroups.addObject(newDataGroup);
318                        groupedData = new NSMutableArray();
319                        valueCommonToGroup = anObject.valueForKeyPath(keyPathToBeUsedForGrouping);
320                    }
321                    groupedData.addObject(anObject);
322                }
323                if (groupedData.count() > 0)
324                {
325                    DataGroup newDataGroup = new DataGroup(keyPathToBeUsedForGrouping,
326                                                           groupedData,
327                                                           attributeKeyPaths(),
328                                                           comparators());
329                    cachedSubGroups.addObject(newDataGroup);
330                }
331            }
332    
333            /** ensure [is_not_expanded_or_has_cached_subgroups] ((isExpanded() && (cachedSubGroups() != null))) || (! isExpanded()); **/
334        }

338        /**
339         * Recurse through all the data group's subGroups and return them in a one-level array.
340         */
341        public NSArray allSubGroups()
342        {
343            /** require [has_subgroups] hasSubGroups(); **/
344    
345            NSMutableArray visibleDataGroups = new NSMutableArray();
346    
347            Enumeration subGroupEnumeration = subGroups().objectEnumerator();
348            while (subGroupEnumeration.hasMoreElements())
349            {
350                DataGroup currentDataGroup = (DataGroup)subGroupEnumeration.nextElement();
351                visibleDataGroups.addObject(currentDataGroup);
352                if (currentDataGroup.isExpanded())
353                {
354                    visibleDataGroups.addObjectsFromArray(currentDataGroup.allSubGroups());    
355                }
356            }
357    
358            return visibleDataGroups;
359    
360            /** ensure [is_not_expanded_or_has_cached_subgroups] ((isExpanded() && (cachedSubGroups() != null))) || (! isExpanded()); **/
361        }

407        public void expandAll()
408        {
409            if (hasSubGroups() && (nextAttributeKeyPath().indexOf("@") == -1))
410            {
411                toggleExpansion();
412    
413                Enumeration subGroupEnum = subGroups().objectEnumerator();
414                while (subGroupEnum.hasMoreElements())
415                {
416                    DataGroup aSubGroup = (DataGroup) subGroupEnum.nextElement();
417                    aSubGroup.expandAll();
418                }
419            }
420        }

428          [comparators_not_null] comparators != null;
