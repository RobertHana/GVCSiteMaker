001    package net.global_village.woextensions.securecomponents;
002    
003    import com.webobjects.appserver.*;
004    
005    /**
006     * A Text Input control which can conditionally display itself as a TextArea or a string.  This allows the UI to dynamically change based on object state or user permissions.  It must be placed inside a WOForm.  When isEditable is bound to true it displays a WOText containing value() when isEditable is bound to false it displays a WOString containing value().  The cols and rows binding can be used to control the size of the WOText and the maxLength binding can be used to limit how much can be typed into the WOText.
007     *
008     * @author Copyright (c) 2001-2005  Global Village Consulting, Inc.  All rights reserved.
009     * This software is published under the terms of the Educational Community License (ECL) version 1.0,
010     * a copy of which has been included with this distribution in the LICENSE.TXT file.
011     * @version $Revision: 4$
012     */
013    public class SecureTextAreaComponent extends SecureComponent
014    {
015    
016        /**
017         * Designated constructor.
018         */
019        public SecureTextAreaComponent(WOContext aContext)
020        {
021            super(aContext);
022        }
023    
024        
025    
026        /**
027         * Accessor for value binding.  The binding provides the displayed content of the WOComponent
028         *
029         * @return the value bound to value
030         */
031        public Object value()
032        {
033            return valueForBinding("value");
034        }
035    
036      
037       
038        /**
039         * Accessor for name binding.  This bidding can be used to assign a name to the WOComponent for use with JavaScript
040         *
041         * @return the value bound to name
042         */
043        public String componentName()
044        {
045            return (String) valueForBinding("name");
046        }
047    
048    
049    
050        /**
051         * Accessor for maxLength binding.  This bidding can be used to assign a maxLength to the WOTextField
052         *
053         * @return the value bound to maxLength
054         */
055        public String maxLength()
056        {
057            return (String) valueForBinding("maxLength");
058        }
059    
060    
061    
062        /**
063         * Accessor for cols binding.  This bidding can be used to control the size of the WOText
064         *
065         * @return the value bound to cols
066         */
067        public String cols()
068        {
069            return (String) valueForBinding("cols");
070        }
071    
072    
073    
074        /**
075         * Accessor for rows binding.  This bidding can be used to control the size of the WOText
076         *
077         * @return the value bound to rows
078         */
079        public String rows()
080        {
081            return (String) valueForBinding("rows");
082        }
083    }
