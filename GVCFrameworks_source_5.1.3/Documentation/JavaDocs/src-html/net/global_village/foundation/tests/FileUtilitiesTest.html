012    /**
013     * Tests for FileUtilities.
014     *
015     * @see net.global_village.foundation.FileUtilities
016     *
017     * @author Copyright (c) 2006-2006 Harte-Hanks Shoppers, Inc.
018     */
019    public class FileUtilitiesTest extends TestCase
020    {
021    
022    
023        public FileUtilitiesTest(String name)
024        {
025            super(name);
026        }
027    
028    
029    
030        /**
031         * Test basic functionality.
032         */
033        public void testBasic()
034        {
035            try
036            {
037                InputStream testDelimitedData = NSBundle.bundleForClass(FileUtilities.class).inputStreamForResourcePath("Tests/TabDelimitedFile.csv");
038                NSArray testData = FileUtilities.dataFromDelimitedFile(testDelimitedData, ",", null);
039                assertEquals(7, testData.count());
040    
041                // Test row with all fields
042                NSDictionary row2 = (NSDictionary)testData.objectAtIndex(1);
043                assertEquals("02851033B713", row2.objectForKey("Ad Key"));
044                assertEquals("AdControlBatchMessages", row2.objectForKey("Used In Tests"));
045                assertEquals("Suspended PAD Show", row2.objectForKey("Comments"));
046    
047                // Test row with missing fields
048                NSDictionary row6 = (NSDictionary)testData.objectAtIndex(5);
049                System.out.println(row6);
050                assertEquals("02917948M713", row6.objectForKey("Ad Key"));
051                assertEquals("", row6.objectForKey("Used In Tests"));
052                assertEquals("", row6.objectForKey("Comments"));
053            }
054            catch (IOException e)
055            {
056                fail("Threw IOException " + e.getLocalizedMessage());
057            }
058            catch (IllegalArgumentException e)
059            {
060                fail("Threw IllegalArgumentException " + e.getLocalizedMessage());
061            }
062        }
063    
064    }
