030        /**
031         * A convenience method wrapping <code>array.sortedArrayUsingComparator(comparator)</code> which converts <code>NSComparator.ComparisonException</code> into a <code>RuntimeException</code>.
032         *
033         * @param array the list of objects to sort
034         * @param comparator the <code>NSComparator</code> instance to use to sort array
035         * @return the objects in array sorted by comparator
036         * @exception RuntimeException if <code>NSArray.sortedArrayUsingComparator</code> raises a <code>NSComparator.ComparisonException</code> exception
037         */
038        public static NSArray sortedArrayUsingComparator(NSArray array, NSComparator comparator)
039        {
040            /** require
041            [valid_array_param] array != null;
042            [valid_comparator_param] comparator != null; **/
043    
044            NSArray orderedList = null;
045    
046            try
047            {
048                orderedList = array.sortedArrayUsingComparator( comparator );
049            }
050            catch (com.webobjects.foundation.NSComparator.ComparisonException e)
051            {
052                throw new ExceptionConverter(e);
053            }
054    
055            JassAdditions.post("NSArrayAdditions", "sortedArrayUsingComparator", orderedList.count() == array.count());
056            return orderedList;
057    
058            /** ensure [valid_result] Result != null; **/
059        }

089        /**
090         * A string specific method to replace NSArray.containsObject which uses equals() for comparisons.
091         * This method ignores case when determining if an array contains a string.
092         * 
093         * @param array <b>must be an array of java.lang.String
094         * @param targetString the string to find in array
095         * @return true if target can be found in array, ignoring case
096         */
097        public static boolean caseInsensitiveContains(NSArray array, String targetString) 
098        {
099            /** require [valid_array] array != null;  
100                        [valid_targetString] targetString != null;       **/
101            
102            boolean contains = false;
103            
104            Enumeration arrayEnum = array.objectEnumerator();
105            while (arrayEnum.hasMoreElements() && ! contains) {
106                contains = targetString.equalsIgnoreCase((String) arrayEnum.nextElement());
107            }
108            
109            return contains;
110        }

114        /**
115         * Inserts objectsToInsert into array at indexToInsertAt.
116         * 
117         * @param array the array to insert objects to
118         * @param objectsToInsert the objects to insert
119         * @param indexToInsertAt the index in array to insert the objects at
120         */
121        public static void insertObjectsAtIndex(NSMutableArray array, NSArray objectsToInsert, int indexToInsertAt) 
122        {
123            /** require
124            [valid_array] array != null;  
125            [valid_range] objectsToInsert != null;
126            [valid_index] indexToInsertAt > -1 && indexToInsertAt <= array.count(); **/
127    
128            Enumeration reverseEnumerator = objectsToInsert.reverseObjectEnumerator();
129            while (reverseEnumerator.hasMoreElements())
130            {
131                Object anObject = reverseEnumerator.nextElement();
132                array.insertObjectAtIndex(anObject, indexToInsertAt);
133            }
134        }

138        /**
139         * Moves the objects in the given range of the given mutable array to the given
140         * index.
141         * 
142         * @param array the array to move objects in
143         * @param range the range of objects to move
144         * @param moveToIndex the index in array to move the objects to
145         */
146        public static void moveObjectsInRangeToIndex(NSMutableArray array, NSRange range, int moveToIndex) 
147        {
148            /** require
149            [valid_array] array != null;  
150            [valid_range] range != null && ! range.containsLocation(moveToIndex);
151            [valid_index] moveToIndex > -1 && moveToIndex <= array.count(); **/
152    
153            // if the moveTo index is after the range, subtract the length of the range
154            // since we will be removing those before we insert
155            if (moveToIndex > range.location() + range.length())
156            {
157                moveToIndex -= range.length();
158            }
159            NSArray objectsToMove = array.subarrayWithRange(range);
160            array.removeObjectsInRange(range);
161            insertObjectsAtIndex(array, objectsToMove, moveToIndex);
162        }
