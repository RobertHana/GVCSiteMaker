001    package net.global_village.foundation;
002    
003    import java.text.*;
004    
005    
006    /**
007     * A simple formatter for java.lang.Boolean that formats into two string values and parses two
008     * string values case-insensitively.
009     *
010     * @author Copyright (c) 2001-2005  Global Village Consulting, Inc.  All rights reserved.
011     * This software is published under the terms of the Educational Community License (ECL) version 1.0,
012     * a copy of which has been included with this distribution in the LICENSE.TXT file.
013     */
014    public class BooleanFormatter extends Format
015    {
016        protected String trueString;
017        protected String falseString;
018        protected String lowerTrueString;
019        protected String lowerFalseString;
020    
021    
022    
023        /**
024         * Designated constructor.
025         *  
026         * @param trueValue the true value for parsing and formatting
027         * @param falseValue the false value for parsing and formatting
028         */
029        public BooleanFormatter(String trueValue, String falseValue)
030        {
031            super();
032            /** require [valid_true] trueValue != null;  [valid_false] falseValue != null;   **/
033            trueString = trueValue;
034            falseString = falseValue;
035            lowerTrueString = trueValue.toLowerCase();
036            lowerFalseString = falseValue.toLowerCase();    
037        }
038        
039        
040        /**
041         * Covenience constructor creates formatter for 'true' and 'false'.
042         */
043        public BooleanFormatter()
044        {
045            this("true", "false");
046        }
047    
048    
049    
050    
051        /* (non-Javadoc)
052         * @see java.text.Format#format(java.lang.Object, java.lang.StringBuffer, java.text.FieldPosition)
053         */
054        public StringBuffer format(Object aBoolean, StringBuffer stringBuffer, FieldPosition fieldPosition)
055        {
056            stringBuffer.append(((java.lang.Boolean)aBoolean).booleanValue() ? trueString : falseString);
057            
058            return stringBuffer;
059        }
060    
061    
062    
063        /* (non-Javadoc)
064         * @see java.text.Format#parseObject(java.lang.String, java.text.ParsePosition)
065         */
066        public Object parseObject(String string, ParsePosition position)
067        {
068            Object parsedObject = null;
069            
070            String targetString = string.substring(position.getIndex()).toLowerCase();
071            
072            if (targetString.startsWith(lowerTrueString))
073            {
074                parsedObject = java.lang.Boolean.TRUE;
075                position.setIndex(position.getIndex() + trueString.length());
076            }
077            else if (targetString.startsWith(lowerFalseString))
078            {
079                parsedObject = java.lang.Boolean.FALSE;
080                position.setIndex(position.getIndex() + falseString.length());
081            }
082            
083            return parsedObject;
084        }
085    
086    
087    
088        /** invariant
089         [valid_trueString] trueString != null; 
090         [valid_falseString] falseString != null; 
091         [valid_lowerTrueString] lowerTrueString != null; 
092         [valid_lowerFalseString] lowerFalseString != null; 
093         [true_string_matches] trueString.equalsIgnoreCase(lowerTrueString);
094         [false_string_matches] falseString.equalsIgnoreCase(lowerFalseString);
095         **/    
096    
097    }
