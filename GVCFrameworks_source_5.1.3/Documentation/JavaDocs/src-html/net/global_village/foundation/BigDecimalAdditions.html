028        /**
029         * Checks if the given <code>BigDecimal</code> is between the two values.
030         *
031         * @param value the value being compared
032         * @param lower the lower value
033         * @param upper the upper value
034         * @param lowerInclusive should the lower comparison be inclusive or exclusive?
035         * @param upperInclusive should the upper comparison be inclusive or exclusive?
036         * @return <code>true</code> if value is between (inclusively or exclusively, as determined by the two inclusive params) <code>upper</code> and <code>lower</code>, <code>false</code> otherwise
037         */
038        public static boolean between(BigDecimal value, BigDecimal lower, BigDecimal upper, boolean lowerInclusive, boolean upperInclusive)
039        {
040            /** require
041            [valid_value_param] value != null;
042            [valid_lower_param] lower != null;
043            [valid_upper_param] upper != null; **/
044    
045            int lowerComparisonValue = lowerInclusive ? 0 : 1;
046            int upperComparisonValue = upperInclusive ? 0 : -1;
047            return (value.compareTo(lower) >= lowerComparisonValue) && (value.compareTo(upper) <= upperComparisonValue);
048        }

073        /**
074         * Compares two BigDecimals for equality within (1 / 10^<minimum scale>) tolerance.  First changes the scale of the most precise number by lowering it to the scale of the least precise, rounding as necessary.  Then subtracts the two numbers, takes the absolute value and multiply's that by 10^<minimum scale>.  If that number is less than or equal to 1, then the numbers are considered equal.  This is designed for use when comparing two values, one of which may have been rounded during calculation.  Note that this uses the ROUND_HALF_DOWN rounding mode to make the more precise param less precise.  This may affect the result you expect, so be careful.<p>
075         * For example:<br>
076         * 2.6543  (scale = 4)   compared to<br>
077         * 2.6     (scale = 1)   will result in true (as will 2.7 and 2.8)<br>
078         * 2.6543  (scale = 4)   compared to<br>
079         * 2.9     (scale = 1)   will result in false (as will 2.5)<br>
080         *
081         * @param op1 one of the BigDecimals to compare
082         * @param op2 one of the BigDecimals to compare
083         * @return <code>true</code> if the two <code>BigDecimals</code> are equal to within (1 / 10^<minimum scale>), after setting both scales to the least precise scale, <code>false</code> otherwise
084         */
085        public static boolean equalsWithScale(BigDecimal op1, BigDecimal op2)
086        {
087            /** require [valid_op1_param] op1 != null; [valid_op2_param] op2 != null; **/
088    
089            int op1Scale = op1.scale();
090            int op2Scale = op2.scale();
091            if (op1Scale > op2Scale)
092            {
093                op1 = op1.setScale(op2Scale, BigDecimal.ROUND_HALF_DOWN);
094            }
095            else if (op2Scale > op1Scale)
096            {
097                op2 = op2.setScale(op1Scale, BigDecimal.ROUND_HALF_DOWN);
098            }
099    
100            BigDecimal tester = op1.subtract(op2).abs();
101            tester = tester.movePointRight(op1.scale());
102    
103            if (tester.compareTo(new BigDecimal("1")) <= 0)
104                return true;
105            else
106                return false;
107        }
