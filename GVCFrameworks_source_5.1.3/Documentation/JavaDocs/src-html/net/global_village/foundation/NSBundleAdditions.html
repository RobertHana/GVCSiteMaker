032        /**
033         * Returns any bundle except for the given bundle.  This is used by the unit tests to return a bundle in which the test resources should not be found.
034         *
035         * @param exceptBundle the bundle that we shouldn't return
036         * @return any bundle but the given bundle
037         */
038        public static NSBundle anyBundle(NSBundle exceptBundle)
039        {
040            /** require [valid_param] exceptBundle != null; **/
041    
042            Enumeration bundles = NSBundle.frameworkBundles().objectEnumerator();
043    
044            while (bundles.hasMoreElements())
045            {
046                NSBundle aBundle = (NSBundle)bundles.nextElement();
047                if ( ! aBundle.equals(exceptBundle))
048                {
049                    JassAdditions.post("NSBundleAdditions", "anyBundle", ! aBundle.equals(exceptBundle));
050                    return aBundle;
051                }
052            }
053    
054            throw new RuntimeException("anyBundle() could not return a valid result - possibly there is only one bundle!");
055    
056            /** ensure [valid_result] Result != null; **/
057        }

061        /**
062         * Returns the localized string table <code>tableName</code>, or <code>null</code> if the file could not be found.<p>
063         *
064         * This does not use <code>Collection.collectionWithContentsOfFile()</code> because the path returned by <code>resourcePathForLocalizedResourceNamed()</code> is not fully qualified (and there is no way to get the FQ name!).<p>
065         *
066         * Note: uses the <em>server's</em> locale for localization.  This is not appropriate for web apps.
067         *
068         * @param bundle the bundle in which to look for <code>tableName</code>.
069         * @param tableName the string table for which to search.
070         * @return a string table or <code>null</code>.
071         *
072         * @deprecated not functional as of WO 5.4, default resources now in Default.lproj for WOResourceManager and NSBundle can't find them
073         */
074        protected static NSDictionary findTableWithName(NSBundle bundle, String tableName)
075        {
076            /** require [valid_bundle_param] bundle != null; [valid_tableName_param] tableName != null; **/
077    
078            NSDictionary table = null;
079            String tablePath = bundle.resourcePathForLocalizedResourceNamed(tableName, null);
080    
081            if (tablePath != null)
082            {
083                byte[] tableBytes = bundle.bytesForResourcePath(tablePath);
084                String tableString = new String(tableBytes).trim();
085    
086                // If the file is a .strings file, propertyListFromString() doesn't recongnize it without the "{" and "}"
087                if ( ! (tableString.startsWith("{") || tableString.startsWith("<?xml")) )
088                {
089                    throw new Error("The table " + tableName + " must be wrapped in '{' and '}'");
090                }
091    
092                try
093                {
094                    table = (NSDictionary)NSPropertyListSerialization.propertyListFromString(tableString);
095                }
096                catch (IllegalArgumentException e)
097                {
098                    // If this happens, try adding {} around the table or removing them.  Usage has been inconsistent.
099                    throw new RuntimeException("String table " + tableName + " in bundle " + bundle.name() + " has a syntax error: " + e.getMessage());
100                }
101                // Since we are only using this to get .strings files and the CustomInfo.plist, this will always return a dictionary.
102            }
103    
104            return table;
105        }

152        /**
153         * Returns a localized version of <code>key</code> in table <code>&lt;tableName&gt;.strings</code>.<p>
154         *
155         * Note: uses the <em>server's</em> locale for localization.  This is not appropriate for web apps.
156         *
157         * @param bundle the bundle in which to look for the table.
158         * @param key the string to localize.
159         * @param tableName the string table to search.
160         * @return a localized version of <code>key</code> or <code>null</code> if <code>key</code> could not be found.
161         *
162         * @deprecated not functional as of WO 5.4, default resources now in Default.lproj for WOResourceManager and NSBundle can't find them
163         */
164        protected static String findLocalizedStringForKey(NSBundle bundle, String key, String tableName)
165        {
166            /** require
167            [valid_bundle_param] bundle != null;
168            [valid_key_param] key != null;
169            [valid_tableName_param] tableName != null; **/
170    
171            String mungedTableName = tableName + StringsFileExtension;
172              String localizedString = null;
173            if (tableExistsWithName(bundle, mungedTableName))
174            {
175                NSDictionary table = tableWithName(bundle, mungedTableName);
176                localizedString = (String)table.objectForKey(key);
177            }
178            return localizedString;
179        }

233         *
234         * Returns a localized version of <code>key</code> in table <code>tableName</code>.<p>
235         *
236         * Note: uses the <em>server's</em> locale for localization.  This is not appropriate for web apps.
237         *
238         * @param bundle the bundle in which to look for the table.
239         * @param key the string to localize.
240         * @param defaultValue the value to return if the localized string for <code>key</code> can't be found in the table.
241         * @param tableName the string table to search.
242         * @return a localized version of <code>key</code>. If <code>key</code> could not be localized, returns <code>defaultValue</code>.
243         *
244         * @deprecated not functional as of WO 5.4, default resources now in Default.lproj for WOResourceManager and NSBundle can't find them
245         */
246        public static String localizedStringForKey(NSBundle bundle, String key, String defaultValue, String tableName)
247        {
248            /** require
249            [valid_bundle_param] bundle != null;
250            [valid_key_param] key != null;
251            [valid_defaultValue_param] defaultValue != null;
252            [valid_tableName_param] tableName != null; **/
253    
254            String rv;
255    
256            if (localizedStringExistsForKey(bundle, key, tableName))
257            {
258                rv = localizedStringForKey(bundle, key, tableName);
259            }
260            else
261            {
262                rv = defaultValue;
263            }
264    
265            return rv;

365        /**
366         * Replacement for the infoDictionary() method, which is now deprecated.  Looks for a file called <bundlename>.plist and returns the dictionary created from that file.
367         *
368         * @param bundle the bundle in which to look for the file.
369         * @return the dictionary created from a file.  Returns an empty dictionary if the file does not exist.
370         */
371        public static NSDictionary infoDictionary(NSBundle bundle)
372        {
373            /** require [valid_param] bundle != null; **/
374    
375            NSDictionary theDictionary;
376    
377            // For compatability first try <bundle name>.plist
378            if (NSBundleAdditions.tableExistsWithName(bundle, bundle.name() + ".plist"))
379            {
380                theDictionary = NSBundleAdditions.tableWithName(bundle, bundle.name() + ".plist");
381            }
382            // If that is not there try Info.plist which is more properly the WO5 successor to the info dictionary
383            else if (NSBundleAdditions.tableExistsWithName(bundle, "CustomInfo.plist"))
384            {
385                theDictionary = NSBundleAdditions.tableWithName(bundle, "CustomInfo.plist");
386            }
387            else
388            {
389                theDictionary = new NSDictionary();
390            }
391    
392            return theDictionary;
393    
394            /** ensure [valid_result] Result != null; **/
395        }

399        /**
400         * Given a path to a framework, this converts it to a path to a java zip file in that framework (a java zip file may or may not exist in the framework - that makes no difference to this method).
401         *
402         * @param frameworkPath the path that this will convert to point to the java zip
403         * @return path to the Java classes that may (or may not) exist in the given framework
404         */
405        protected static String frameworkPathToJavaZipPath(String frameworkPath)
406        {
407            /** require [valid_param] frameworkPath != null; **/
408    
409            String stringTail = NSPathUtilities.lastPathComponent(frameworkPath);
410            stringTail = NSPathUtilities.stringByDeletingPathExtension(stringTail);
411            stringTail = stringTail.toLowerCase();
412            stringTail = NSPathUtilities.stringByAppendingPathExtension(stringTail, "zip");
413    
414            String javaZipPath = NSPathUtilities.stringByAppendingPathComponent(frameworkPath, "Resources");
415            javaZipPath = NSPathUtilities.stringByAppendingPathComponent(javaZipPath, "Java");
416            javaZipPath = NSPathUtilities.stringByAppendingPathComponent(javaZipPath, stringTail);
417    
418            return javaZipPath;
419    
420            /** ensure [valid_result] Result != null; **/
421        }

425        /**
426         * Fix a problem with <code>bundleForClass</code> which causes a call to a class in the main bundle to return null.  We catch that and return <code>mainBundle()</code> if so.
427         *
428         * @param aClass the class for which this returns the bundle
429         * @return the bundle for the given class
430         * @see com.webobjects.foundation.NSBundle#bundleForClass(Class)
431         */
432        public static NSBundle bundleForClass(java.lang.Class aClass)
433        {
434            /** require [valid_param] aClass != null; **/
435    
436            if (NSBundle.bundleForClass(aClass) == null)
437            {
438                return NSBundle.mainBundle();
439            }
440            else
441            {
442                return NSBundle.bundleForClass(aClass);
443            }
444    
445            /** ensure [valid_result] Result != null; **/
446        }

450        /**
451         * Returns a string containing the contents of the resource at resourcePath.  If resourcePath does not
452         * indicate a resource, or is null, the empty string is returned.
453         *
454         * @param bundle NSBundle to get resource from
455         * @param resourcePath path to the resource relative to the top level resources directory
456         * @return a string containing the contents of the resource at resourcePath
457         */
458        public static String contentsOfResourceAtPath(NSBundle bundle, String resourcePath) {
459            /** require [valid_bundle] bundle != null;   **/
460            try {
461                return new String(bundle.bytesForResourcePath(resourcePath), _NSStringUtilities.UTF8_ENCODING);
462            }
463            catch (UnsupportedEncodingException e) {
464                throw new ExceptionConverter(e);
465            }
466            /** ensure [valid_result] Result != null; **/
467        }
