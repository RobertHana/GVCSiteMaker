042        /**
043         * Test urlEncodedDictionary()
044         */
045        public void testUrlEncodedDictionary()
046        {
047            NSDictionary dict = new NSDictionary("value", "key");
048            assertEquals("key=value", NSDictionaryAdditions.urlEncodedDictionary(dict));
049    
050            dict = new NSDictionary(new String[] {"value1", "value2"}, new String[] {"key1", "key2"});
051            assertTrue(NSDictionaryAdditions.urlEncodedDictionary(dict).equals("key1=value1&key2=value2") || NSDictionaryAdditions.urlEncodedDictionary(dict).equals("key2=value2&key1=value1"));
052    
053            // Test URL encoding
054            dict = new NSDictionary("value!with.strange(chars)", "key with spaces");
055            assertEquals("key+with+spaces=value%21with.strange%28chars%29", NSDictionaryAdditions.urlEncodedDictionary(dict));
056    
057            // Test dictionary handling
058            dict = new NSDictionary(new NSArray("value"), "key");
059            assertEquals("key=value", NSDictionaryAdditions.urlEncodedDictionary(dict));
060    
061            dict = new NSDictionary(new NSArray(new String[] { "value", "other value"}), "key");
062            assertEquals("key=value", NSDictionaryAdditions.urlEncodedDictionary(dict));
063    
064        }
