001    package net.global_village.woextensions.components;
002    
003    import com.webobjects.appserver.*;
004    
005    
006    /**
007     * This component when used in conjunction with an HTML table and a
008     * <code>WORepetition</code> will give the desired alternating background colors on rows.
009     * Use this <code>WOComponent</code> instead of the standard <code>TR</code> tag.  Pass in the index of
010     * the <code>WORepetition</code> that envelopes this <code>WOComponent</code>.
011     *
012     * @author Copyright (c) 2001-2005  Global Village Consulting, Inc.  All rights reserved.
013     * This software is published under the terms of the Educational Community License (ECL) version 1.0,
014     * a copy of which has been included with this distribution in the LICENSE.TXT file.
015     * @version $Revision: 2$
016     */
017    public class AlternatingColorRowComponent extends WOComponent
018    {
019        public Integer rowNumber;
020        protected String valign;
021    
022    
023        /**
024         * Designated constructor.
025         */
026        public AlternatingColorRowComponent(WOContext aContext)
027        {
028            super(aContext);
029        }
030    
031    
032    
033        /**
034         * Returns either "<code>primaryBackgroundColor</code>" or "<code>secondaryBackgroundColor</code>"
035         * depending on <code>rowNumber</code>.
036         *
037         * @return background color CSS class
038         */
039        public String backgroundCSSClass()
040        {
041            //rowNumber = (Integer) valueForBinding("rowNumber");
042            String backgroundCSSClass;
043    
044            if (rowNumber.intValue() % 2 == 0)
045            {
046                backgroundCSSClass = "primaryBackgroundColor";
047            }
048            else
049            {
050                backgroundCSSClass = "secondaryBackgroundColor";
051            }
052             
053            return backgroundCSSClass;
054    
055            /** ensure
056            [result_not_null] Result != null;
057            [correct_result] Result == "primaryBackgroundColor" || Result == "secondaryBackgroundColor"; **/
058        }
059    
060    
061    
062        public String valign()
063        {
064            return valign;
065        }
066    
067    
068    
069        /**
070         * Only valid HTML tags are accepted.
071         *
072         * @param newValign one of "<code>top</code>", "<code>middle</code>" or "<code>bottom</code>"
073         */
074        public void setValign(String newValign)
075        {
076            /** require [valid_param] (newValign.equalsIgnoreCase("top") || newValign.equalsIgnoreCase("middle") || newValign.equalsIgnoreCase("bottom")); **/
077            valign = newValign;
078        }
079    
080    
081    
082    }
