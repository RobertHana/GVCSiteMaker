004    import com.webobjects.appserver.*;
005    
006    /**
007    * If isPermissable is true displays a checkbox controlled by isChecked, if isPermissable is not true displays a string value for isChecked.  If isChecked is true string is "Yes", "No" otherwise.  It must be placed inside a WOForm.
008     *

010     * This software is published under the terms of the Educational Community License (ECL) version 1.0,
011     * a copy of which has been included with this distribution in the LICENSE.TXT file.
012     * @version $Revision: 4$
013     */  
014    public class SecureCheckBoxComponent extends SecureComponent
015    {
016    
017        /**
018         * Designated constructor.
019         */
020        public SecureCheckBoxComponent(WOContext aContext)
021        {
022            super(aContext);
023        }
024    
025    
026        
027        /**
028         * Accessor for isChecked binding.  The binding provides the displayed content of the WOComponent
029         *
030         * @return the value bound to isChecked
031         */
032        public boolean isChecked()
033        {
034            return  booleanValueForBinding("isChecked", false);
035        }
036    
037    
038    
039        /**
040         * Returns a string value for isChecked() Yes if isChecked is true, No otherwise
041         *
042         * @return a string value for isChecked() Yes if isChecked is true, No otherwise
043         */
044        public String stringValue()
045        {
046            String stringValue = "No";
047    
048            if (isChecked())
049            {
050                stringValue = "Yes";
051            }
052            
053            return stringValue;
054        }
055    
056    }
