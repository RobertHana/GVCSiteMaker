016    public class TestKeyedLookup extends KeyedLookup
017    {
018        // now for wrapped stuff from superclasses
019    
020        
021        /**
022         * A convenience method which returns the object of this entity where <code>key == aKey</code>.  Throws an exception if there are no matching objects, or more than one matching object.
023         *
024         * @param aKey key of specific object to look for.
025         * @param editingContext the <code>EOEditingContext</code> to fetch from
026         * @return object associated with <code>aKey</code>
027         */
028        static public TestKeyedLookup objectForKey( String aKey, EOEditingContext editingContext)
029        {
030            /** require
031            [aKey_not_null] aKey != null;
032            [editingContext_not_null] editingContext != null;  **/
033    
034            return (TestKeyedLookup) KeyedLookup.objectForKey( editingContext, "TestKeyedLookup", aKey);
035    
036            /** ensure [Result_not_null] Result != null; **/
037        }
038    
039    
040        
041        /**
042         * A convenience method which returns all objects of this Entity sorted by <code>NameComparator</code>.  
043         *
044         * @param editingContext the <code>EOEditingContext</code> to fetch from
045         * @return NSArray of all objects of the entity specified by <code>entityName</code> and sorted by <code>NameComparator</code>
046         */
047        static public NSArray orderedObjects(EOEditingContext editingContext)
048        {
049            /** require [editingContext_not_null] editingContext != null;  **/
050            
051            return KeyedLookup.orderedObjects( editingContext, "TestKeyedLookup", Lookup.NameComparator );
052    
053            /** ensure [Result_not_null] Result != null; **/
054        }
055    
056    }
