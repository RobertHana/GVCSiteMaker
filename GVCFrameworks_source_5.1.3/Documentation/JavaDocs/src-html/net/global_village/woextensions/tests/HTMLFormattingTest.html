014    public class HTMLFormattingTest extends TestCase
015    {
016    
017        /**
018         * Constructor for HTMLFormattingTest.
019         * @param name name of test
020         */
021        public HTMLFormattingTest(String name)
022        {
023            super(name);
024        }
025    
026    
027        /**
028         * Basic test of string replacement logic.
029         */
030        final public void testReplaceCarriageReturnsWithHTMLInString()
031        {
032            String result = HTMLFormatting.replaceFormattingWithHTML("This is a\n\ntest of\nthe code");
033            assertTrue("newlines converted", result.equals("This is a<br/><br/>test of<br/>the code"));
034            
035            result = HTMLFormatting.replaceFormattingWithHTML("This is a\r\rtest of\rthe code");
036            assertTrue("CR converted", result.equals("This is a<br/><br/>test of<br/>the code"));
037    
038            result = HTMLFormatting.replaceFormattingWithHTML("This is a\n\rtest of\rthe code");
039            assertTrue("newline-CR converted", result.equals("This is a<br/>test of<br/>the code"));
040    
041            result = HTMLFormatting.replaceFormattingWithHTML("This is a\r\ntest of\rthe code");
042            assertTrue("CR-newline converted", result.equals("This is a<br/>test of<br/>the code"));
043    
044            result = HTMLFormatting.replaceFormattingWithHTML(null);
045            assertTrue("null handling", result.equals(""));
046    
047        }
048    
049    }
