001    package net.global_village.virtualtables;

003    
004    
005    /**
006     * Holds the data for a virtual table.  Each field in a table will be associated with a specific column, which defines the type of data that the field must contain.  Each different data type is a separate subclass of this.<p>
007     *
008     * Creating new VirtualField subclasses<br>
009     * Creating a new VirtualFIeld subclass is usually simple (just do it!), there are a few cases where you may need to do other things:
010     * <ul>
011     *   <li>If the "value" of the field is <em>not</em> an attribute of the EO (as is the case with VirtualLookupColumns (q.v.)), you'll also need to create a VirtualColumn subclass and override importAttributeName() and valueForImportedValue()
012     * </ul>
013     *
014     * @author Copyright (c) 2001-2005  Global Village Consulting, Inc.  All rights reserved.
015     * This software is published under the terms of the Educational Community License (ECL) version 1.0,
016     * a copy of which has been included with this distribution in the LICENSE.TXT file.
017     * @version $Revision: 4$
018     */
019    public class VirtualField extends _VirtualField
020    {
021    
022    
023        /**
024         * Returns the field's column name.
025         *
026         * @return this field's column name
027         */
028        public String name()
029        {
030            return column().name();
031            /** ensure [valid_result] Result != null; **/
032        }
033    
034    
035    
036        /**
037         * The generic accessor for fields.  Subclasses should override this to return their data.
038         *
039         * @return this field's value
040         */
041        public Object value()
042        {
043            throw new Error("Subclass responsibility");
044        }
045    
046    
047    
048        /**
049         * The generic setter for fields.  Subclasses should override this to set their data.
050         *
051         * @param newValue the new value of the field
052         */
053        public void setValue(Object newValue)
054        {
055            throw new Error("Subclass responsibility");
056        }
057    
058    
059    
060        /**
061         * Returns a copy of this field's value.  This is for use when duplicating and copying rows.
062         *
063         * @return a copy of this field's value
064         */
065        public Object valueCopy()
066        {
067            // In most cases the value is immutable so we can just return it.
068            return value();
069        }
070    
071    
072    
073        /**
074         * Returns <code>true</code> if any fields in another table refers to this field.  This is the case if a lookup field in a row of another table uses this field as the source of its lookup data.  If this returns <code>true</code>, then this field cannot be deleted (prevented by Deny deletion rule in VirutalField).
075         *
076         * @return <code>true</code> if any fields in another table refers to this field.
077         */
078        public boolean hasReferringFields()
079        {
080            return referringFields().count() > 0;
081        }
082    
083    
084    
085        /**
086         * Used by the table export process to perform additional processing on an exported field value.
087         *
088         * @param sender an object that this method can use to further distinguish what should be done with the value
089         * @return the value to use during table export
090         */
091        public Object valueForExport(Object sender)
092        {
093            return value();
094        }
095    
096    
097    }
