015    /**
016     *  EOEditingContext overrides the saveChanges method of com.apple.yellow.eocontrol.EOEditingContext so that custom, localized validation messages can be produced for adaptor level exceptions.
017     * <br>
018     *  TODO:
019     *  - consider if this functionality could be better (or more consistently) handled by subclassing one of the EOAdaptor... or EODatabase... classes.  The problem with this EOEditingContext approach is that not all adpators raise the proper exceptions, and that the exceptions raised may not have the needed information in them.  Subclassing at a lower level may get around this.
020     * <br>
021     *    Notes:
022     *      (1) Violations of width / scale / precision limits (assuming EOModel does not match db)
023     *          generate a NSInternalInconsistencyException which is not caught by this poser.  Observed with FrontBase.
024     * <br>
025     *      (2) An object having a FK reference that does not exist causes EOF to attempt to fetch that object which results
026     *          in an EOAdaptorOptimisticLockingFailure or an NSInternalInconsistencyException.  Observed with FrontBase
027     * <br>
028     *      (3) Oracle returns the proper type of exception, but with an ORA error code rather than a more standard and
029     *          useful SQLSTATE code.  This will make interpreting the exceptions more difficult.
030     * <br>
031     *      (4) FrontBase raises an NSInternalInconsistencyException instead of an EOGeneralAdaptorException for everything
032     *          other than optimistic locking failures.  These exceptions contain very little in the way of useful information,
033     *          other than the name and reason.  The userInfo consists of a list of EODatabaseOperation and EODatabaseContext
034     *          which give no indication of what has failed.  The exception reason contains the actual exception as SQLSTATE
035     *          values.  For example a reason will look like:
036     *              Exception <FrontBase exception #> (SQLSTATE class:SQLSTATE code). <Text message>
037     *          For example:
038     *              Exception 357 (23:000). Integrity constraint violation...
039     *          The (23:000) indicates that is it an integrity constraint violation and could be used to manufacture a more
040     *          explicit exception.  FrontBase has promised to address this issue in a future release.
041     *
042     * @author Copyright (c) 2001-2005  Global Village Consulting, Inc.  All rights reserved.

044     * a copy of which has been included with this distribution in the LICENSE.TXT file.
045     * @version $Revision: 23$

047     * TODO: Add a delegate to interpret exceptions?

115        /**
116        * Calls super.saveChanges() and interprets any exceptions raised.<br><br>
117        * This EOEditingContext also clears the undo stack after a successful save.  This is done as a work around for a bug in EOF. There is a rather serious bug when validateForDelete() fails to allow a deletion.  This will occur if you are using the Deny delete rule, the relationship is mandatory, or you have a custom validateForDelete() method.  The error occurs in this scenario:<br>
118        * 1. The editing context has multiple generations, meaning that saveChanges() has been called one or more times after one or more EOs has been created / inserted / updated.<br>
119        * 2. An EO is deleted from the editing context by deleteObject().<br>
120        * 3. saveChanges() fails due to an NSValidation.ValidationException raised in validateForDelete.<br>
121        *
122        * This result of this appears to be that undo() is called on the editing context's undo manager too many times.  Instead of rolling back to the state when saveChanges() was called it rolls back past several of previous saveChanges()!  The result of this is that the editing context shows a historical state that does not match the object store or the database.
123         */
124        public void saveChanges()
125        {
126            try
127            {
128                super.saveChanges();
129            }
130            catch (EOFValidationException e)
131            {
132                if (logger.isTraceEnabled())
133                {
134                    logger.trace("Save failed with EOFValidationException: ", e);
135                    logger.trace("    userInfo: " + e.userInfo());
136                }
137                throw e;
138            }
139            catch (EOGeneralAdaptorException e)
140            {
141                if (logger.isTraceEnabled())
142                {
143                    logger.trace("Save failed with EOGeneralAdaptorException: ", e);
144                    logger.trace("    userInfo: " + e.userInfo());
145                }
146                RuntimeException interpretedException =  interpretEOGeneralAdaptorException(e);
147    
148                if (logger.isTraceEnabled())
149                {
150                    logger.trace("    interpreted to : " + interpretedException.getLocalizedMessage());
151                }
152    
153                throw interpretedException;
154            }
155        }

159        /**
160         * Examines the exception and dispatches it to one of the more specific interpret...Exception methods.
161         *
162         * @param theException the exception as raised from saveChanges()
163         * @return an exception with a localized error message and a customized userInfo dictionary
164         */
165        public RuntimeException interpretEOGeneralAdaptorException(EOGeneralAdaptorException theException)
166        {
167            /** require [valid_param] theException != null; **/
168    
169            // In case we can not interpret this exception, ensure that it is returned as is.
170            RuntimeException interpretedException = theException;
171    
172            if (NSExceptionAdditions.isOptimisticLockingFailure(theException))
173            {
174                interpretedException = interpretOptimisticLockingFailure(theException);
175            }
176            else if (NSExceptionAdditions.isIntegrityConstraintViolation(theException))
177            {
178                interpretedException = interpretIntegrityConstraintViolation(theException);
179            }
180            else if (NSExceptionAdditions.isAdaptorOperationFailureException(theException))
181            {
182                int failedOperator = NSExceptionAdditions.failedAdaptorOperator(theException);
183    
184                switch (failedOperator)
185                {
186                    // case EODatabaseOperation.EOAdaptorLockOperator : interpretedException = interpretLockingFailureFromException(theException); break;
187    
188                    // This should never happen with integrity constraint violations handled above and barring model exception
189                    case EODatabaseOperation.AdaptorInsertOperator :
190                        interpretedException = interpretInsertFailureFromException(theException);
191                        break;
192    
193                    // case EODatabaseOperation.EOAdaptorUpdateOperator : interpretedException = interpretUpdateFailureFromException(theException); break;
194    
195                    // case  EODatabaseOperation.EOAdaptorDeleteOperator : interpretedException = interpretDeleteFailureFromException(theException); break;
196    
197                    // case  EODatabaseOperation.EOAdaptorStoredProcedureOperator : interpretedException = interpretStoredProcedureFailureFromException(theException); break;
198                    default:
199                        // Later, once the other cases are implemented we can do this:
200                        //                default :
201                        //                    interpretedException = new RuntimeException("Unknown Adaptor Operator (" + failedOperator +")");
202                }
203            }
204    
205            if (interpretedException instanceof EOFValidationException)
206            {
207                ((EOFValidationException)interpretedException).setOriginalException(theException);
208            }
209    
210            return interpretedException;
211        }

215        /**
216         * Interprets the passed exception as an optimistic locking failure (concurrent update error) and returns an exception with a localized error message and a customized userInfo dictionary.  Note that the operation which caused the exception could have been an update or a delete.  The same message is returned in either case.  If you need to customize the message, look at <code>NSExceptionAdditions.failedAdaptorOperation(exception)</code> to determine if it is an EOAdaptorUpdateOperator or an EOAdaptorDeleteOperator.
217         *
218         * @param theException the exception as raised from saveChanges()
219         * @return an exception with a localized error message and a customized userInfo dictionary
220         */
221        public EOFValidationException interpretOptimisticLockingFailure(EOGeneralAdaptorException theException)
222        {
223            /** require
224            [valid_param] theException != null;
225            [exception_is_optimistic_locking_failure] NSExceptionAdditions.isOptimisticLockingFailure(theException); **/
226    
227            EOEntity entity = NSExceptionAdditions.entitySaveFailedOn(theException, this);
228            EOEnterpriseObject object = NSExceptionAdditions.objectSaveFailedOn(theException);
229            return new EOFValidationException(object, entity.name(), null, EOFValidation.OptimisticLockingFailure, null);
230        }

235         * Interprets the passed exception as an optimistic locking failure (concurrent update error) and returns an exception with a localized error message and a customized userInfo dictionary.  Note that the operation which caused the exception could have been an update or a delete.  The same message is returned in either case.  If you need to customize the message, look at <code>NSExceptionAdditions.failedAdaptorOperation(exception)</code> to determine if it is an EOAdaptorUpdateOperator or an EOAdaptorDeleteOperator.
236         *
237         * @param theException the exception as raised from saveChanges()
238         * @return an exception with a localized error message and a customized userInfo dictionary
239         */
240        public EOFValidationException interpretIntegrityConstraintViolation(EOGeneralAdaptorException theException)
241        {
242            /** require
243            [valid_param] theException != null;
244            [exception_is_optimistic_locking_failure] NSExceptionAdditions.isIntegrityConstraintViolation(theException); **/
245    
246            EOEntity entity = NSExceptionAdditions.entitySaveFailedOn(theException, this);
247            EOEnterpriseObject object = NSExceptionAdditions.isAdaptorOperationFailureException(theException) ?
248                                            NSExceptionAdditions.objectSaveFailedOn(theException) : null;
249            String violatedIntegrityConstraintName = NSExceptionAdditions.violatedIntegrityConstraintName(theException);
250            return new EOFValidationException(object, entity.name(), violatedIntegrityConstraintName, EOFValidation.IntegrityConstraintViolation, NSExceptionAdditions.violatedIntegrityConstraintName(theException));
251        }

341        {
342            try
343            {
344                // If private_processingChanges then a notification was received after calling super's constructor and
345                // before our constructor set this.  There are no registered objects yet, so we can ignore this
346                if (private_processingChanges != null && private_processingChanges.getBoolean(this))
347                {
348                    NSDictionary changedObjects = _objectBasedChangeInfoForGIDInfo(changes);
349                    if (changedObjects.count() > 0)
350                    {
351                        // Reset this flag so this exception does not get raised again and again until
352                        // this editing context is garbage collected.
353                        private_processingChanges.setBoolean(this, false);
354                        throw new RuntimeException("While processingRecentChanges(), objects refetched " +
355                            "and _processObjectStoreChanges called for: " + changedObjects);
356                    }
357                }
358            }
359            catch (IllegalAccessException access)
360            {
361                throw new ExceptionConverter(access);
362            }
363    
364            super._processObjectStoreChanges(changes);
365        }
