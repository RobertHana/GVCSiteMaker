001    package com.houdah.webobjects.eoaccess.coercion;
002    
003    import com.houdah.webobjects.eocontrol.qualifiers.InSubqueryQualifier;
004    import com.webobjects.eoaccess.EOEntity;
005    import com.webobjects.eocontrol.EOQualifier;
006    import com.webobjects.foundation.NSDictionary;
007    
008    /**
009     * <p>
010     * Support class to apply qualifier attribute coercion to instances of
011     * InSubqueryQualifier.
012     * </p>
013     * 
014     * <p>
015     * This sample code is provided for educational purposes. It is mainly to be
016     * considered a source of information and a way to spark off discussion. I make
017     * no warranty, expressed or implied, about the quality of this code or its
018     * usefulness in any particular situation. Use this code or any code based on it
019     * at your own risk. Basically, enjoy the read, but don't blame me for anything.
020     * </p>
021     * 
022     * @author Pierre Bernard http://homepage.mac.com/i_love_my/webobjects.html
023     */
024    public class InSubqueryQualifierCoercionSupport implements QualifierAttributeCoercion.Support
025    {
026            /**
027             * Applies value coercion to values passed to the supplied qualifier.<BR>
028             * 
029             * @param qualifier
030             *            the qualifier to validate
031             * @param entity
032             *            the entity this qualifier is to be applied to
033             * @param the
034             *            updated qualifier - may share branches with the original
035             *            qualifier
036             */
037            public EOQualifier coerceQualifierAttributes(EOQualifier qualifier, EOEntity entity)
038            {
039                    InSubqueryQualifier isQualifier = (InSubqueryQualifier) qualifier;
040                    EOQualifier subQualifier = isQualifier.subQualifier();
041                    NSDictionary substitutions = new NSDictionary(QualifierAttributeCoercion
042                                    .coerceQualifierAttributes(subQualifier, entity), InSubqueryQualifier.SUB_QUALIFIER);
043    
044                    return isQualifier.clone(substitutions);
045            }
046    }
