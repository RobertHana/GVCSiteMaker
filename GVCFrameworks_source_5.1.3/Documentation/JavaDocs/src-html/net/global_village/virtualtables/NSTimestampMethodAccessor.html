001    package net.global_village.virtualtables;
002    
003    import java.util.*;
004    
005    import ognl.*;
006    
007    import com.webobjects.foundation.*;
008    
009    
010    /**
011     * Adds methods to NSTimestamp in the manner of a category.
012     *
013     * @author Copyright (c) 2006-2006 Harte-Hanks Shoppers, Inc.
014     */
015    public class NSTimestampMethodAccessor extends ObjectMethodAccessor
016    {
017    
018        /**
019         * Adds:
020         * <ul>
021         * <li>addYMDHMS(years, months, days, hours, minutes, seconds) as alias for timestampByAddingGregorianUnits</li>
022         * </ul>
023         *
024         * @see ognl.MethodAccessor#callMethod(java.util.Map, java.lang.Object, java.lang.String, java.lang.Object[])
025         */
026        public Object callMethod(Map map, Object target, String name, Object[] aobj) throws MethodFailedException
027        {
028            if ("addYMDHMS".equals(name))
029            {
030                return ((NSTimestamp)target).timestampByAddingGregorianUnits(((Number)aobj[0]).intValue(),
031                                                                             ((Number)aobj[1]).intValue(),
032                                                                             ((Number)aobj[2]).intValue(),
033                                                                             ((Number)aobj[3]).intValue(),
034                                                                             ((Number)aobj[4]).intValue(),
035                                                                             ((Number)aobj[5]).intValue());
036            }
037    
038            return super.callMethod(map, target, name, aobj);
039        }
040    
041    
042    
043    }
