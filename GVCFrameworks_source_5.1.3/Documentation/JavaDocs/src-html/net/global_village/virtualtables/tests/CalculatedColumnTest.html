035        /**
036         * Sets up the fixtures.
037         * @exception Exception an exception that the setup may throw
038         */
039        public void setUp() throws java.lang.Exception
040        {
041            super.setUp();
042    
043            // Create a table containing a value that can be looked up in an expression
044            lookupTable = VirtualTable.createVirtualTable(editingContext(), "Lookup_Table");
045            VirtualColumn lookupSourceColumn = (VirtualColumn)lookupTable.newColumn("lookupSourceColumn", ColumnType.NumberColumnType);
046            VirtualColumn lookupRelationalColumn = (VirtualColumn)lookupTable.newColumn("nameForLookupValue", ColumnType.StringColumnType);
047            VirtualColumn lookupExpressionColumn = (VirtualColumn)lookupTable.newColumn("expression", ColumnType.CalculatedColumnType);
048            ((VirtualCalculatedColumn)lookupExpressionColumn).setExpression("lookupSourceColumn * 10");
049            editingContext().saveChanges();
050    
051            // Create a value to be looked up
052            VirtualRow rowWithLookupValue = lookupTable.newRow();
053            rowWithLookupValue.takeValueForKey(new BigDecimal(7), lookupSourceColumn.name());
054            rowWithLookupValue.takeValueForKey("Big Number Seven!", lookupRelationalColumn.name());
055            VirtualField lookupField = rowWithLookupValue.fieldNamed(lookupSourceColumn.name());
056            editingContext().saveChanges();
057    
058    
059            // Create the table that will have calculated columns
060            testTable = VirtualTable.createVirtualTable(editingContext(), "Test Table");
061            testTable.newColumn("Some Number", ColumnType.NumberColumnType);
062            testTable.newColumn("Other Number", ColumnType.NumberColumnType);
063            testTable.newColumn("string Column", ColumnType.StringColumnType);
064            testTable.newColumn("boolean Column", ColumnType.BooleanColumnType);
065            testTable.newColumn("date", ColumnType.TimestampColumnType);
066    
067            // Here are the calculated columns that will be tested
068            expression1 = (VirtualCalculatedColumn)testTable.newColumn("expression", ColumnType.CalculatedColumnType);
069            expression1.setExpression("SomeNumber");
070            expression2 = (VirtualCalculatedColumn)testTable.newColumn("second calculation", ColumnType.CalculatedColumnType);
071            expression2.setExpression("expression + 27");
072            expression3 = (VirtualCalculatedColumn)testTable.newColumn("final calculation", ColumnType.CalculatedColumnType);
073            expression3.setExpression("lookupColumn + 17");
074    
075            // Create a lookup column into lookupTable
076            VirtualLookupColumn lookupColumn = (VirtualLookupColumn)testTable.newColumn("lookupColumn", ColumnType.LookupColumnType);
077            lookupColumn.setLookupColumn(lookupSourceColumn);
078    
079            // Add rows of data that will be used to text the expressions
080            row = testTable.newRow();
081            row.takeValueForKey("string value", "string Column");
082            row.takeValueForKey(Boolean.TRUE, "boolean Column");
083            row.takeValueForKey(new BigDecimal("100"), "Some Number");
084    
085            row2 = testTable.newRow();
086            row2.takeValueForKey("string value2", "stringColumn");
087            row2.takeValueForKey(Boolean.FALSE, "booleanColumn");
088            row2.takeValueForKey(new BigDecimal("101"), "Some Number");
089    
090            // Create a row using looking a value up in the lookup table
091            row3 = testTable.newRow();
092            row3.takeValueForKey("string value2", "stringColumn");
093            row3.takeValueForKey(Boolean.FALSE, "booleanColumn");
094            row3.takeValueForKey(new BigDecimal("101"), "Some Number");
095            row3.takeValueForKey(new NSTimestamp(), "date");
096            row3.takeValueForKey(lookupField, lookupColumn.name());
097    
098            editingContext().saveChanges();
099        }

237        /**
238         * Test that relational expressions are validated correctly.
239         */
240        public void testRelationalExtensionsAndCyclicExpression()
241        {
242            // There was a bug where a cycle was detected when the name of the calculated field appeared
243            // in a relational expression in its formula.  This happens when a field of that same name
244            // (expression in this case below) was in another (lookup) table and referred to in the expression
245            // This was fixed by using Column objects, not names, to track cycles.
246            try
247            {
248                expression1.setExpression("lookupColumn" + VirtualRowPropertyAccessor.RelationalNavigationMarker + "expression");
249                expression1.validateExpression(expression1.expression());
250                assertEquals(new BigDecimal(70), row3.valueForFieldNamed("expression"));
251            }
252            catch (EOFValidationException e)
253            {
254                NSLog.out.appendln(e);
255                fail("failed to allow relational expression (" + e.getLocalizedMessage() + ")");
256            }
257            finally
258            {
259                editingContext().revert();
260            }
261        }
