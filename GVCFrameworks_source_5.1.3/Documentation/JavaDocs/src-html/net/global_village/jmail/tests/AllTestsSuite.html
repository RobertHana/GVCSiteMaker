020        public static final String MAIL_SERVER_KEY = "MailServer";
021        public static final String POP_ACCOUNT_KEY = "POPAccount";
022        public static final String MAIL_USER_KEY = "MailUser";
023        public static final String MAIL_USER_PASSWORD_KEY = "MailUserPassword";
024        public static final String TEST_CONFIGURATION_FILE = "TestConfiguration.plist";
025        private NSDictionary testConfigurationDictionary;
026    
027        public static String TEST_SERVER;
028        public static String TEST_ACCOUNT;
029        public static String TEST_USER;
030        public static String TEST_PASSWORD;

040        public AllTestsSuite(String name)
041        {
042            super(name);
043            TEST_SERVER =   (String)testConfigurationDictionary().objectForKey(MAIL_SERVER_KEY);
044            TEST_ACCOUNT =  (String)testConfigurationDictionary().objectForKey(POP_ACCOUNT_KEY);
045            TEST_USER =     (String)testConfigurationDictionary().objectForKey(MAIL_USER_KEY);
046            TEST_PASSWORD = (String)testConfigurationDictionary().objectForKey(MAIL_USER_PASSWORD_KEY);
047    
048            addTest(new TestSuite(MessageTest.class));
049            addTest(new TestSuite(MailDeliveryTest.class));
050            addTest(new TestSuite(IMAPInBoxTest.class));

063        /**
064         * Returns the configuration dictionary for this test run.  This class gets the
065         * dictionary from the TestConfigurationDictionary.plist of the framework in
066         * the working directory when the tests are run.<br>
067         * <br>
068         * The dictionary is expected to have this format:
069         * <code>
070         * {
071         *     MailServer = "mail.someserver.com";
072         *     POPAccount = "you@someserver.com";
073         *     MailUser = "you";
074         *     MailUserPassword = "secret";
075         * }
076         * </code>
077         *
078         * @return the configuration dictionary for this test run
079         */
080        protected NSDictionary testConfigurationDictionary()
081        {
082            if (testConfigurationDictionary == null)
083            {
084                NSBundle bundle = NSBundle.mainBundle();
085                if (NSBundleAdditions.tableExistsWithName(bundle, TEST_CONFIGURATION_FILE))
086                {
087                    testConfigurationDictionary = NSBundleAdditions.tableWithName(bundle, TEST_CONFIGURATION_FILE);
088                }
089                else
090                {
091                    throw new IllegalStateException("Can't find configuration in " + TEST_CONFIGURATION_FILE);
092                }
093    
094                // Validate dictionary contents
095                if (testConfigurationDictionary.objectForKey(MAIL_SERVER_KEY) == null)
096                {
097                    throw new IllegalArgumentException(MAIL_SERVER_KEY +
098                        " key missing from testConfigurationDictionary()");
099                }
100    
101                if (testConfigurationDictionary.objectForKey(POP_ACCOUNT_KEY) == null)
102                {
103                    throw new IllegalArgumentException(POP_ACCOUNT_KEY +
104                        " key missing from testConfigurationDictionary()");
105                }
106    
107                if (testConfigurationDictionary.objectForKey(MAIL_USER_KEY) == null)
108                {
109                    throw new IllegalArgumentException(MAIL_USER_KEY +
110                        " key missing from testConfigurationDictionary()");
111                }
112    
113                if (testConfigurationDictionary.objectForKey(MAIL_USER_PASSWORD_KEY) == null)
114                {
115                    throw new IllegalArgumentException(MAIL_USER_PASSWORD_KEY +
116                        " key missing from testConfigurationDictionary()");
117                }
118            }
119    
120            return testConfigurationDictionary;
121    
122            /** ensure
123            [valid_result] Result != null;
124            [mail_server_key_is_present] Result.objectForKey(MAIL_SERVER_KEY) != null;
125            [pop_account_key_is_present] Result.objectForKey(POP_ACCOUNT_KEY) != null;
126            [mail_user_key_is_present] Result.objectForKey(MAIL_USER_KEY) != null;
127            [mail_user_password_key_is_present] Result.objectForKey(MAIL_USER_PASSWORD_KEY) != null; **/
128        }
