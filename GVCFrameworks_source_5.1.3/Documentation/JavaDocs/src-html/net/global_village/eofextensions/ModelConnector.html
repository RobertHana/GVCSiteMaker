010    /**
011     * <p>ModelConnector handles configuring the connection dictionary and adaptor name
012     * in EOModels at runtime.  It can ignore some models if they are not needed or if
013     * they need different connection information. Note that this class updates the
014     * connection dictionary <em>as the models are loaded</em>.  This means that if a model gets
015     * loaded before you instantiate this class, it's connection dictionary will not be
016     * updated.</p>
017     *
018     * <p>If an EOModel already has a set adaptor type (e.g. JDBC, JDNI) and that type
019     * is not the same as our <code>adaptorType()</code> then the model will also be
020     * ignored.  This prevents one connector (e.g. JDBC) from trashing models
021     * (e.g. JDNI) models.</p>
022     *
023     *
024     * @author Copyright (c) 2001-2006  Global Village Consulting, Inc.  All rights reserved.
025     * This software is published under the terms of the Educational Community License (ECL) version 1.0,
026     * a copy of which has been included with this distribution in the LICENSE.TXT file.
027     * @version $Revision: 16$
028     */

038        /**
039         * Designated constructor for the default model group, creates a ModelConnector to use
040         * the indicated connection dictionary and adaptor name when updating all models not named
041         * in <code>ingoredModels</code>.  Only models with an adaptorType of <code>theAdaptorName</code>
042         * or with no adaptorName, in the default model group are updated.
043         *
044         * @param aConnectionDictionary the connection dictionary to connect all the EOModels with
045         * @param theAdaptorName the name of the EOAdaptor that aConnectionDictionary is for
046         * @param theModelNamesToIgnore NSArray of the names of models to <b>not</b> update with connectionDictionary()
047         */
048        public ModelConnector(NSDictionary aConnectionDictionary,
049                String theAdaptorName,
050                NSArray theModelNamesToIgnore)
051        {
052            // Be careful not to call EOModelGroup.defaultGroup() or this.modelGroup() until the observer is registered
053            super();
054            /** require
055            [valid_aConnectionDictionary_param] aConnectionDictionary != null;
056            [valid_adaptorName] theAdaptorName != null;
057            [valid_theModelNamesToIgnore_param] theModelNamesToIgnore != null; **/
058    
059            connectionDictionary = aConnectionDictionary;
060            adaptorName = theAdaptorName;
061            modelNamesToIgnore = theModelNamesToIgnore;
062    
063            // This notification method does not check which group the model is in, thus avoiding infinite recursion
064            addObserver("handleModelAddedInDefaultGroupNotification");
065    
066            // Causes the EOModels to be loaded from the bundles referenced on the classpath.
067            EOModelGroup.defaultGroup().models();
068    
069            // EOF bug work around
070            if (modelGroup().models().count() > 0)
071            {
072                registerModels();
073            }
074    
075            // Now switch the notification method to the one that checks for model groups
076            removeObserver();
077            addObserver("handleModelAddedInCustomGroupNotification");
078        }

098        /**
099         * Designated constructor for custom model groups, creates a ModelConnector to use the
100         * indicated connection dictionary and adaptor name when updating all models not named in
101         * <code>ingoredModels</code>.  Only models with an adaptorType of <code>theAdaptorName</code>
102         * or with no adaptorName, and in the given model group, are updated.
103         *
104         * <p><b>Using a custom model group</b><br>
105         * Since this class sets the connection dictionary of models <em>as they are loaded</em>, you need
106         * to order your code carefully when using a custom model group:<br>
107         * <code>
108         * EOModelGroup myModelGroup = new EOModelGroup();
109         * new ModelConnector(dict, adaptor, ignore, myModelGroup);
110         * myModelGroup.addModel(myModel);
111         * </code></p>
112         *
113         * @param aConnectionDictionary the connection dictionary to connect all the EOModels with
114         * @param theAdaptorName the name of the EOAdaptor that aConnectionDictionary is for
115         * @param theModelNamesToIgnore NSArray of the names of models to <b>not</b> update with connectionDictionary()
116         * @param theModelGroup restricts connections to models in this model group, <code>null</code> indicates the default model group, do <b>not</b> pass in EOModelGroup.defaultGroup()
117         */
118        public ModelConnector(NSDictionary aConnectionDictionary,
119                String theAdaptorName,
120                NSArray theModelNamesToIgnore,
121                EOModelGroup theModelGroup)
122        {
123            super();
124            /** require
125            [valid_aConnectionDictionary_param] aConnectionDictionary != null;
126            [valid_adaptorName] theAdaptorName != null;
127            [valid_theModelNamesToIgnore_param] theModelNamesToIgnore != null;
128            [valid_theModelGroup_param] theModelGroup != null; **/
129    
130            connectionDictionary = aConnectionDictionary;
131            adaptorName = theAdaptorName;
132            modelNamesToIgnore = theModelNamesToIgnore;
133            modelGroup = theModelGroup;
134    
135            // Establish this instance as a handler of this notification so that the EOModels can be updated with the proper connection dictionary.
136            addObserver("handleModelAddedInCustomGroupNotification");
137    
138            // EOF bug work around
139            if (modelGroup().models().count() > 0)
140            {
141                registerModels();
142            }
143        }

167        /**
168         * Register for notifications.
169         *
170         * @param methodName the method name that will handle the notifications
171         */
172        protected void addObserver(String methodName)
173        {
174            NSSelector modelAddedSelector = new NSSelector(methodName, new Class[] { NSNotification.class } );
175            NSNotificationCenter.defaultCenter().addObserver(this, modelAddedSelector, EOModelGroup.ModelAddedNotification, null);
176        }

179        /**
180         * Unregister us from notifications so that we are not messaged.
181         */
182        protected void removeObserver()
183        {
184            if (NSLog.debugLoggingAllowedForLevel(NSLog.DebugLevelDetailed))
185            {
186                NSLog.debug.appendln("ModelConnector: removing observer for: " + this);
187            }
188    
189            // It is not clear to me if this can happen with NSNotificationCenter's weak references but this can not hurt
190            NSNotificationCenter.defaultCenter().removeObserver(this);
191        }

196         * Unregister us from notifications so that we are not messaged.
197         */

205        /**
206         * Registers all of the EOModels that we have switched connection dictionaries for with their
207         * EODatabase provided that the model's name is not in modelNamesToIgnore().  This was done to
208         * work around a bug in this situation: two EOModels, an EOEntity in one is a sub-class of an
209         * EOEntity in the other, and an instance of the sub-class is created before any entities in the
210         * model containing the superclass are created.  This results in this exception:<br>
211         * <br>
212         * java.lang.NullPointerException<br>
213         * at com.webobjects.jdbcadaptor.FrontbasePlugIn._newPrimaryKey(FrontbasePlugIn.java:215)<br>
214         * at com.webobjects.jdbcadaptor.FrontbasePlugIn.newPrimaryKeys(FrontbasePlugIn.java:205)<br>
215         * at com.webobjects.jdbcadaptor.JDBCChannel.primaryKeysForNewRowsWithEntity(JDBCChannel.java:544)<br>
216         * at com.webobjects.eoaccess.EODatabaseContext._batchNewPrimaryKeysWithEntity(EODatabaseContext.java:5697)<br>
217         * at com.webobjects.eoaccess.EODatabaseContext.prepareForSaveWithCoordinator(EODatabaseContext.java:5676)<br>
218         * at com.webobjects.eocontrol.EOObjectStoreCoordinator.saveChangesInEditingContext(EOObjectStoreCoordinator.java:389)<br>
219         * at com.webobjects.eocontrol.EOEditingContext.saveChanges(EOEditingContext.java:2486)<br>
220         * <br>
221         * It appears that when the sub-class referred to its parent that it was not found in the database.  This code just registeres each EOModel with it's database.
222         */
223        public void registerModels()
224        {
225            /** require [has_models] modelGroup().models().count() > 0; **/
226    
227            // There are probably other and possibly better ways to accomplish this.
228            String anEntityName =  modelWithEntities().entityNames().lastObject();
229            EODatabaseDataSource ds = new EODatabaseDataSource(new EOEditingContext(), anEntityName);
230            EODatabaseContext dbContext = ds.databaseContext();
231            dbContext.lock();
232            try
233            {
234                EODatabase db = dbContext.database();
235    
236                NSArray allModels = modelGroup().models();
237                Enumeration modelEnumerator = allModels.objectEnumerator();
238                while (modelEnumerator.hasMoreElements())
239                {
240                    EOModel aModel = (EOModel)modelEnumerator.nextElement();
241    
242                    if (connectionDictionary().equals(aModel.connectionDictionary()))
243                    {
244                        if (NSLog.debugLoggingAllowedForLevel(NSLog.DebugLevelDetailed))
245                        {
246                            NSLog.debug.appendln("ModelConnector: registering model: " + aModel.name());
247                        }
248    
249                        if ( ! db.addModelIfCompatible(aModel))
250                        {
251                            NSLog.debug.appendln("Correct adaptor type? " + db.adaptor().name().equals(aModel.adaptorName()));
252                            NSLog.debug.appendln("db.adaptor().name()  " + db.adaptor().name());
253                            NSLog.debug.appendln("aModel.adaptorName()  " + aModel.adaptorName());
254                            NSLog.debug.appendln("Connection dictionaries match? " + db.adaptor().canServiceModel(aModel));
255                            NSLog.debug.appendln("aModel.connectionDictionary()  " + aModel.connectionDictionary());
256                            NSLog.debug.appendln("db.adaptor().connectionDictionary() " + db.adaptor().connectionDictionary());
257                            throw new RuntimeException("Unhandled condition: model '" + aModel.name() + "' is not compatible with database");
258                        }
259                    }
260                }
261            }
262            finally
263            {
264                dbContext.unlock();
265            }
266        }

270        /**
271         * Get an eo model from the model group that contains at least one entity.
272         *
273         * @return an eo model with at least one entity
274         */
275        public EOModel modelWithEntities()
276        {
277            /** require [has_models] modelGroup().models().count() > 0; **/
278    
279            EOModel resultModel = null;
280            NSArray allModels = modelGroup().models();
281            Enumeration modelEnumerator = allModels.objectEnumerator();
282            while (modelEnumerator.hasMoreElements())
283            {
284                EOModel aModel = (EOModel)modelEnumerator.nextElement();
285                if ((aModel.entityNames().count() > 0) &&  ( ! modelNamesToIgnore().containsObject(aModel.name())))
286                {
287                    resultModel = aModel;
288                    break;
289                }
290            }
291            return resultModel;
292    
293            /** ensure [valid_result] Result != null; **/
294        }

298        /**
299         * Connect the passed in model using this object's connection dictionary and adaptor name.
300         *
301         * @param theModel the model we are connecting
302         */
303        public void connectModel(EOModel theModel)
304        {
305            /** require [valid_param] theModel != null; **/
306    
307            if (NSLog.debugLoggingAllowedForLevel(NSLog.DebugLevelDetailed))
308            {
309                NSLog.debug.appendln("ModelConnector: Connecting model: " + theModel.name() +
310                        " with dictionary " + connectionDictionary());
311            }
312    
313            theModel.setConnectionDictionary(connectionDictionary());
314            theModel.setAdaptorName(adaptorName());
315    
316            /** ensure
317            [connection_dictionary_set] theModel.connectionDictionary().equals(connectionDictionary());
318            [adaptor_name_set] theModel.adaptorName().equals(adaptorName()); **/
319        }

323        /**
324         * The object in the notification is the model being added.  If this model's name is not in modelNamesToIgnore(), overrides the connection dictionary in the loaded model with connectionDictionary().
325         *
326         * @param aNotification notification of which EOModel is being added
327         */
328        public void handleModelAddedInDefaultGroupNotification(NSNotification aNotification)
329        {
330            /** require [valid_param] aNotification != null; **/
331    
332            EOModel theModel = (EOModel) aNotification.object();
333    
334            if ( ! isIgnoringModel(theModel))
335            {
336                connectModel(theModel);
337            }
338            else if (NSLog.debugLoggingAllowedForLevel(NSLog.DebugLevelDetailed))
339            {
340                NSLog.debug.appendln("ModelConnector: Skipping model: " + theModel.name());
341            }
342    
343            /** ensure [not_ignored_means_connection_disctionary_set]
344            ( ! isIgnoringModel((EOModel)aNotification.object())) ? ((EOModel)aNotification.object()).connectionDictionary().equals(connectionDictionary()) : true; **/
345        }
346    
347    
348    
349        /**
350         * The object in the notification is the model being added.  If this model's name is not in modelNamesToIgnore(), overrides the connection dictionary in the loaded model with connectionDictionary().
351         *
352         * @param aNotification notification of which EOModel is being added
353         */
354        public void handleModelAddedInCustomGroupNotification(NSNotification aNotification)
355        {
356            /** require [valid_param] aNotification != null; **/
357    
358            EOModel theModel = (EOModel)aNotification.object();
359    
360            if (( ! isIgnoringModel(theModel)) && (modelGroup().models().containsObject(theModel)))
361            {
362                connectModel(theModel);
363            }
364            else if (NSLog.debugLoggingAllowedForLevel(NSLog.DebugLevelDetailed))
365            {
366                NSLog.debug.appendln("ModelConnector: Skipping model: " + theModel.name());
367            }
368    
369            /** ensure [not_ignored_and_in_our_model_group_means_connection_disctionary_set]
370            (( ! isIgnoringModel((EOModel)aNotification.object())) && (modelGroup().models().containsObject((EOModel)aNotification.object()))) ? ((EOModel)aNotification.object()).connectionDictionary().equals(connectionDictionary()) : true; **/
371        }

375        /**
376         * Returns <code>true</code> if <code>aModel</code> will be ignored by this
377         * EOModelConnector instance.  The model will be ignored if it is in the list
378         * of models to ignore, if it has an adaptor type which is not compatible
379         * (not null and not our adaptor type), or if it does not belong to the model
380         * group that we are connecting.
381         *
382         * @param aModel the EOModel to check
383         * @return <code>true</code> if <code>aModel</code> will be ignored by this
384         * EOModelConnector instance
385         */
386        public boolean isIgnoringModel(EOModel aModel)
387        {
388            /** require [aModel_valid] aModel != null; **/
389            return (modelNamesToIgnore().containsObject(aModel.name()) ) ||
390                   ( (aModel.adaptorName() != null) &&
391                     ( ! aModel.adaptorName().equals(adaptorName())) );
392        }
