024    /**
025     * Get file format, image resolution, number of bits per pixel and optionally 
026     * number of images, comments and physical resolution from 
027     * JPEG, GIF, BMP, PCX, PNG, IFF, RAS, PBM, PGM, PPM, PSD and SWF files 
028     * (or input streams).
029     * <p>
030     * Use the class like this:
031     * <pre>
032     * ImageInfo ii = new ImageInfo();
033     * ii.setInput(in); // in can be InputStream or RandomAccessFile
034     * ii.setDetermineImageNumber(true); // default is false
035     * ii.setCollectComments(true); // default is false
036     * if (!ii.check()) {
037     *   System.err.println("Not a supported image file format.");
038     *   return;
039     * }
040     * System.out.println(ii.getFormatName() + ", " + ii.getMimeType() + 
041     *   ", " + ii.getWidth() + " x " + ii.getHeight() + " pixels, " + 
042     *   ii.getBitsPerPixel() + " bits per pixel, " + ii.getNumberOfImages() +
043     *   " image(s), " + ii.getNumberOfComments() + " comment(s).");
044     * </pre>
045     * You can also use this class as a command line program.
046     * Call it with a number of image file names as parameters:
047     * <pre>
048     *   java ImageInfo *.jpg *.png *.gif
049     * </pre>
050     * or call it without parameters and pipe data to it:
051     * <pre>
052     *   cat image.jpg | java ImageInfo
053     * </pre>
054     * <p>
055     * Known limitations:
056     * <ul>
057     * <li>When the determination of the number of images is turned off, GIF bits 
058     *  per pixel are only read from the global header.
059     *  For some GIFs, local palettes change this to a typically larger
060     *  value. To be certain to get the correct color depth, call
061     *  setDetermineImageNumber(true) before calling check().
062     *  The complete scan over the GIF file will take additional time.</li>
063     * <li>Transparency information is not included in the bits per pixel count.
064     *  Actually, it was my decision not to include those bits, so it's a feature! ;-)</li>
065     * </ul>
066     * <p>
067     * Requirements:
068     * <ul>
069     * <li>Java 1.1 or higher</li>
070     * </ul>
071     * <p>
072     * The latest version can be found at <a href="http://www.geocities.com/marcoschmidt.geo/image-info.html">http://www.geocities.com/marcoschmidt.geo/image-info.html</a>.

225            /**
226             * Call this method after you have provided an input stream or file
227             * using {@link #setInput(InputStream)} or {@link #setInput(DataInput)}.
228             * If true is returned, the file format was known and you information
229             * about its content can be retrieved using the various getXyz methods.
230             * @return if information could be retrieved from input
231             */
232            public boolean check() {
233                    format = -1;
234                    width = -1;
235                    height = -1;
236                    bitsPerPixel = -1;
237                    numberOfImages = 1;
238                    physicalHeightDpi = -1;
239                    physicalWidthDpi = -1;
240                    comments = null;
241                    try {
242                            int b1 = read() & 0xff;
243                            int b2 = read() & 0xff;
244                            if (b1 == 0x47 && b2 == 0x49) {
245                                    return checkGif();
246                            }
247                            else
248                            if (b1 == 0x89 && b2 == 0x50) {
249                                    return checkPng();
250                            }
251                            else
252                            if (b1 == 0xff && b2 == 0xd8) {
253                                    return checkJpeg();
254                            }
255                            else
256                            if (b1 == 0x42 && b2 == 0x4d) {
257                                    return checkBmp();
258                            }
259                            else
260                            if (b1 == 0x0a && b2 < 0x06) {
261                                    return checkPcx();
262                            }
263                            else
264                            if (b1 == 0x46 && b2 == 0x4f) {
265                                    return checkIff();
266                            }
267                            else
268                            if (b1 == 0x59 && b2 == 0xa6) {
269                                    return checkRas();
270                            }
271                            else
272                            if (b1 == 0x50 && b2 >= 0x31 && b2 <= 0x36) {
273                                    return checkPnm(b2 - '0');
274                            }
275                            else
276                            if (b1 == 0x38 && b2 == 0x42) {
277                                    return checkPsd();
278                            }
279                            else
280                            if (b1 == 0x46 && b2 == 0x57) {
281                                    return checkSwf();
282                            }
283                            else {
284                                    return false;
285                            }
286                    } catch (IOException ioe) {
287                            return false;
288                    }
289            }

291            private boolean checkBmp() throws IOException {
292                    byte[] a = new byte[44];
293                    if (read(a) != a.length) {
294                            return false;
295                    }
296                    width = getIntLittleEndian(a, 16);
297                    height = getIntLittleEndian(a, 20);
298                    if (width < 1 || height < 1) {
299                            return false;
300                    }
301                    bitsPerPixel = getShortLittleEndian(a, 26);
302                    if (bitsPerPixel != 1 && bitsPerPixel != 4 &&
303                        bitsPerPixel != 8 && bitsPerPixel != 16 &&
304                        bitsPerPixel != 24 && bitsPerPixel != 32) {
305                        return false;
306                    }
307                    int x = getIntLittleEndian(a, 36);
308                    if (x > 0) {
309                            setPhysicalWidthDpi(x);
310                    }
311                    int y = getIntLittleEndian(a, 40);
312                    if (y > 0) {
313                            setPhysicalHeightDpi(y);
314                    }
315                    format = FORMAT_BMP;
316                    return true;
317            }

319            private boolean checkGif() throws IOException {
320                    final byte[] GIF_MAGIC_87A = {0x46, 0x38, 0x37, 0x61};
321                    final byte[] GIF_MAGIC_89A = {0x46, 0x38, 0x39, 0x61};
322                    byte[] a = new byte[11]; // 4 from the GIF signature + 7 from the global header
323                    if (read(a) != 11) {
324                            return false;
325                    }
326                    if ((!equals(a, 0, GIF_MAGIC_89A, 0, 4)) &&
327                            (!equals(a, 0, GIF_MAGIC_87A, 0, 4))) {
328                            return false;
329                    }
330                    format = FORMAT_GIF;
331                    width = getShortLittleEndian(a, 4);
332                    height = getShortLittleEndian(a, 6);
333                    int flags = a[8] & 0xff;
334                    bitsPerPixel = ((flags >> 4) & 0x07) + 1;
335                    if (!determineNumberOfImages) {
336                            return true;
337                    }
338                    // skip global color palette
339                    if ((flags & 0x80) != 0) {
340                            int tableSize = (1 << ((flags & 7) + 1)) * 3;
341                            skip(tableSize);
342                    }
343                    numberOfImages = 0;
344                    int blockType;
345                    do
346                    {
347                            blockType = read();
348                            switch(blockType)
349                            {
350                                    case(0x2c): // image separator
351                                    {
352                                            if (read(a, 0, 9) != 9) {
353                                                    return false;
354                                            }
355                                            flags = a[8] & 0xff;
356                                            int localBitsPerPixel = (flags & 0x07) + 1;
357                                            if (localBitsPerPixel > bitsPerPixel) {
358                                                    bitsPerPixel = localBitsPerPixel;
359                                            }
360                                            if ((flags & 0x80) != 0) {
361                                                    skip((1 << localBitsPerPixel) * 3);
362                                            }
363                                            skip(1); // initial code length
364                                            int n;
365                                            do
366                                            {
367                                                    n = read();
368                                                    if (n > 0) {
369                                                            skip(n);
370                                                    }
371                                                    else
372                                                    if (n == -1) {
373                                                            return false;
374                                                    }
375                                            }
376                                            while (n > 0);
377                                            numberOfImages++;
378                                            break;
379                                    }
380                                    case(0x21): // extension
381                                    {
382                                            int extensionType = read();
383                                            if (collectComments && extensionType == 0xfe) {
384                                                    StringBuffer sb = new StringBuffer();
385                                                    int n;
386                                                    do
387                                                    {
388                                                            n = read();
389                                                            if (n == -1) {
390                                                                    return false;
391                                                            }
392                                                            if (n > 0) {
393                                                                    for (int i = 0; i < n; i++) {
394                                                                            int ch = read();
395                                                                            if (ch == -1) {
396                                                                                    return false;
397                                                                            }
398                                                                            sb.append((char)ch);
399                                                                    }
400                                                            }
401                                                    }
402                                                    while (n > 0);
403                                            } else {
404                                                    int n;
405                                                    do
406                                                    {
407                                                            n = read();
408                                                            if (n > 0) {
409                                                                    skip(n);
410                                                            }
411                                                            else
412                                                            if (n == -1) {
413                                                                    return false;
414                                                            }
415                                                    }
416                                                    while (n > 0);
417                                            }
418                                            break;
419                                    }
420                                    case(0x3b): // end of file
421                                    {
422                                            break;
423                                    }
424                                    default:
425                                    {
426                                            return false;
427                                    }
428                            }
429                    }
430                    while (blockType != 0x3b);
431                    return true;
432            }

434            private boolean checkIff() throws IOException {
435                    byte[] a = new byte[10];
436                    // read remaining 2 bytes of file id, 4 bytes file size 
437                    // and 4 bytes IFF subformat
438                    if (read(a, 0, 10) != 10) {
439                            return false;
440                    }
441                    final byte[] IFF_RM = {0x52, 0x4d};
442                    if (!equals(a, 0, IFF_RM, 0, 2)) {
443                            return false;
444                    }
445                    int type = getIntBigEndian(a, 6);
446                    if (type != 0x494c424d && // type must be ILBM...
447                        type != 0x50424d20) { // ...or PBM
448                        return false;
449                    }
450                    // loop chunks to find BMHD chunk
451                    do {
452                            if (read(a, 0, 8) != 8) {
453                                    return false;
454                            }
455                            int chunkId = getIntBigEndian(a, 0);
456                            int size = getIntBigEndian(a, 4);
457                            if ((size & 1) == 1) {
458                                    size++;
459                            }
460                            if (chunkId == 0x424d4844) { // BMHD chunk
461                                    if (read(a, 0, 9) != 9) {
462                                            return false;
463                                    }
464                                    format = FORMAT_IFF;
465                                    width = getShortBigEndian(a, 0);
466                                    height = getShortBigEndian(a, 2);
467                                    bitsPerPixel = a[8] & 0xff;
468                                    return (width > 0 && height > 0 && bitsPerPixel > 0 && bitsPerPixel < 33);
469                            } else {
470                                    skip(size);
471                            }
472                    } while (true);
473            }

475            private boolean checkJpeg() throws IOException {
476                    byte[] data = new byte[12];
477                    while (true) {
478                            if (read(data, 0, 4) != 4) {
479                                    return false;
480                            }
481                            int marker = getShortBigEndian(data, 0);
482                            int size = getShortBigEndian(data, 2);
483                            if ((marker & 0xff00) != 0xff00) {
484                                    return false; // not a valid marker
485                            }
486                            if (marker == 0xffe0) { // APPx 
487                                    if (size < 14) {
488                                            return false; // APPx header must be larger than 14 bytes
489                                    }
490                                    if (read(data, 0, 12) != 12) {
491                                            return false;
492                                    }
493                                    final byte[] APP0_ID = {0x4a, 0x46, 0x49, 0x46, 0x00};
494                                    if (equals(APP0_ID, 0, data, 0, 5)) {
495                                            if (data[7] == 1) {
496                                                    setPhysicalWidthDpi(getShortBigEndian(data, 8));
497                                                    setPhysicalHeightDpi(getShortBigEndian(data, 10));
498                                            }
499                                            else
500                                            if (data[7] == 2) {
501                                                    int x = getShortBigEndian(data, 8);
502                                                    int y = getShortBigEndian(data, 10);
503                                                    setPhysicalWidthDpi((int)(x * 2.54f));
504                                                    setPhysicalHeightDpi((int)(y * 2.54f));
505                                            }
506                                    }
507                                    skip(size - 14);
508                            }
509                            else
510                            if (collectComments && size > 2 && marker == 0xfffe) { // comment
511                                    size -= 2;
512                                    byte[] chars = new byte[size];
513                                    if (read(chars, 0, size) != size) {
514                                            return false;
515                                    }
516                                    String comment = new String(chars, "iso-8859-1");
517                                    comment = comment.trim();
518                                    System.out.println(comment);
519                                    addComment(comment);
520                            }
521                            else
522                            if (marker >= 0xffc0 && marker <= 0xffcf && marker != 0xffc4 && marker != 0xffc8) {
523                                    if (read(data, 0, 6) != 6) {
524                                            return false;
525                                    }
526                                    format = FORMAT_JPEG;
527                                    bitsPerPixel = (data[0] & 0xff) * (data[5] & 0xff);
528                                    width = getShortBigEndian(data, 3);
529                                    height = getShortBigEndian(data, 1);
530                                    return true;
531                            } else {
532                                    skip(size - 2);
533                            }
534                    }
535            }

537            private boolean checkPcx() throws IOException {
538                    byte[] a = new byte[64];
539                    if (read(a) != a.length) {
540                            return false;
541                    }
542                    if (a[0] != 1) { // encoding, 1=RLE is only valid value
543                            return false;
544                    }
545                    // width / height
546                    int x1 = getShortLittleEndian(a, 2);
547                    int y1 = getShortLittleEndian(a, 4);
548                    int x2 = getShortLittleEndian(a, 6);
549                    int y2 = getShortLittleEndian(a, 8);
550                    if (x1 < 0 || x2 < x1 || y1 < 0 || y2 < y1) {
551                            return false;
552                    }
553                    width = x2 - x1 + 1;
554                    height = y2 - y1 + 1;
555                    // color depth
556                    int bits = a[1];
557                    int planes = a[63];
558                    if (planes == 1 &&
559                        (bits == 1 || bits == 2 || bits == 4 || bits == 8)) {
560                            // paletted
561                            bitsPerPixel = bits;
562                    } else
563                    if (planes == 3 && bits == 8) {
564                            // RGB truecolor
565                            bitsPerPixel = 24;
566                    } else {
567                            return false;
568                    }
569                    setPhysicalWidthDpi(getShortLittleEndian(a, 10));
570                    setPhysicalHeightDpi(getShortLittleEndian(a, 10));
571                    format = FORMAT_PCX;
572                    return true;
573            }

575            private boolean checkPng() throws IOException {
576                    final byte[] PNG_MAGIC = {0x4e, 0x47, 0x0d, 0x0a, 0x1a, 0x0a};
577                    byte[] a = new byte[24];
578                    if (read(a) != 24) {
579                            return false;
580                    }
581                    if (!equals(a, 0, PNG_MAGIC, 0, 6)) {
582                            return false;
583                    }
584                    format = FORMAT_PNG;
585                    width = getIntBigEndian(a, 14);
586                    height = getIntBigEndian(a, 18);
587                    bitsPerPixel = a[22] & 0xff;
588                    int colorType = a[23] & 0xff;
589                    if (colorType == 2 || colorType == 6) {
590                            bitsPerPixel *= 3;
591                    }
592                    return true;
593            }

595            private boolean checkPnm(int id) throws IOException {
596                    if (id < 1 || id > 6) {
597                            return false;
598                    }
599                    final int[] PNM_FORMATS = {FORMAT_PBM, FORMAT_PGM, FORMAT_PPM};
600                    format = PNM_FORMATS[(id - 1) % 3];
601                    boolean hasPixelResolution = false;
602                    String s;
603                    while (true)
604                    {
605                            s = readLine();
606                            if (s != null) {
607                                    s = s.trim();
608                            }
609                            if (s == null || s.length() < 1) {
610                                    continue;
611                            }
612                            if (s.charAt(0) == '#') { // comment
613                                    if (collectComments && s.length() > 1) {
614                                            addComment(s.substring(1));
615                                    }
616                                    continue;
617                            }
618                            if (!hasPixelResolution) { // split "343 966" into width=343, height=966
619                                    int spaceIndex = s.indexOf(' ');
620                                    if (spaceIndex == -1) {
621                                            return false;
622                                    }
623                                    String widthString = s.substring(0, spaceIndex);
624                                    spaceIndex = s.lastIndexOf(' ');
625                                    if (spaceIndex == -1) {
626                                            return false;
627                                    }
628                                    String heightString = s.substring(spaceIndex + 1);
629                                    try {
630                                            width = Integer.parseInt(widthString);
631                                            height = Integer.parseInt(heightString);
632                                    } catch (NumberFormatException nfe) {
633                                            return false;
634                                    }
635                                    if (width < 1 || height < 1) {
636                                            return false;
637                                    }
638                                    if (format == FORMAT_PBM) {
639                                            bitsPerPixel = 1;
640                                            return true;
641                                    }
642                                    hasPixelResolution = true;
643                            }
644                            else
645                            {
646                                    int maxSample;
647                                    try {
648                                            maxSample = Integer.parseInt(s);
649                                    } catch (NumberFormatException nfe) {
650                                            return false;
651                                    }
652                                    if (maxSample < 0) {
653                                            return false;
654                                    }
655                                    for (int i = 0; i < 25; i++) {
656                                            if (maxSample < (1 << (i + 1))) {
657                                                    bitsPerPixel = i + 1;
658                                                    if (format == FORMAT_PPM) {
659                                                            bitsPerPixel *= 3;
660                                                    }
661                                                    return true;
662                                            }
663                                    }
664                                    return false;
665                            }
666                    }
667            }

669            private boolean checkPsd() throws IOException {
670                    byte[] a = new byte[24];
671                    if (read(a) != a.length) {
672                            return false;
673                    }
674                    final byte[] PSD_MAGIC = {0x50, 0x53};
675                    if (!equals(a, 0, PSD_MAGIC, 0, 2)) {
676                            return false;
677                    }
678                    format = FORMAT_PSD;
679                    width = getIntBigEndian(a, 16);
680                    height = getIntBigEndian(a, 12);
681                    int channels = getShortBigEndian(a, 10);
682                    int depth = getShortBigEndian(a, 20);
683                    bitsPerPixel = channels * depth;
684                    return (width > 0 && height > 0 && bitsPerPixel > 0 && bitsPerPixel <= 64);
685            }

687            private boolean checkRas() throws IOException {
688                    byte[] a = new byte[14];
689                    if (read(a) != a.length) {
690                            return false;
691                    }
692                    final byte[] RAS_MAGIC = {0x6a, (byte)0x95};
693                    if (!equals(a, 0, RAS_MAGIC, 0, 2)) {
694                            return false;
695                    }
696                    format = FORMAT_RAS;
697                    width = getIntBigEndian(a, 2);
698                    height = getIntBigEndian(a, 6);
699                    bitsPerPixel = getIntBigEndian(a, 10);
700                    return (width > 0 && height > 0 && bitsPerPixel > 0 && bitsPerPixel <= 24);
701            }

704            private boolean checkSwf() throws IOException {
705                    //get rid of the last byte of the signature, the byte of the version and 4 bytes of the size
706                    byte[] a = new byte[6];
707                    if (read(a) != a.length) {
708                            return false;
709                    }
710                    format = FORMAT_SWF;
711                    int bitSize = (int)readUBits( 5 );
712                    int maxX = (int)readSBits( bitSize );
713                    int maxY = (int)readSBits( bitSize );
714                    width = maxX/20; //cause we're in twips
715                    height = maxY/20;  //cause we're in twips
716                    setPhysicalWidthDpi(72);
717                    setPhysicalHeightDpi(72);
718                    return (width > 0 && height > 0);
719            }

1083        /**
1084         * Read an unsigned value from the given number of bits
1085         */
1086            public long readUBits( int numBits ) throws IOException
1087            {
1088                    if (numBits == 0) {
1089                            return 0;
1090                    }
1091                    int bitsLeft = numBits;
1092                    long result = 0;
1093                    if (bitPos == 0) { //no value in the buffer - read a byte
1094                            if (in != null) {
1095                                    bitBuf = in.read();
1096                            } else {
1097                                    bitBuf = din.readByte();
1098                            }
1099                            bitPos = 8;
1100                    }
1101            
1102                while( true )
1103            {
1104                int shift = bitsLeft - bitPos;
1105                if( shift > 0 )
1106                {
1107                    // Consume the entire buffer
1108                    result |= bitBuf << shift;
1109                    bitsLeft -= bitPos;
1110    
1111                    // Get the next byte from the input stream
1112                    if (in != null) {
1113                      bitBuf = in.read();
1114                    } else {
1115                      bitBuf = din.readByte();
1116                    }
1117                    bitPos = 8;
1118                }
1119                else
1120                {
1121                    // Consume a portion of the buffer
1122                    result |= bitBuf >> -shift;
1123                    bitPos -= bitsLeft;
1124                    bitBuf &= 0xff >> (8 - bitPos);       // mask off the consumed bits
1125    
1126                    return result;
1127                }
1128            }        
1129        }

1131            /**
1132         * Read a signed value from the given number of bits
1133         */
1134        private int readSBits( int numBits ) throws IOException
1135        {
1136            // Get the number as an unsigned value.
1137            long uBits = readUBits( numBits );
1138    
1139            // Is the number negative?
1140            if( ( uBits & (1L << (numBits - 1))) != 0 )
1141            {
1142                // Yes. Extend the sign.
1143                uBits |= -1L << numBits;
1144            }
1145    
1146            return (int)uBits;        
1147        }
