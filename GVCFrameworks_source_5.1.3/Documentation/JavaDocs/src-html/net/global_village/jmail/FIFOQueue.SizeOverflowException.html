001    package net.global_village.jmail;
002    
003    import java.util.*;
004    
005    
006    /**
007     * Simple implementation of FIFO queue to support ThreadedMailAgent.
008     * 
009     * This code is adapted from ERQueue from the ERJavaMail framework of Project Wonder.  This file 
010     * is protected under the GNU Lesser Public License.
011     * 
012     * @author Camille Troillard <tuscland@mac.com>
013     * @author Sacha Mallais <sacha@global-village.net>
014     */
015    public class FIFOQueue extends Vector {
016    
017        protected int _maxSize = 0;
018        public int maxSize () { return _maxSize; }
019        public void setMaxSize (int size) { _maxSize = size; }
020    
021        public static class SizeOverflowException extends Exception {
022            public SizeOverflowException () { super (); }
023        }
024        
025        public FIFOQueue() {
026            this (0);
027        }
028    
029        public FIFOQueue(int maxSize) {
030            super ();
031            this.setMaxSize (maxSize);
032        }
033    
034        public Object push (Object item) throws SizeOverflowException {
035            if ((_maxSize == 0) || (this.size () < _maxSize))
036                this.addElement (item);
037            else
038                throw new SizeOverflowException ();
039    
040            return item;
041        }
042    
043        public synchronized Object pop () {
044            Object element = this.elementAt (0);
045            this.removeElementAt (0);
046            return element;
047        }
048    
049        public synchronized Object peek () {
050            return this.elementAt (0);
051        }
052    
053        public boolean empty () {
054            return (this.size () == 0);
055        }
056    
057        public synchronized int search (Object o) {
058            return this.indexOf (o);
059        }
060    }
