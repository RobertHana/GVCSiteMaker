001    package net.global_village.foundation;
002    
003    import java.text.*;
004    
005    
006    /**
007     * This class simply extends (renames) <code>java.text.DecimalFormat</code> as documentation that it 
008     * returns <code>java.lang.Long</code>.
009     *
010     * @author Copyright (c) 2001-2005  Global Village Consulting, Inc.  All rights reserved.
011     * This software is published under the terms of the Educational Community License (ECL) version 1.0,
012     * a copy of which has been included with this distribution in the LICENSE.TXT file.
013     */
014    public class LongFormatter extends DecimalFormat
015    {
016    
017        /** Convenience instance.  **/
018        public static final LongFormatter Formatter = new LongFormatter();
019    
020        /** Formats output without any commas. */
021        public static final LongFormatter NoCommaFormatter = new LongFormatter("0");
022    
023        
024        
025        /**
026         * @see DecimalFormat#DecimalFormat()
027         */
028        public LongFormatter()
029        {
030            super();
031        }
032    
033        
034        
035        /**
036         * @see DecimalFormat#DecimalFormat(String, DecimalFormatSymbols)
037         */
038        public LongFormatter(String pattern, DecimalFormatSymbols symbols)
039        {
040            super(pattern, symbols);
041        }
042    
043        
044        
045        /**
046         * @see DecimalFormat#DecimalFormat(String)
047         */
048        public LongFormatter(String pattern)
049        {
050            super(pattern);
051        }
052        
053        
054    }
