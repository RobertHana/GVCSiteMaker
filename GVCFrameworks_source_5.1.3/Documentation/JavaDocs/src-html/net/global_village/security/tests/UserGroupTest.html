083        /**
084         * Test userGroupExists()
085         */
086        public void testUserGroupExists()
087        {
088            // occaisionally get errors on this test but usually sweeping will get rid of them.
089            assertTrue( ! UserGroup.userGroupExists("Some ficticious group that does not exist", editingContext));
090            assertTrue(UserGroup.userGroupExists(userGroupOne.name(), editingContext));
091    
092            // Test DBC
093            try { UserGroup.userGroupExists(null, editingContext); fail("DBC failed to catch a null string name parameter"); }
094            catch (RuntimeException e) {  }
095    
096            try { UserGroup.userGroupExists("", editingContext); fail("DBC failed to catch an empty string name parameter");
097            } catch (RuntimeException e) { }
098    
099            try { UserGroup.userGroupExists(userGroupOne.name(), null); fail("DBC failed to catch a null editing context parameter");
100            } catch (Exception e) { }
101        }

167        /**
168         * Test compareByName()
169         */
170        public void testCompareByName()
171        {
172            // additional groups to compare to
173            UserGroup userGroupTwo = (UserGroup) EOUtilities.objectMatchingKeyAndValue(editingContext,
174                                                                                       "GVCUserGroup",
175                                                                                       "name",
176                                                                                       SecurityTestNameConstants.kGroupTwoName);
177            UserGroup userGroupThree = (UserGroup) EOUtilities.objectMatchingKeyAndValue(editingContext,
178                                                                                         "GVCUserGroup",
179                                                                                         "name",
180                                                                                         SecurityTestNameConstants.kGroupThreeName);
181            UserGroup userGroupFour = (UserGroup) EOUtilities.objectMatchingKeyAndValue(editingContext,
182                                                                                        "GVCUserGroup",
183                                                                                        "name",
184                                                                                        SecurityTestNameConstants.kGroupFourName);
185            // new style tests
186            assertTrue(userGroupFour.compareByName(userGroupOne) < 0);
187            assertTrue(userGroupOne.compareByName(userGroupThree) < 0);
188            assertTrue(userGroupThree.compareByName(userGroupTwo) < 0);
189            // negative test
190            assertTrue( ! (userGroupFour.compareByName(userGroupTwo) > 0));
191            // old style tests  
192            assertTrue(userGroupOne.compareByName(userGroupTwo) < 0);
193            assertTrue(userGroupOne.compareByName(userGroupOne) == 0);
194            assertTrue(userGroupTwo.compareByName(userGroupOne) > 0);
195            // Test DBC
196            try { userGroupOne.compareByName(null); fail("DBC failed to catch null parameter"); }
197            catch (Exception e) { }
198        }

213        /**
214          * Test activeUsers()
215          */
216        public void testActiveUsers()
217        {
218             // userGroupOne contains userOne who has not has it's account disabled userTwo is not in userGroupOne
219             NSArray activeUsersInGroupOne = userGroupOne.activeUsers();
220             assertTrue("activeUsersInGroupOne.count() > 0", activeUsersInGroupOne.count() > 0); // only one active user currently
221             assertTrue("activeUsersInGroupOne.containsObject(userOne)", activeUsersInGroupOne.containsObject(userOne));
222             assertTrue("! activeUsersInGroupOne.containsObject(userTwo)", ! activeUsersInGroupOne.containsObject(userTwo));
223    
224             // test on group with more then one member now including a disabled member!
225             UserGroup userGroupFour = (UserGroup) EOUtilities.objectMatchingKeyAndValue(editingContext,
226                                                                                         "GVCUserGroup",
227                                                                                         "name",
228                                                                                         SecurityTestNameConstants.kGroupFourName);
229             NSArray activeUsersInGroupFour = userGroupFour.activeUsers();
230             assertTrue("activeUsersInGroupFour.count() == 3", activeUsersInGroupFour.count() == 3);
231             assertEquals("activeUsersInGroupFour.objectAtIndex(0).equals(userOne)", activeUsersInGroupFour.objectAtIndex(0), userOne);
232             assertEquals("activeUsersInGroupFour.lastObject().equals(userTwo)", activeUsersInGroupFour.lastObject(), userTwo);  // userThree is in the middle
233        }

238         * Test allUserGroup()
239         */
240        public void testOrderedListOfAllUserGroups()
241        {
242            UserGroup userGroupTwo = (UserGroup) EOUtilities.objectMatchingKeyAndValue(editingContext,
243                                                                                 "GVCUserGroup",
244                                                                                 "name",
245                                                                                 SecurityTestNameConstants.kGroupTwoName);
246            UserGroup userGroupThree = (UserGroup) EOUtilities.objectMatchingKeyAndValue(editingContext,
247                                                                                         "GVCUserGroup",
248                                                                                         "name",
249                                                                                         SecurityTestNameConstants.kGroupThreeName);
250            UserGroup userGroupFour = (UserGroup) EOUtilities.objectMatchingKeyAndValue(editingContext,
251                                                                                        "GVCUserGroup",
252                                                                                        "name",
253                                                                                        SecurityTestNameConstants.kGroupFourName);
254            NSArray sortedGroups = UserGroup.orderedListOfAllUserGroups(editingContext);
255            assertTrue(sortedGroups.indexOfObject(userGroupFour) < sortedGroups.indexOfObject(userGroupOne));
256            assertTrue(sortedGroups.indexOfObject(userGroupOne) < sortedGroups.indexOfObject(userGroupThree));
257            assertTrue(sortedGroups.indexOfObject(userGroupThree) < sortedGroups.indexOfObject(userGroupTwo));
258        }
