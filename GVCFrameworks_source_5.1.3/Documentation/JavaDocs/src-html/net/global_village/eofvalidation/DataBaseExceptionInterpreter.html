014     */
015    public abstract class DataBaseExceptionInterpreter
016    {
017    
018        /**
019         * Returns the name of the violated constraint from an exception caused by an
020         * integrity constraint violation.
021         *
022         * @param exception exception caused by an integrity constraint violation
023         * @return name of the violated constraint
024         */
025        public abstract String violatedIntegrityConstraintName(SQLException exception);
026    
027    
028    
029        /**
030         * Returns the name of the affected table from an exception caused by an
031         * integrity constraint violation.
032         *
033         * @param exception exception caused by an integrity constraint violation
034         * @return name of the affected table
035         */
036        public abstract String tableNameFromSQLException(SQLException exception);
037    
038    
039    }
