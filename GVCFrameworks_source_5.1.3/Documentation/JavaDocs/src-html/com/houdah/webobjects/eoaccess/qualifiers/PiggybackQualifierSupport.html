001    package com.houdah.webobjects.eoaccess.qualifiers;
002    
003    import com.houdah.webobjects.eocontrol.qualifiers.PiggybackQualifier;
004    import com.webobjects.eoaccess.EOEntity;
005    import com.webobjects.eoaccess.EOSQLExpression;
006    import com.webobjects.eoaccess.EOQualifierSQLGeneration.Support;
007    import com.webobjects.eocontrol.EOQualifier;
008    
009    /**
010     * <p>
011     * Support class for the PiggybackQualifier.
012     * </p>

022     * @author Pierre Bernard http://homepage.mac.com/i_love_my/webobjects.html
023     */
024    public class PiggybackQualifierSupport extends QualifierGenerationSupport
025    {
026            // Public instance methods
027    
028            /*
029             * (non-Javadoc)
030             * 
031             * @see com.webobjects.eoaccess.EOQualifierSQLGeneration.Support#sqlStringForSQLExpression(com.webobjects.eocontrol.EOQualifier,
032             *      com.webobjects.eoaccess.EOSQLExpression)
033             */
034            public String sqlStringForSQLExpression(EOQualifier qualifier, EOSQLExpression expression)
035            {
036                    PiggybackQualifier pbQualifier = (PiggybackQualifier) qualifier;
037                    EOQualifier subQualifier = pbQualifier.qualifier();
038                    Support support = supportForClass(subQualifier.getClass());
039    
040                    if (support == null) {
041                            throw new IllegalArgumentException("Qualifier " + subQualifier
042                                            + " has no support class");
043                    }
044    
045                    StringBuffer buffer = new StringBuffer();
046    
047                    buffer.append("(");
048                    buffer.append(support.sqlStringForSQLExpression(subQualifier, expression));
049                    buffer.append(")");
050    
051                    return buffer.toString();
052            }
053    
054            /*
055             * (non-Javadoc)
056             * 
057             * @see com.webobjects.eoaccess.EOQualifierSQLGeneration.Support#schemaBasedQualifierWithRootEntity(com.webobjects.eocontrol.EOQualifier,
058             *      com.webobjects.eoaccess.EOEntity)
059             */
060            public EOQualifier schemaBasedQualifierWithRootEntity(EOQualifier qualifier, EOEntity entity)
061            {
062                    PiggybackQualifier pbQualifier = (PiggybackQualifier) qualifier;
063                    EOQualifier subQualifier = pbQualifier.qualifier();
064                    Support support = supportForClass(subQualifier.getClass());
065    
066                    if (support == null) {
067                            throw new IllegalArgumentException("Qualifier " + subQualifier
068                                            + " has no support class");
069                    }
070    
071                    EOQualifier schemaBasedSubQualifier = support.schemaBasedQualifierWithRootEntity(
072                                    subQualifier, entity);
073    
074                    if (subQualifier == schemaBasedSubQualifier) {
075                            return pbQualifier;
076                    }
077                    else {
078                            PiggybackQualifier schemaBasedQualifier = new PiggybackQualifier(
079                                            schemaBasedSubQualifier);
080    
081                            schemaBasedQualifier.userInfo().addEntriesFromDictionary(pbQualifier.userInfo());
082    
083                            return schemaBasedQualifier;
084                    }
085            }
086    
087            /*
088             * (non-Javadoc)
089             * 
090             * @see com.webobjects.eoaccess.EOQualifierSQLGeneration.Support#qualifierMigratedFromEntityRelationshipPath(com.webobjects.eocontrol.EOQualifier,
091             *      com.webobjects.eoaccess.EOEntity, java.lang.String)
092             */
093            public EOQualifier qualifierMigratedFromEntityRelationshipPath(EOQualifier qualifier,
094                            EOEntity entity, String relationshipPath)
095            {
096                    PiggybackQualifier pbQualifier = (PiggybackQualifier) qualifier;
097                    EOQualifier subQualifier = pbQualifier.qualifier();
098                    Support support = supportForClass(subQualifier.getClass());
099    
100                    if (support == null) {
101                            throw new IllegalArgumentException("Qualifier " + subQualifier
102                                            + " has no support class");
103                    }
104    
105                    EOQualifier migratedSubQualifier = support.qualifierMigratedFromEntityRelationshipPath(
106                                    subQualifier, entity, relationshipPath);
107    
108                    if (subQualifier == migratedSubQualifier) {
109                            return pbQualifier;
110                    }
111                    else {
112                            PiggybackQualifier migratedQualifier = new PiggybackQualifier(migratedSubQualifier);
113    
114                            migratedQualifier.userInfo().addEntriesFromDictionary(pbQualifier.userInfo());
115    
116                            return migratedQualifier;
117                    }
118            }
119    }
