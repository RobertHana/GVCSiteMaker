002

004

006    
007    import net.global_village.foundation.GVCBoolean;
008    
009

017    public class BooleanTest extends TestCase
018    {
019        private GVCBoolean trueValue;
020        private GVCBoolean falseValue;
021    
022    
023        public BooleanTest(String name)
024        {
025            super(name);
026        }
027    
028    
029    
030        /**
031         * Sets up the fixtures.
032         * @exception Exception an exception that the setup may throw
033         */
034        public void setUp() throws java.lang.Exception
035        {
036            super.setUp();
037            trueValue = new GVCBoolean(true);
038            falseValue = new GVCBoolean(false);
039        }
040    
041    
042    
043        /**
044         * Test the various factory methods and isTrue and isFalse.
045         */
046        public void testFactoryMethods()
047        {
048            assertTrue(trueValue.isTrue());
049            assertTrue(falseValue.isFalse());
050    
051            trueValue = GVCBoolean.booleanWithBoolean(true);
052            falseValue = GVCBoolean.booleanWithBoolean(false);
053    
054            assertTrue(trueValue.isTrue());
055            assertTrue(falseValue.isFalse());
056    
057            trueValue = GVCBoolean.booleanWithString("Y");
058            falseValue = GVCBoolean.booleanWithString("N");
059    
060            assertTrue(trueValue.isTrue());
061            assertTrue(falseValue.isFalse());
062    
063            assertTrue(GVCBoolean.yes().isTrue());
064            assertTrue(GVCBoolean.no().isFalse());
065    
066            try
067            {
068                GVCBoolean.booleanWithString("n");
069                fail("Expected exception did not occur: missing precondition on booleanWithString()");
070            }
071            catch (RuntimeException e) { }
072        }
073    
074    
075    
076        /**
077         * Test booleanValue() and stringValue().
078         */
079        public void testSpecialAccessors()
080        {
081            assertTrue(trueValue.booleanValue() == true);
082            assertTrue(falseValue.booleanValue() == false);
083    
084            assertTrue(trueValue.stringValue().equals("Y"));
085            assertTrue(falseValue.stringValue().equals("N"));
086        }
087    
088    
089    
090        /**
091         * Test toString().
092         */
093        public void testToString()
094        {
095            assertTrue(trueValue.toString().equals("Yes"));
096            assertTrue(falseValue.toString().equals("No"));
097        }
098    
099    
100    
101        /**
102         * Test negate().
103         */
104        public void testNegate()
105        {
106            assertTrue(trueValue.negate().equals(falseValue));
107            assertTrue(falseValue.negate().equals(trueValue));
108        }
109    
110    
111    
112        /**
113         * Test equals().
114         */
115        public void testEquals()
116        {
117            assertTrue(trueValue.equals(new GVCBoolean(true)));
118            assertTrue(falseValue.equals(new GVCBoolean(false)));
119            assertTrue( ! trueValue.equals(new GVCBoolean(false)));
120            assertTrue( ! falseValue.equals(new GVCBoolean(true)));
121            assertTrue( ! falseValue.equals(new NSArray()));
122        }
123    
124    
125    
126        /**
127         * Test compareTo().
128         */
129        public void testCompareTo()
130        {
131            assertTrue(trueValue.compareTo(falseValue) >= 1);
132            assertTrue(falseValue.compareTo(trueValue) <= -1);
133            assertTrue(trueValue.compareTo(trueValue) == 0);
134            assertTrue(falseValue.compareTo(falseValue) == 0);
135        }
136    
137    
138    
139        /**
140         * Test clone().
141         */
142        public void testClone()
143        {
144            assertNotNull(trueValue.clone());
145            assertTrue(trueValue.clone().equals(new GVCBoolean(true)));
146        }
147    
148    
149    
150    }
