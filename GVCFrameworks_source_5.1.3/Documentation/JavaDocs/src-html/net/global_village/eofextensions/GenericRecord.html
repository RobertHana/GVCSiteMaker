027         * @return primary key value for this object
028         */
029        public static Object primaryKeyValue(com.webobjects.eocontrol.EOGenericRecord record)
030        {
031            /** require [is_in_editingContext] record.editingContext() != null;  **/
032            NSDictionary pkDictionary = EOUtilities.primaryKeyForObject(record.editingContext(), record);
033            if (pkDictionary.count() > 1)
034            {
035                throw new RuntimeException("primaryKeyValue called for entity with compound key: " + record.entityName());
036            }
037    
038            return pkDictionary.allValues().lastObject();
039            /** ensure [valid_result] Result != null;  **/
040        }

044        /**
045         * Type coercing version of EOUtilities.objectMatchingKeyAndValue.  Uses EOAttribute.validateValue
046         * to convert value into the correct type (e.g. String to Number) if it is not of the correct type.
047         * FrontBase does this in the plugin, but other databases do not (MySQL, MSSQL).
048         *
049         * @param ec editing context to return EO in
050         * @param entityName name of the EOEntity to fetch object for
051         * @param attributeName name of attribute corresponding to value
052         * @param value the value of the attribute to qualify on
053         *
054         * @return EO matching value for attributeName
055         * @throws EOObjectNotAvailableException if product with given ID does not exist
056         */
057        public static EOEnterpriseObject objectMatchingKeyAndValue(EOEditingContext ec,
058                                                                   String entityName,
059                                                                   String attributeName,
060                                                                   Object value)
061        {
062            /** require [valid_ec] ec != null;
063                        [valid_entityName] entityName != null && EOUtilities.entityNamed(ec, entityName) != null;
064                        [valid_ec] attributeName != null && EOUtilities.entityNamed(ec, entityName).attributeNamed(attributeName) != null;
065             **/
066            EOEntity entity = EOUtilities.entityNamed(ec, entityName);
067            value = entity.attributeNamed(attributeName).validateValue(value);
068            return EOUtilities.objectMatchingKeyAndValue(ec, entityName, attributeName, value);
069        }

151        /**
152         * Detecting WO 5 dummy fault EOs.  The code below is a translation of code posted here:
153         * From http://wodeveloper.com/omniLists/eof/2001/November/msg00023.html
154         *
155         * As of WO4.5, EOF resolves faults for objects that don't exist in the database to a dummy object with mostly empty attributes (except those
156        that are set in -init), whereas earlier versions of EOF threw a _fireFault exception.  This method detects whether the receiver is such a dummy object.
157         *
158         *  A distinguishing feature of these dummy EOs is that they don't have a corresponding database context snapshot, so we use this to detect its dummy-ness.  This seems more robust than checking if all or most attributes are empty, since these can be changed by client code.
159         *
160         *  Dummy fault EOs should be removed from the editingContext (using forgetObject) before it is invalidated (invalidateAllObjects or invalidateObjectWithGlobalID), otherwise an unrecoverable "decrementSnapshotCountForGlobalID - unable to decrement snapshot count for object with global ID" exception is thrown by EOF.<br><br>
161         *
162         * See also: http://www.omnigroup.com/mailman/archive/eof/2001-May/001988.html andfollowing.
163         */
164        public boolean isDummyFaultEO()
165        {
166            return EOObject.isDummyFaultEO(this);
167        }

171        /**
172         * EOCopying interface. Returns a copy of this object, copying related objects as well.  The actual copy mechanism (by reference, shallow, deep, or custom) for each object is up to the object being copied.  If a copy already exists in <code>copiedObjects</code>, then that is returned instead of making a new copy.  This allows complex graphs of objects, including those with cycles, to be copied without producing duplicate objects.  The grapch of copied objects will be the same regardless of where copy is started with two exceptions: it is started on a reference copied object or a reference copied object is the only path between two disconnected parts of the graph.  In these cases the reference copied object prevents the copy from following the graph further.
173         *
174         * @param copiedObjects - the copied objects keyed on the EOGlobalID of the object the copy was made from.
175         * @return a copy of this object
176         */
177        public EOEnterpriseObject copy(NSMutableDictionary copiedObjects)
178        {
179            /** require [copiedObjects_not_null] copiedObjects != null;  **/
180    
181           return EOCopying.DefaultImplementation.copy(copiedObjects, this);
182    
183            /** ensure [copy_made] Result != null; [copy_recorded] copiedObjects.objectForKey(editingContext().globalIDForObject(this)) != null;  **/
184        }

192         */

294        /**
295         * Returns true if the first object is equal to the second object.  This method treats an object with a value NSKeyValueCoding.NullValue to be equal to null.
296         * They are equal if they are either both null or both are not null and first.equals(second) evaluates to true.
297         *
298         * @param first the first object to compare
299         * @param second the second object to compare
300         * @return true if the first object is equal to the second object
301         */
302        public static boolean equals(Object first, Object second)
303        {
304            boolean firstIsNull = first == null || first.equals(NSKeyValueCoding.NullValue);
305            boolean secondIsNull = second == null || second.equals(NSKeyValueCoding.NullValue);
306    
307            return ((firstIsNull && secondIsNull) || ((first != null) && (first.equals(second))));
308        }

357        /**
358         * Returns the committed snapshot for this EO.  The committed snapshot is the values
359         * last fetched from or saved to the database and do not include any local, unsaved
360         * edits.
361         *
362         * This method exists because
363         * {@link com.webobjects.eocontrol.EOEditingContext#committedSnapshotForObject EOEditingContext.committedSnapshotForObject()}
364         * gives unexpected results for newly inserted objects if
365         * {@link com.webobjects.eocontrol.EOEditingContext#processRecentChanges() EOEditingContext.processRecentChanges()}
366         * has been called. This method always returns a dictionary whose values are
367         * all NSKeyValueCoding.NullValue in the case of a newly inserted object.
368         *
369         * @return the committed snapshot for this EO
370         */
371        public NSDictionary committedSnapshot() {
372            if (!isNewEO()) {
373                return editingContext().committedSnapshotForObject(this);
374            }
375    
376            NSArray keys = allPropertyKeys();
377            NSMutableDictionary allNullDict = new NSMutableDictionary(keys.count());
378            ERXDictionaryUtilities.setObjectForKeys(allNullDict, NSKeyValueCoding.NullValue, keys);
379            return allNullDict;
380        }

384        /**
385         * Returns the committed value associated with <code>key</code> based from its committed snapshot.
386         *
387         * @param key the key to be searched for
388         * @return the value for <code>key</code>, or <code>null</code> if no value is associated with key
389         */
390        public Object committedValueFor(String key)
391        {
392            /** require [key_not_null] key != null;  **/
393            Object value = committedSnapshot().objectForKey(key);
394            return value.equals(NSKeyValueCoding.NullValue) ? null : value;
395        }

399        /**
400         * Overridden to allow bindings like eo.committedValueFor.name.
401         *
402         * @see #committedValueFor(String)
403         * @see com.webobjects.eocontrol.EOCustomObject#valueForKeyPath(java.lang.String)
404         *
405         * @param keyPath keyPath to return value for
406         * @return committed value for key if keyPath starts with "committedValueFor."
407         * @throws IllegalArgumentException if keyPath starts with "committedValueFor." and has a key path after that (not a simple key)
408         */
409        public Object valueForKeyPath(String keyPath)
410        {
411            if (keyPath.startsWith("committedValueFor."))
412            {
413                String key = keyPath.substring(keyPath.indexOf('.') + 1);
414                if (key.indexOf('.') > -1)
415                {
416                    throw new IllegalArgumentException(keyPath + " contains a key path, not a simple key");
417                }
418                return committedValueFor(key);
419            }
420    
421            return super.valueForKeyPath(keyPath);
422        }

426        /**
427         * Overrides validateForSave to call validateForSave(NSMutableArray) and add the result of that
428         * to what the super class produces.  Subclasses should implement validateForSave(NSMutableArray)
429         * rather than the standard EOF method.  Doing so makes validateForSave(NSMutableArray) much
430         * shorter and easier to write.
431         *
432         * @see com.webobjects.eocontrol.EOCustomObject#validateForSave()
433         * @see #validateForSave(NSMutableArray)
434         * @throws ValidationException if validation error are found
435         */
436        public void validateForSave() throws ValidationException
437        {
438            NSMutableArray exceptions = new NSMutableArray();
439    
440            try
441            {
442                super.validateForSave();
443            }
444            catch ( NSValidation.ValidationException e)
445            {
446                exceptions.addObject(e);
447            }
448    
449            validateForSave(exceptions);
450    
451            if (exceptions.count() > 0)
452            {
453                throw NSValidation.ValidationException.aggregateExceptionWithExceptions(exceptions);
454            }
455        }
