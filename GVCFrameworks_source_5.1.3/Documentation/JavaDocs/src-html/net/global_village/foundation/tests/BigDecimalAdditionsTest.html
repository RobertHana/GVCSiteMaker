015    public class BigDecimalAdditionsTest extends TestCase
016    {
017        BigDecimal test1;
018        BigDecimal test2;
019        BigDecimal test3;
020        BigDecimal test4;
021        BigDecimal test5;
022        BigDecimal test6;
023        BigDecimal test7;
024        BigDecimal test8;
025        BigDecimal test9;
026        BigDecimal test10;
027        BigDecimal test11;
028        BigDecimal test12;
029        
030    
031        public BigDecimalAdditionsTest(String name)
032        {
033            super(name);
034        }
035    
036    
037    
038        /**
039         * Sets up the fixtures.
040         * @exception Exception an exception that the setup may throw
041         */
042        public void setUp() throws java.lang.Exception
043        {
044            super.setUp();
045            test1 = new BigDecimal("2.6543");
046            test2 = new BigDecimal("2.7");
047            test3 = new BigDecimal("2.6");
048            test4 = new BigDecimal("2.5");
049            test5 = new BigDecimal("2.8");
050            test6 = new BigDecimal("2.9");
051            test7 = new BigDecimal("12.6");
052            test8 = new BigDecimal("2.65432");
053            test9 = new BigDecimal("2.65439");
054            test10 = new BigDecimal("2.6544");
055            test11 = new BigDecimal("2.6543");
056            test12 = new BigDecimal("2.65449");
057        }
058    
059    
060    
061        /**
062         * Test equalsWithScale.
063         */
064        public void testEqualsWithScale()
065        {
066            assertTrue(BigDecimalAdditions.equalsWithScale(test1, test2));
067            assertTrue(BigDecimalAdditions.equalsWithScale(test1, test3));
068            assertTrue( ! BigDecimalAdditions.equalsWithScale(test1, test4));
069            assertTrue(BigDecimalAdditions.equalsWithScale(test1, test5));
070            assertTrue( ! BigDecimalAdditions.equalsWithScale(test1, test6));
071            assertTrue( ! BigDecimalAdditions.equalsWithScale(test1, test7));
072            assertTrue(BigDecimalAdditions.equalsWithScale(test1, test8));
073            assertTrue(BigDecimalAdditions.equalsWithScale(test1, test9));
074            assertTrue(BigDecimalAdditions.equalsWithScale(test1, test10));
075            assertTrue(BigDecimalAdditions.equalsWithScale(test1, test11));
076            assertTrue( ! BigDecimalAdditions.equalsWithScale(test1, test12));
077        }
078    
079    
080    
081        /**
082         * Test equalsWithScale.
083         */
084        public void testBetween()
085        {
086            assertTrue(BigDecimalAdditions.between(test3, test4, test2, true));
087            assertTrue(BigDecimalAdditions.between(test3, test3, test2, true));
088            assertTrue(BigDecimalAdditions.between(test3, test4, test3, true));
089            assertTrue(BigDecimalAdditions.between(test3, test3, test3, true));
090            assertTrue( ! BigDecimalAdditions.between(test3, test3, test2, false));
091            assertTrue( ! BigDecimalAdditions.between(test3, test4, test3, false));
092            assertTrue( ! BigDecimalAdditions.between(test3, test3, test3, false));
093        }
