007    /**
008     * A component that displays information of an array of Enterprise Objects in a drill-down effect.
009     *
010     * <b>It provides the following functionalities/flexibilities:</b><br>
011     * <ul>
012     * <li>display data in a collapsible nested list (drill down hierarchy) based on which attributes are being displayed
013     * <li>expand and collapse rows to show or hide detail data
014     * <li>reorder columns when title is clicked, with order controlled by client
015     * <li>perform specific client action when a row is clicked
016     * <li>perform simple calculations
017     * <li>allows client to control sort ordering of rows
018     * <li>displays list in alternating colour rows and columns
019     * </ul>
020     *
021     * <b>To use this component, the following bindings need to be setup:</b><br>
022     * Required Bindings:
023     * <ul>
024     * <li>objectsToReport - the array of Enterprise Objects to be displayed in the component
025     * <li>attributeKeyPaths - the ordered list of attribute key paths that determines the attributes to be displayed for each Enterprise Object
026     * <li>attributesToAlwaysDisplay - the list of attribute key paths that needs to be displayed for each row
027     * <li>comparators - the ordered list of compare method names that will be used for sorting its corresponding attribute key paths
028     * <li>lastAttributeToExpand - tells the component the last attribute key path that can be expanded from the list of attributeKeyPaths
029     * <li>titles - the ordered list of titles to be displayed as the titles for each column
030     * <li>onAttributeClick - bind this to the client method to be called when an attribute or item is clicked.  This method is performed before the attribute is expanded or collapsed. In order for this method to be invoked properly, signature of method must be: <code>public WOComponent aMethod(String clickedAttribute, DataGroup aDataGroup)</code>
031     * <li>onTitleClick - bind this to the client method to be called when a title is clicked. In order for this method to be invoked properly, signature of method must be: <code>public WOComponent aMethod(String clickedTitle)</code>
032     * </ul>
033     *
034     * Note: It is assumed by the component that the list of titles, attributeKeyPaths and comparators are parallel with each column displayed.
035     *
036     * Optional Bindings:
037     * <ul>
038     * <li>titleStyle - bind to the Cascading Style Sheet (CSS) class to edit the display style for each title
039     * <li>oddRowStyle - bind to the CSS class to edit the display style for all odd rows in the list
040     * <li>evenRowStyle - bind to the CSS class to edit the display style for all even rows in the list
041     * <li>oddColumnStyle - bind to the CSS class to edit the display style for all odd columns in the list
042     * <li>evenColumnStyle - bind to the CSS class to edit the display style for all odd columns in the list
043     * </ul>
044     *

046     * This software is published under the terms of the Educational Community License (ECL) version 1.0,
047     * a copy of which has been included with this distribution in the LICENSE.TXT file.
048     * @version $Revision: 4$
049     */

077        /**
078         * This method is overridden to generate the initial data groups before this component is displayed.
079         */
080        public void appendToResponse(WOResponse aResponse, WOContext aContext)
081        {
082            /** require
083            [has_title_for_each_attribute] attributeKeyPaths().count() == titles().count();
084            [has_comparator_for_each_attribute] attributeKeyPaths().count() == comparators().count(); **/
085    
086            //These conditions needs to be checked to avoid re-processing of the same data. 
087            if ((! objectsToReport().isEqualToArray(currentData)) || (! attributeKeyPaths().isEqualToArray(currentAttributeKeyPaths)))
088            {
089                //Since masterDataGroup does not have any group attribute, then a dummy attribute must be added to attributeKeyPaths() and comparators() when creating it  
090                String dummyAttribute = "$dummyAttribute%";
091    
092                NSMutableArray attributeKeyPathsWithDummy = new NSMutableArray(dummyAttribute);
093                attributeKeyPathsWithDummy.addObjectsFromArray(attributeKeyPaths());
094    
095                NSMutableArray comparatorsWithDummy = new NSMutableArray(dummyAttribute);
096                comparatorsWithDummy.addObjectsFromArray(comparators());
097    
098                setMasterDataGroup(new DataGroup(dummyAttribute,
099                                                 objectsToReport(),
100                                                 attributeKeyPathsWithDummy,
101                                                 comparatorsWithDummy));
102    
103                //Always show top-level
104                masterDataGroup().toggleExpansion();
105    
106                currentData = objectsToReport();
107                currentAttributeKeyPaths = attributeKeyPaths();
108            }
109            super.appendToResponse(aResponse, aContext);
110        }

278        /**
279         * Returns the corresponding value of attributeKeyPath from aDataGroup.
280         *
281         * @return the corresponding value of attributeKeyPath from aDataGroup.
282         */
283        public Object currentAttribute()
284        {
285            Object theObject = null;
286    
287            if (attributeKeyPath.equals(aDataGroup.groupAttributeKeyPath()) || attributesToAlwaysDisplay().containsObject(attributeKeyPath))
288            {
289                theObject = currentAttributeValue();
290            }
291            return theObject;
292        }
293    
294    
295    
296        /**
297         * Returns the corresponding value of attributeKeyPath from aDataGroup.
298         *
299         * @return the corresponding value of attributeKeyPath from aDataGroup.
300         */
301        public Object currentAttributeValue()
302        {
303            Object theObject = aDataGroup.valueForKeyPath(attributeKeyPath);
304    
305            //In case valueForKeyPath() returns EONullValue.nullValue()
306            if (theObject.equals(NSKeyValueCoding.NullValue))
307            {
308                theObject = null;
309            }
310    
311            return theObject;
312        }

316        /**
317         * Toggles the expansion of a data group.
318         *
319         * @return the parent's WOComponent
320         */
321        public com.webobjects.appserver.WOComponent toggleExpansion()
322        {
323            if (shouldExpandAll())
324            {
325                aDataGroup.expandAll();    
326            }
327            else
328            {
329                aDataGroup.toggleExpansion();            
330            }
331    
332            return parent();
333    
334            /** ensure [valid_result] Result != null; **/
335        }

372        /**
373         * Invokes the method bound to <code>onTitleClick()</code>. 
374         *
375         * @return the WOComponent returned when method is invoked, it returns the parent component when the method invoked raises an exception.
376         */
377        public com.webobjects.appserver.WOComponent performTitleAction()
378        {
379            com.webobjects.appserver.WOComponent nextPage = parent().context().page();
380            
381            try
382            {
383                nextPage = (com.webobjects.appserver.WOComponent) NSSelector.invoke(onTitleClick(), String.class, parent(), aTitle);
384            }
385            catch (Exception e) {}
386    
387            return nextPage;
388        }
389    
390    
391    
392        /**
393         * Invokes the method bound to <code>onAttributeClick()</code>.
394         *
395         * @return the WOComponent returned when method is invoked, it returns the parent component when the method invoked raises an exception.
396         */
397        public com.webobjects.appserver.WOComponent performAttributeAction()
398        {
399            com.webobjects.appserver.WOComponent nextPage = parent().context().page();
400    
401            try
402            {
403                nextPage = (com.webobjects.appserver.WOComponent) NSSelector.invoke(onAttributeClick(),
404                                                                                    String.class,
405                                                                                    DataGroup.class,
406                                                                                    parent(),
407                                                                                    attributeKeyPath,
408                                                                                    aDataGroup);
409            }
410            catch (Exception e) {}
411    
412            //Only toggle expansion if the method onAttributeClick() returns the same page and is not the last attribute
413            if ((nextPage.equals(parent().context().page())) &&
414                (attributeKeyPaths().indexOfObject(attributeKeyPath) <= attributeKeyPaths().indexOfObject(lastAttributeToExpand())))
415            {
416                nextPage = toggleExpansion();
417            }
418            
419            return nextPage;
420        }
