020    /**
021     * <p>
022     * Support class to handle SQL generation for InSubqueryQualifier instances.
023     * </p>
024     * 
025     * <p>
026     * While the implementation of the qualifier is straightforward, the
027     * implementation of the support class is not. Thing get tricky when
028     * relationships are involved. But the real fun comes with relationships based
029     * on multiple joins. In this event the support class actually creates an
030     * instance of yet another custom qualifier to serve as schema based qualifier.
031     * This qualifier in turn has a support class of its own. The reason this
032     * qualifier is not exposed outside the support class is that its implementation
033     * provides only for the behavior needed in the context of
034     * InSubQueryQualifierSupport. Most notably this qualifier does not support
035     * in-memory evaluation.
036     * </p>
037     * 
038     * <p>
039     * While the InSubQueryQualifier has been tested in common situations, it has
040     * not actually been tested for use on multiple join relationships. As per the
041     * disclaimer: use at your own risk.
042     * </p>
043     * 
044     * <p>
045     * This sample code is provided for educational purposes. It is mainly to be
046     * considered a source of information and a way to spark off discussion. I make
047     * no warranty, expressed or implied, about the quality of this code or its
048     * usefulness in any particular situation. Use this code or any code based on it
049     * at your own risk. Basically, enjoy the read, but don't blame me for anything.
050     * </p>
051     * 
052     * @author Pierre Bernard http://homepage.mac.com/i_love_my/webobjects.html
053     */
054    public class InSubqueryQualifierSupport extends QualifierGenerationSupport

065             * (non-Javadoc)
066             * 
067             * @see com.webobjects.eoaccess.EOQualifierSQLGeneration.Support#sqlStringForSQLExpression(com.webobjects.eocontrol.EOQualifier,
068             *      com.webobjects.eoaccess.EOSQLExpression)
069             */
070            public String sqlStringForSQLExpression(EOQualifier qualifier, EOSQLExpression expression)
071            {
072                    InSubqueryQualifier isQualifier = (InSubqueryQualifier) qualifier;
073                    ArrayInSubqueryQualifier aisQualifier = new ArrayInSubqueryQualifier(new NSArray(
074                                    isQualifier.keyPath()), isQualifier.entityName(), new NSArray(isQualifier
075                                    .attributePath()), isQualifier.subQualifier());
076                    EOQualifierSQLGeneration.Support support = supportForClass(ArrayInSubqueryQualifier.class);
077    
078                    return support.sqlStringForSQLExpression(aisQualifier, expression);
079            }

081            /*
082             * (non-Javadoc)
083             * 
084             * @see com.webobjects.eoaccess.EOQualifierSQLGeneration.Support#schemaBasedQualifierWithRootEntity(com.webobjects.eocontrol.EOQualifier,
085             *      com.webobjects.eoaccess.EOEntity)
086             */
087            public EOQualifier schemaBasedQualifierWithRootEntity(EOQualifier qualifier, EOEntity entity)
088            {
089                    InSubqueryQualifier isQualifier = (InSubqueryQualifier) qualifier;
090                    String keyPath = isQualifier.keyPath();
091    
092                    if (isQualifier.entityName() == null) {
093                            EORelationship relationship = relationshipForPath(entity, keyPath);
094    
095                            if (relationship.isFlattened()) {
096                                    relationship = (EORelationship) relationship.componentRelationships().lastObject();
097                            }
098    
099                            EOEntity destEntity = relationship.destinationEntity();
100                            NSArray joins = relationship.joins();
101                            int joinCount = joins.count();
102                            NSMutableArray srcAttributes = new NSMutableArray(joinCount);
103                            NSMutableArray destAttrNames = new NSMutableArray(joinCount);
104                            String path = Qualifier.allButLastPathComponent(keyPath);
105    
106                            for (int i = joinCount - 1; i >= 0; i--) {
107                                    EOJoin join = (EOJoin) joins.objectAtIndex(i);
108                                    String srcAttributeName = join.sourceAttribute().name();
109                                    String destAttributeName = join.destinationAttribute().name();
110    
111                                    srcAttributes.addObject(optimizeQualifierKeyPath(entity, path, srcAttributeName));
112                                    destAttrNames.addObject(destAttributeName);
113                            }
114    
115                            EOQualifier subQualifier = isQualifier.subQualifier();
116                            EOQualifierSQLGeneration.Support support = supportForClass(subQualifier.getClass());
117                            EOQualifier schemaBasedSubQualifier = support.schemaBasedQualifierWithRootEntity(
118                                            subQualifier, destEntity);
119    
120                            return new ArrayInSubqueryQualifier(srcAttributes, destEntity.name(), destAttrNames,
121                                            schemaBasedSubQualifier);
122                    }
123                    else {
124                            EOEntity destEntity = entity.model().modelGroup().entityNamed(isQualifier.entityName());
125                            String attributePath = isQualifier.attributePath();
126    
127                            EOQualifier subQualifier = isQualifier.subQualifier();
128                            EOQualifierSQLGeneration.Support support = supportForClass(subQualifier.getClass());
129                            EOQualifier schemaBasedSubQualifier = support.schemaBasedQualifierWithRootEntity(
130                                            subQualifier, destEntity);
131    
132                            return new ArrayInSubqueryQualifier(new NSArray(optimizeQualifierKeyPath(entity,
133                                            Qualifier.allButLastPathComponent(keyPath), Qualifier
134                                                            .lastPathComponent(keyPath))), isQualifier.entityName(), new NSArray(
135                                            optimizeQualifierKeyPath(destEntity, Qualifier
136                                                            .allButLastPathComponent(attributePath), Qualifier
137                                                            .lastPathComponent(attributePath))), schemaBasedSubQualifier);
138                    }
139            }

187                     * @see com.webobjects.eocontrol.EOQualifier#qualifierWithBindings(com.webobjects.foundation.NSDictionary,
188                     *      boolean)
189                     */
190                    public EOQualifier qualifierWithBindings(NSDictionary bindings, boolean requiresAll)
191                    {
192                            EOQualifier qualifier = subQualifier();
193    
194                            if (qualifier != null) {
195                                    EOQualifier boundQualifier = qualifier.qualifierWithBindings(bindings, requiresAll);
196    
197                                    if (qualifier != boundQualifier) {
198                                            return new ArrayInSubqueryQualifier(keyPathArray(), entityName(),
199                                                            attributePaths(), boundQualifier);
200                                    }
201                            }
202    
203                            return this;
204                    }

210                     */
211                    public void validateKeysWithRootClassDescription(EOClassDescription classDescription)
212                    {
213                            Enumeration keys = keyPathArray().objectEnumerator();
214    
215                            while (keys.hasMoreElements()) {
216                                    Qualifier.validateKeyPathWithRootClassDescription((String) keys.nextElement(),
217                                                    classDescription);
218                            }
219    
220                            if (subQualifier() != null) {
221                                    EOClassDescription subDescription = EOClassDescription
222                                                    .classDescriptionForEntityName(entityName());
223    
224                                    subQualifier().validateKeysWithRootClassDescription(subDescription);
225                            }
226                    }

285            public static class ArrayInSubqueryQualifierSupport extends QualifierGenerationSupport
286            {
287                    // Public instance methods
288    
289                    /*
290                     * (non-Javadoc)
291                     * 
292                     * @see com.webobjects.eoaccess.EOQualifierSQLGeneration.Support#sqlStringForSQLExpression(com.webobjects.eocontrol.EOQualifier,
293                     *      com.webobjects.eoaccess.EOSQLExpression)
294                     */
295                    public String sqlStringForSQLExpression(EOQualifier qualifier, EOSQLExpression expression)
296                    {
297                            ArrayInSubqueryQualifier aisQualifier = (ArrayInSubqueryQualifier) qualifier;
298                            EOEntity entity = expression.entity();
299                            NSArray keyPathArray = aisQualifier.keyPathArray();
300                            int keyPathCount = keyPathArray.count();
301                            boolean hasMultipleElements = (keyPathCount > 1);
302                            StringBuffer buffer = new StringBuffer();
303    
304                            if (hasMultipleElements) {
305                                    buffer.append("(");
306                            }
307    
308                            for (int i = 0; i < keyPathCount; i++) {
309                                    String keyPath = (String) keyPathArray.objectAtIndex(i);
310                                    String attributeString = expression.sqlStringForAttributeNamed(keyPath);
311    
312                                    if (attributeString == null) {
313                                            throw new IllegalStateException(
314                                                            "sqlStringForKeyValueQualifier: attempt to generate SQL for "
315                                                                            + aisQualifier.getClass().getName() + " " + aisQualifier
316                                                                            + " failed because attribute identified by key '" + keyPath
317                                                                            + "' was not reachable from from entity '" + entity.name()
318                                                                            + "'");
319                                    }
320    
321                                    if (i > 0) {
322                                            buffer.append(", ");
323                                    }
324    
325                                    buffer.append(expression.formatSQLString(attributeString, attributeForPath(entity,
326                                                    keyPath).readFormat()));
327                            }
328    
329                            if (hasMultipleElements) {
330                                    buffer.append(")");
331                            }
332    
333                            buffer.append(" IN (");
334    
335                            EOEntity subEntity = entity.model().modelGroup().entityNamed(aisQualifier.entityName());
336                            EOSQLExpression subExpression = expressionForEntity(subEntity);
337                            EOFetchSpecification subFetch = new EOFetchSpecification(subEntity.name(), aisQualifier
338                                            .subQualifier(), null);
339                            NSArray attributePaths = aisQualifier.attributePaths();
340                            int aCount = attributePaths.count();
341                            NSMutableArray attributes = new NSMutableArray(aCount);
342    
343                            for (int a = 0; a < aCount; a++) {
344                                    attributes.addObject(attributeForPath(subEntity, (String) attributePaths
345                                                    .objectAtIndex(a)));
346                            }
347    
348                            StringBuffer subBuffer = new StringBuffer();
349    
350                            subExpression.prepareSelectExpressionWithAttributes(attributes, false, subFetch);
351    
352                            subBuffer.append("SELECT ");
353                            subBuffer.append(subExpression.listString());
354                            subBuffer.append(" FROM ");
355                            subBuffer.append(subExpression.tableListWithRootEntity(subEntity));
356    
357                            boolean hasWhereClause = (subExpression.whereClauseString() != null)
358                                            && (subExpression.whereClauseString().length() > 0);
359    
360                            if (hasWhereClause) {
361                                    subBuffer.append(" WHERE ");
362                                    subBuffer.append(subExpression.whereClauseString());
363                            }
364    
365                            if ((subExpression.joinClauseString() != null)
366                                            && (subExpression.joinClauseString().length() > 0)) {
367                                    if (hasWhereClause) {
368                                            subBuffer.append(" AND ");
369                                    }
370    
371                                    subBuffer.append(subExpression.joinClauseString());
372                            }
373    
374                            replaceTableAliasesInExpressionBuffer(subBuffer, subExpression);
375                            buffer.append(subBuffer);
376                            buffer.append(")");
377                            Enumeration bindVariables = subExpression.bindVariableDictionaries().objectEnumerator();
378                            while (bindVariables.hasMoreElements()) {
379                                    expression.addBindVariableDictionary((NSDictionary) bindVariables.nextElement());
380                            }
381    
382                            return buffer.toString();
383                    }

399                     * @see com.webobjects.eoaccess.EOQualifierSQLGeneration.Support#qualifierMigratedFromEntityRelationshipPath(com.webobjects.eocontrol.EOQualifier,
400                     *      com.webobjects.eoaccess.EOEntity, java.lang.String)
401                     */
402                    public EOQualifier qualifierMigratedFromEntityRelationshipPath(EOQualifier qualifier,
403                                    EOEntity entity, String relationshipPath)
404                    {
405                            ArrayInSubqueryQualifier aisQualifier = (ArrayInSubqueryQualifier) qualifier;
406                            NSArray keyPathArray = aisQualifier.keyPathArray();
407                            int keyPathCount = keyPathArray.count();
408                            NSMutableArray migratedKeyPathArray = new NSMutableArray(keyPathCount);
409                            for (int i = 0; i < keyPathCount; i++) {
410                                    String keyPath = (String) keyPathArray.objectAtIndex(i);
411                                    migratedKeyPathArray.addObject(translateKeyAcrossRelationshipPath(keyPath,
412                                                    relationshipPath, entity));
413                            }
414    
415                            return new ArrayInSubqueryQualifier(migratedKeyPathArray, aisQualifier.entityName(),
416                                            aisQualifier.attributePaths(), aisQualifier.subQualifier());
417                    }
