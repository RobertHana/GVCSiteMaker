082        /**
083         * Test graph equivalence, regardless of where copy is started.
084         */
085        public void testGraphEquivalence()
086        {
087            NSMutableDictionary copyCache = new NSMutableDictionary();
088            rootObject.copy(copyCache);
089    
090            NSMutableDictionary copyCache2 = new NSMutableDictionary();
091            ownedObject1.copy(copyCache2);
092    
093            assertTrue("number of keys matches", copyCache.count() == copyCache2.count());
094            assertTrue("Global IDs matches", new NSSet(copyCache.allKeys()).equals(new NSSet(copyCache2.allKeys())));
095            
096            editingContext().revert();
097        }

102         * Test that all objects were copied
103         */
104        public void testAllObjectCopied()
105        {
106            NSMutableDictionary copyCache = new NSMutableDictionary();
107            rootObject.copy(copyCache);
108    
109            assertTrue("root object copied", copyCache.objectForKey(editingContext().globalIDForObject(rootObject)) != null);
110            assertTrue("first owned object copied", copyCache.objectForKey(editingContext().globalIDForObject(ownedObject1)) != null);
111            assertTrue("second owned object copied", copyCache.objectForKey(editingContext().globalIDForObject(ownedObject2)) != null);
112            assertTrue("first un-owned object copied", copyCache.objectForKey(editingContext().globalIDForObject(unOwnedObject1)) != null);
113            assertTrue("second un-owned object copied", copyCache.objectForKey(editingContext().globalIDForObject(unOwnedObject2)) != null);
114            editingContext().revert();
115        }

120         * Test that objects were copied as designated (copied or by reference)
121         */
122        public void testObjectsCopiedCorrectly()
123        {
124            NSMutableDictionary copyCache = new NSMutableDictionary();
125            rootObject.copy(copyCache);
126    
127            assertTrue("root object copied", copyCache.objectForKey(editingContext().globalIDForObject(rootObject)) != rootObject);
128            assertTrue("first owned object copied", copyCache.objectForKey(editingContext().globalIDForObject(ownedObject1)) != ownedObject1);
129            assertTrue("second owned object copied", copyCache.objectForKey(editingContext().globalIDForObject(ownedObject2)) != ownedObject2);
130            assertTrue("first un-owned object referenced", copyCache.objectForKey(editingContext().globalIDForObject(unOwnedObject1)) == unOwnedObject1);
131            assertTrue("second un-owned object referenced", copyCache.objectForKey(editingContext().globalIDForObject(unOwnedObject2)) == unOwnedObject2);
132            editingContext().revert();
133        }

137        /**
138         * Test that the relationships are not duplicated or ommited.
139         */
140        public void testRelationshipsCopiedCorrectly()
141        {
142            NSMutableDictionary copyCache = new NSMutableDictionary();
143            RootObject copyOfRoot = (RootObject) rootObject.copy(copyCache);
144    
145    
146            Enumeration relationshipEnumerator = rootObject.toManyRelationshipKeys().objectEnumerator();
147            while (relationshipEnumerator.hasMoreElements())
148            {
149                String relationshipName = (String)relationshipEnumerator.nextElement();
150                NSArray originalObjects = (NSArray)rootObject.valueForKey(relationshipName);
151                NSArray copiedObjects = (NSArray)copyOfRoot.valueForKey(relationshipName);
152    
153                assertTrue("to many relationship has same # of objects: " + relationshipName,
154                           originalObjects.count() == copiedObjects.count());
155            }
156    
157            relationshipEnumerator =  rootObject.toOneRelationshipKeys().objectEnumerator();
158            while (relationshipEnumerator.hasMoreElements())
159            {
160                String relationshipName = (String)relationshipEnumerator.nextElement();
161                EOEnterpriseObject originalObject = (EOEnterpriseObject)rootObject.valueForKey(relationshipName);
162                EOEnterpriseObject copiedObject = (EOEnterpriseObject)copyOfRoot.valueForKey(relationshipName);
163    
164                assertTrue("to-one relationship has compatible objects: " + relationshipName,
165                           ((originalObject == null) &&  (copiedObject == null) ||
166                            (originalObject != null) &&  (copiedObject != null)));
167            }
168    
169            editingContext().revert();
170        }

174        /**
175         * Test that the relationships are not duplicated or ommited.
176         */
177        public void testAttributesCopiedCorrectly()
178        {
179            NSMutableDictionary copyCache = new NSMutableDictionary();
180            rootObject.setAValue(null);     // Check that null values which are set to non-null in awakeFromInsertion are properly copied.
181            RootObject copyOfRoot = (RootObject) rootObject.copy(copyCache);
182    
183    
184            Enumeration attributeEnumerator = rootObject.attributeKeys().objectEnumerator();
185            while (attributeEnumerator.hasMoreElements())
186            {
187                String attributeName = (String)attributeEnumerator.nextElement();
188    
189                Object originalValue = rootObject.valueForKey(attributeName);
190                Object copiedValue = copyOfRoot.valueForKey(attributeName);
191    
192                if ( ! attributeName.equals("oid"))
193                {
194                    assertTrue("attribute has same value: " + attributeName,
195                               (originalValue == copiedValue) || originalValue.equals(copiedValue));
196                }
197                else
198                {
199                    assertTrue("exposed PK not copied",
200                               (originalValue != null) && (copiedValue == null));
201                }
202            }
203    
204            editingContext().revert();
205        }

209        /**
210         * Test that copied objects inserted correctly.
211         */
212        public void testCopiedObjectsInserted()
213        {
214            NSMutableDictionary copyCache = new NSMutableDictionary();
215            rootObject.copy(copyCache);
216            EOEditingContext ec = editingContext();
217            NSArray insertedObjects = ec.insertedObjects();
218    
219            assertTrue("root object inserted",
220                       insertedObjects.containsObject(copyCache.objectForKey(ec.globalIDForObject(rootObject))));
221            assertTrue("first owned object inserted",       insertedObjects.containsObject(copyCache.objectForKey(ec.globalIDForObject(ownedObject1))));
222            assertTrue("second owned object inserted",      insertedObjects.containsObject(copyCache.objectForKey(ec.globalIDForObject(ownedObject2))));
223    
224            assertTrue("first un-owned object not inserted",
225                       ! insertedObjects.containsObject(copyCache.objectForKey(ec.globalIDForObject(unOwnedObject1))));
226            assertTrue("second un-owned object not inserted",
227                       ! insertedObjects.containsObject(copyCache.objectForKey(ec.globalIDForObject(unOwnedObject2))));
228            
229            editingContext().revert();
230        }
