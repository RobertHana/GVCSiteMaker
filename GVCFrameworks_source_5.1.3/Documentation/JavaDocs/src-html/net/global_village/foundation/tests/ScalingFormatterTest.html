018    public class ScalingFormatterTest extends TestCase
019    {

021        /**
022         * Constructor.
023         * @param name of test
024         */
025        public ScalingFormatterTest(String name)
026        {
027            super(name);
028        }
029    
030    
031    
032        /*
033         * Test for StringBuffer format(Object, StringBuffer, FieldPosition)
034         */
035        public void testFormatObjectStringBufferFieldPosition()
036        {
037            Format formatter = new ScalingFormatter(.125, IntegerFormatter.Formatter);
038    
039            StringBuffer sb = formatter.format(new BigDecimal(800), new StringBuffer(), new FieldPosition(0));
040            assertEquals("100", sb.toString());
041    
042            sb = formatter.format(new Integer(800), new StringBuffer(), new FieldPosition(0));
043            assertEquals("100", sb.toString());
044    
045            sb = formatter.format(new Double(800), new StringBuffer(), new FieldPosition(0));
046            assertEquals("100", sb.toString());
047    
048            formatter = new ScalingFormatter(8, IntegerFormatter.Formatter);
049    
050            sb = formatter.format(new BigDecimal(800), new StringBuffer(), new FieldPosition(0));
051            assertEquals("6400", sb.toString());
052    
053            sb = formatter.format(new Integer(800), new StringBuffer(), new FieldPosition(0));
054            assertEquals("6400", sb.toString());
055    
056            sb = formatter.format(new Double(800), new StringBuffer(), new FieldPosition(0));
057            assertEquals("6400", sb.toString());
058        }
059    
060    
061    
062        /*
063         * Test for java.lang.Object parseObject(java.lang.String)
064         */
065        public void testParseObjectString()
066        {
067            Format formatter = new ScalingFormatter(.125, IntegerFormatter.Formatter);
068            try
069            {
070                assertTrue(formatter.parseObject("100").equals(new Integer(800)));
071    
072                formatter = new ScalingFormatter(8, IntegerFormatter.Formatter);
073                assertTrue(formatter.parseObject("6400").equals(new Integer(800)));
074            }
075            catch (ParseException e)
076            {
077                fail(e.getMessage());
078            }
079        }
