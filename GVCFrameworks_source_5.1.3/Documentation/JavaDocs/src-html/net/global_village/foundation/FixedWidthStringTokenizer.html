008    /**
009     * A String tokenizer that accepts a string of an array of widths and returns the string separated into strings of width lengths.

013     * a copy of which has been included with this distribution in the LICENSE.TXT file.
014     * @version $Revision: 7$
015     */
016    public class FixedWidthStringTokenizer extends AbstractStringTokenizer
017    {
018        protected NSArray tokenWidths;
019        protected Enumeration widthEnumerator;
020    
021    
022        /**
023         * Designated constructor.
024         *
025         * @param str the string to tokenize
026         * @param widths the widths of the tokens
027         * @param names the names of the tokens
028         */
029        public FixedWidthStringTokenizer(String str, NSArray widths, NSArray names)
030        {
031            super(str, names);
032            /** require
033            [valid_str_param] str != null;
034            [valid_widths_param] widths != null;
035            [widths_equals_string_length] ((Number)widths.valueForKeyPath("@sum")).intValue() == str.length();
036            [has_name_for_each_width] names != null ? widths.count() == names.count() : true; **/
037            JassAdditions.pre("FixedWidthStringTokenizer", "<init>", Collection.collectionContainsInstancesOf(widths, Integer.class));
038    
039            tokenWidths = widths;
040            widthEnumerator = widths.objectEnumerator();
041        }
042    
043    
044    
045        /**
046         * Equivilant to <code>FixedWidthStringTokenizer(str, widths, null)</code>.
047         *
048         * @param str the string to tokenize
049         * @param widths the widths of the tokens
050         */
051        public FixedWidthStringTokenizer(String str, NSArray widths)
052        {
053            this(str, widths, null);
054            /** require
055            [valid_str_param] str != null;
056            [valid_widths_param] widths != null;
057            [widths_equals_string_length] ((Number)widths.valueForKeyPath("@sum")).intValue() == str.length(); **/
058            JassAdditions.pre("FixedWidthStringTokenizer", "<init>", Collection.collectionContainsInstancesOf(widths, Integer.class));
059        }
060    
061    
062        
063        /**
064         * Returns the next token as an Object.
065         *
066         * @return the next token
067         */
068        public Object nextElement()
069        {
070            /** require [has_more_elements] hasMoreElements(); **/
071            
072            int tokenLength = ((Integer)widthEnumerator.nextElement()).intValue();
073    
074            if (nameEnumerator != null)
075            {
076                nameOfCurrentToken = (String)nameEnumerator.nextElement();
077            }
078    
079            Object token = stringToTokenize.substring(mark, mark + tokenLength);
080            mark += tokenLength;
081            return token;
082    
083            /** ensure [valid_result] Result != null; **/
084        }
085    
086    
087        /** invariant
088        [has_string_to_tokenize] stringToTokenize != null;
089        [has_token_widths] tokenWidths != null;
090        [has_width_enumerator] widthEnumerator != null;
091        [mark_is_consistant] mark <= stringToTokenize.length(); **/
092    
093    }
