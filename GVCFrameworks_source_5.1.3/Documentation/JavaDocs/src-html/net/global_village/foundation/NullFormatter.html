001    package net.global_village.foundation;
002    
003    import java.text.*;
004    
005    
006    /**
007     * A null object formatter that just returns its input.  Useful for when the API needs a formatter
008     * but there is no formatting to do.
009     *
010     * @author Copyright (c) 2001-2005  Global Village Consulting, Inc.  All rights reserved.
011     * This software is published under the terms of the Educational Community License (ECL) version 1.0,
012     * a copy of which has been included with this distribution in the LICENSE.TXT file.
013     */
014    public class NullFormatter extends Format
015    {
016        
017        /** Convenience instance.  **/
018        public static final NullFormatter Formatter = new NullFormatter();
019        
020    
021    
022        /**
023         * Designated constructor.
024         */
025        public NullFormatter()
026        {
027            super();
028        }
029    
030    
031    
032        /* (non-Javadoc)
033         * @see java.text.Format#format(java.lang.Object, java.lang.StringBuffer, java.text.FieldPosition)
034         */
035        public StringBuffer format(Object object, StringBuffer stringBuffer, FieldPosition fieldPosition)
036        {
037            stringBuffer.append(object);
038            
039            return stringBuffer;
040        }
041    
042    
043    
044        /* (non-Javadoc)
045         * @see java.text.Format#parseObject(java.lang.String, java.text.ParsePosition)
046         */
047        public Object parseObject(String string, ParsePosition position)
048        {
049            position.setIndex(string.length());
050            return string;
051        }
052        
053    }
