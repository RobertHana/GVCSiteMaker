113         *
114         * @param theObject the object to be evaluated

118        public static EOProperty propertyWithName(EOEnterpriseObject theObject,
119                                                  String propertyName)
120        {
121            /** require
122            [valid_theObject_param] theObject != null;
123            [valid_propertyName_param] propertyName != null;
124            [object_is_entity] entityForSelf(theObject) != null;
125            [property_exists] EOObject.entityForSelf(theObject).attributeNamed(propertyName) != null ||
126                              EOObject.entityForSelf(theObject).relationshipNamed(propertyName) != null;            **/
127    
128            EOProperty theProperty = EOObject.entityForSelf(theObject).attributeNamed(propertyName);
129    
130            if (theProperty == null)
131            {
132                theProperty = EOObject.entityForSelf(theObject).relationshipNamed(propertyName);
133            }
134    
135            return theProperty;
136    
137            /** ensure [valid_result] Result != null; **/
138        }

142        /**
143         * Private method to support <code>keyPathHasEntity()</code> and <code>entityFromKeyPath()</code>
144         *
145         * @param theObject the object to be evaluated
146         * @param aKeyPath the key path of the entity
147         * @return the EOEntity associated with the object and key path
148         */
149        protected static EOEntity _findEntityFromKeyPath(Object theObject,
150                                                       String aKeyPath)
151        {
152            /** require
153            [valid_theObject_param] theObject != null;
154            [valid_aKeyPath_param] aKeyPath != null;
155            [keypath_is_valid] StringAdditions.isValidPropertyKeyPath(aKeyPath); **/
156    
157            EOEntity entity = null;
158    
159            String objectKeyPath = StringAdditions.objectKeyPathFromKeyPath(aKeyPath);
160            // valueForKeyPath raises if keypath not found
161            try
162            {
163                EOEnterpriseObject object = (EOEnterpriseObject)NSKeyValueCodingAdditions.Utility.valueForKeyPath(theObject, objectKeyPath);
164                entity = entityForSelf(object);
165            }
166            catch (Exception e)
167            {
168                entity = null;
169            }
170    
171            return entity;
172        }

398        /**
399         * Detecting WO 5 dummy fault EOs.  The code below is a translation of code posted here:
400         * From http://wodeveloper.com/omniLists/eof/2001/November/msg00023.html
401         *
402         * As of WO4.5, EOF resolves faults for objects that don't exist in the database to a dummy object with mostly empty attributes (except those
403        that are set in -init), whereas earlier versions of EOF threw a _fireFault exception.  This method detects whether the receiver is such a dummy object.
404         *
405         *  A distinguishing feature of these dummy EOs is that they don't have a corresponding database context snapshot, so we use this to detect its dummy-ness.  This seems more robust than checking if all or most attributes are empty, since these can be changed by client code.
406         *
407         *  Dummy fault EOs should be removed from the editingContext (using forgetObject) before it is invalidated (invalidateAllObjects or invalidateObjectWithGlobalID), otherwise an unrecoverable "decrementSnapshotCountForGlobalID - unable to decrement snapshot count for object with global ID" exception is thrown by EOF.<br><br>
408         *
409         * See also: http://www.omnigroup.com/mailman/archive/eof/2001-May/001988.html andfollowing.
410         */
411        public static boolean isDummyFaultEO(EOEnterpriseObject anObject)
412        {
413            /** require [valid_param] anObject != null;  **/
414            JassAdditions.pre("EOObject", "isDummyFaultEO [object_in_ec]", anObject.editingContext() != null);
415            JassAdditions.pre("EOObject", "isDummyFaultEO [object_has_global_id]", anObject.editingContext().globalIDForObject(anObject) != null);
416    
417            boolean isDummyFaultEO = false;
418    
419            // This method will fail if the object is still a fault.  We fire the fault so that the correct result is returned.
420            if (anObject.isFault())
421            {
422                try
423                {
424                    anObject.willRead();
425                }
426                catch (EOObjectNotAvailableException e)
427                {
428                    // The DatabaseContextDeletgate restores the "throw on failure to fetch" functionality and throws EOObjectNotAvailableException
429                    // That is caught here and true returned so that this method will return the expected result even when this delegate is used
430                    return true;
431                }
432            }
433    
434            EOEditingContext ec = anObject.editingContext();
435            EOGlobalID globalID = ec.globalIDForObject(anObject);
436    
437            // NB. objects with temporary globalIDs legitimately have no DB snapshots, since these are by definition not yet saved to the database.
438            if (! globalID.isTemporary())
439            {
440                // Find the EODatabaseContext instance associated with anObject, or null if no databaseContext association can be found.
441                EOObjectStoreCoordinator rootStore = (EOObjectStoreCoordinator)ec.rootObjectStore();
442                rootStore.lock();
443                try
444                {
445                    EODatabaseContext dbContext = (EODatabaseContext)rootStore.objectStoreForObject(anObject);
446                    if (dbContext == null)
447                    {
448                        throw new RuntimeException("Could not locate EODatabaseContext for object '" +
449                                                   anObject + "' in editingContext: " + ec);
450                    }
451    
452                    isDummyFaultEO = (dbContext.snapshotForGlobalID(globalID) == null);
453                }
454                finally
455                {
456                    rootStore.unlock();
457                }
458            }
459    
460            return isDummyFaultEO;
461        }

467        /**
468         * Forces this object to be re-read from the database; the snapshot and all editing contexts
469         * holding this object are updated. Merging or overwriting of the changes is then handled by
470         * the editing context delegate. This method takes no action if the object is pending insertion.
471         *
472         * @param anObject the object to be evaluated
473         * @throws EOObjectNotAvailableException if the object being refreshed has been deleted
474         */
475        public static void refreshObject(EOEnterpriseObject anObject) throws EOObjectNotAvailableException
476        {
477            /** require [valid_theObject_param] anObject != null;
478                        [in_ec] anObject.editingContext() != null;            **/
479    
480            // We can't just call editingContext().refreshObject() or editingContext().refaultObject() the object here.
481            // That will result in fresh data only if the object is in a single editing context.
482            // This seems like the safest way to get fresh data and inform all of the other
483            // editing contexts that they need to update.
484            EOEditingContext ec = anObject.editingContext();
485            if ( ! ec.insertedObjects().containsObject(anObject))
486            {
487                EOQualifier thisObject = EOUtilities.qualifierForEnterpriseObject(ec, anObject);
488                EOFetchSpecification fetchSpec = new EOFetchSpecification(entityForSelf(anObject).name(), thisObject, NSArray.EmptyArray);
489                fetchSpec.setRefreshesRefetchedObjects(true);
490                NSArray results = ec.objectsWithFetchSpecification(fetchSpec);
491    
492                // If we did not fetch any objects, then it has been deleted.  This EOF stack does not know
493                // this yet, so we need to invalidate the object and then fire the fault to trigger an exception
494                // (if the dbCtxt delegate throws), otherwise anObject becomes a Dummy Fault EO
495                if (results.count() == 0)
496                {
497                    ec.invalidateObjectsWithGlobalIDs(new NSArray(globalID(anObject)));
498                    anObject.willRead();
499                }
500            }
501        }

505        /**
506         * Reverts any changes in this object's values back to the state last fetched.  This will not refetch the
507         * data from the database.  This method takes no action if the object is pending insertion or deletion.
508         *
509         * @param anObject the object to be evaluated
510         */
511        public static void revertToSaved(EOEnterpriseObject anObject)
512        {
513            /** require [valid_theObject_param] anObject != null;
514                        [in_ec] anObject.editingContext() != null;            **/
515            EOEditingContext ec = anObject.editingContext();
516            if ( ! (ec.insertedObjects().containsObject(anObject) || ec.deletedObjects().containsObject(anObject)))
517            {
518                // Calling editingContext().refaultObject(this) will also works but may result in a trip to the database
519                // This is less work
520                anObject.updateFromSnapshot(ec.committedSnapshotForObject(anObject));
521            }
522    
523            /** ensure [snapshots_match] (anObject.editingContext().insertedObjects().containsObject(anObject) ||
524                                          anObject.editingContext().deletedObjects().containsObject(anObject)) ||
525                                         anObject.snapshot().equals(anObject.editingContext().committedSnapshotForObject(anObject)); **/
526        }
