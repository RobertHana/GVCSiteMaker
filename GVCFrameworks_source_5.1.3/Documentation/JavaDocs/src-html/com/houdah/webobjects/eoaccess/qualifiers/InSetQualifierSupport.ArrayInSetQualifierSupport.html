027    /**
028     * <p>
029     * Support class to handle SQL generation for InSetQualifier instances.
030     * </p>
031     * 
032     * <p>
033     * The code for this qualifier contains one ugly bit. In order to determine the
034     * string used to represent a match value in the SQL query I have the JDBC
035     * adaptor generate SQL for a key-value qualifier and extract relevant bits from
036     * it.
037     * </p>
038     * 
039     * <p>
040     * As it stands, the adaptor plugins apply database vendor specific
041     * transformation to values in a key-value qualifier. One such transformation is
042     * right padding values when querying against a fixed width char column.
043     * Unfortunately these transformations are not directly available from the
044     * public methods in the adaptor or plugin
045     * </p>
046     * 
047     * This leaves two alternatives. The above described hack or reproducing the
048     * desired behavior with code like the one below. The second approach has two
049     * distinct disadvantages. For one it is only partial: it takes care only of
050     * right padding. Other transformations may be needed. Most importantly though
051     * the below code is database vendor specific. In spite of having vendor
052     * specific behavior grouped in the plugin it gets scattered over various
053     * qualifier SQL generation support classes
054     * </p>
055     * 
056     * <pre>
057     * protected boolean needsPadding(EOSQLExpression expression, String keyPath)
058     * {
059     *      EOEntity entity = expression.entity();
060     *      EOAttribute attribute = attributeForPath(entity, keyPath);
061     * 
062     *      return needsPadding(attribute);
063     * }
064     * 
065     * protected boolean needsPadding(EOAttribute attribute)
066     * {
067     *      String valueType = attribute.valueType();
068     * 
069     *      return ((valueType != null) &amp;&amp; (valueType.startsWith(&quot;c&quot;)));
070     * }
071     * 
072     * protected String padSqlStringForValueAsNeeded(EOSQLExpression expression, String sqlStringForValue,
073     *              String keyPath)
074     * {
075     *      EOEntity entity = expression.entity();
076     *      EOAttribute attribute = attributeForPath(entity, keyPath);
077     * 
078     *      if (needsPadding(attribute)) {
079     *              sqlStringForValue = padSqlStringForValue(attribute, sqlStringForValue);
080     *      }
081     * 
082     *      return sqlStringForValue;
083     * }
084     * 
085     * protected String padSqlStringForValue(EOAttribute attribute, String sqlStringForValue)
086     * {
087     *      // TODO: This is ugly as it supposes that we have an Oracle database.
088     *      // This should be handled by the plugin
089     *      return &quot;RPAD(&quot; + sqlStringForValue + &quot;, &quot; + attribute.width() + &quot;)&quot;;
090     * }
091     * </pre>
092     * 
093     * <p>
094     * This sample code is provided for educational purposes. It is mainly to be
095     * considered a source of information and a way to spark off discussion. I make
096     * no warranty, expressed or implied, about the quality of this code or its
097     * usefulness in any particular situation. Use this code or any code based on it
098     * at your own risk. Basically, enjoy the read, but don't blame me for anything.
099     * </p>
100     * 
101     * @author Pierre Bernard http://homepage.mac.com/i_love_my/webobjects.html
102     */

113             * 
114             * @see com.webobjects.eoaccess.EOQualifierSQLGeneration.Support#sqlStringForSQLExpression(com.webobjects.eocontrol.EOQualifier,
115             *      com.webobjects.eoaccess.EOSQLExpression)
116             */
117            public String sqlStringForSQLExpression(EOQualifier qualifier, EOSQLExpression expression)
118            {
119                    InSetQualifier isQualifier = (InSetQualifier) qualifier;
120                    String keyPath = isQualifier.keyPath();
121                    NSSet values = isQualifier.values();
122                    Enumeration e = values.objectEnumerator();
123                    NSMutableSet valueArrays = new NSMutableSet(values.count());
124    
125                    while (e.hasMoreElements()) {
126                            valueArrays.addObject(new NSArray(e.nextElement()));
127                    }
128    
129                    ArrayInSetQualifier aisQualifier = new ArrayInSetQualifier(new NSArray(keyPath),
130                                    valueArrays);
131                    EOQualifierSQLGeneration.Support support = supportForClass(ArrayInSetQualifier.class);
132    
133                    return support.sqlStringForSQLExpression(aisQualifier, expression);
134            }

136            /*
137             * (non-Javadoc)
138             * 
139             * @see com.webobjects.eoaccess.EOQualifierSQLGeneration.Support#schemaBasedQualifierWithRootEntity(com.webobjects.eocontrol.EOQualifier,
140             *      com.webobjects.eoaccess.EOEntity)
141             */
142            public EOQualifier schemaBasedQualifierWithRootEntity(EOQualifier qualifier, EOEntity entity)
143            {
144                    InSetQualifier isQualifier = (InSetQualifier) qualifier;
145                    String keyPath = isQualifier.keyPath();
146                    EORelationship relationship = relationshipForPath(entity, keyPath);
147    
148                    if (relationship != null) {
149                            if (relationship.isFlattened()) {
150                                    relationship = (EORelationship) relationship.componentRelationships().lastObject();
151                            }
152    
153                            NSArray joins = relationship.joins();
154                            int joinCount = joins.count();
155                            NSMutableArray destAttrNames = new NSMutableArray(joinCount);
156                            NSMutableArray optimizedPaths = new NSMutableArray(joinCount);
157    
158                            for (int i = joinCount - 1; i >= 0; i--) {
159                                    String destAttributeName = ((EOJoin) joins.objectAtIndex(i)).destinationAttribute()
160                                                    .name();
161    
162                                    destAttrNames.addObject(destAttributeName);
163                                    optimizedPaths.addObject(optimizeQualifierKeyPath(entity, keyPath,
164                                                    destAttributeName));
165                            }
166    
167                            NSMutableSet newValues = new NSMutableSet(isQualifier.values().count());
168                            Enumeration values = isQualifier.values().objectEnumerator();
169    
170                            while (values.hasMoreElements()) {
171                                    Object value = values.nextElement();
172    
173                                    if (value == NSKeyValueCoding.NullValue || (value instanceof EOQualifierVariable)) {
174                                            NSMutableArray destValues = new NSMutableArray(joinCount);
175    
176                                            for (int j = 0; j < joinCount; j++) {
177                                                    destValues.addObject(value);
178                                            }
179    
180                                            newValues.addObject(destValues);
181                                    }
182                                    else {
183                                            EOEnterpriseObject enterpriseObject = (EOEnterpriseObject) value;
184                                            EOObjectStoreCoordinator objectStoreCoordinator = (EOObjectStoreCoordinator) enterpriseObject
185                                                            .editingContext().rootObjectStore();
186                                            NSDictionary destValues = objectStoreCoordinator.valuesForKeys(destAttrNames,
187                                                            enterpriseObject);
188    
189                                            newValues.addObject(destValues.objectsForKeys(destAttrNames,
190                                                            NSKeyValueCoding.NullValue));
191                                    }
192                            }
193    
194                            if (destAttrNames.count() == 1) {
195                                    NSMutableSet singleValues = new NSMutableSet(newValues.count());
196                                    Enumeration newValEnum = newValues.objectEnumerator();
197    
198                                    while (newValEnum.hasMoreElements()) {
199                                            NSArray newValue = (NSArray) newValEnum.nextElement();
200    
201                                            singleValues.addObject(newValue.objectAtIndex(0));
202                                    }
203    
204                                    return nullValueAwareQualifier(new InSetQualifier((String) optimizedPaths
205                                                    .objectAtIndex(0), singleValues));
206                            }
207                            else {
208                                    return new ArrayInSetQualifier(optimizedPaths, newValues);
209                            }
210                    }
211                    else {
212                            return nullValueAwareQualifier(isQualifier);
213                    }
214            }

231            // Protected instance methods
232    
233            protected EOQualifier nullValueAwareQualifier(InSetQualifier isQualifier)
234            {
235                    if (isQualifier.values().containsObject(NSKeyValueCoding.NullValue)) {
236                            EOQualifier nullQualifier = new EOKeyValueQualifier(isQualifier.keyPath(),
237                                            EOQualifier.QualifierOperatorEqual, NSKeyValueCoding.NullValue);
238    
239                            return new EOOrQualifier(new NSArray(new EOQualifier[] {
240                                            nullQualifier,
241                                            new InSetQualifier(isQualifier.keyPath(), isQualifier.values()
242                                                            .setBySubtractingSet(new NSSet(NSKeyValueCoding.NullValue))) }));
243                    }
244                    else {
245                            return isQualifier;
246                    }
247            }

269                    /*
270                     * (non-Javadoc)
271                     * 
272                     * @see com.webobjects.eocontrol.EOQualifier#qualifierWithBindings(com.webobjects.foundation.NSDictionary,
273                     *      boolean)
274                     */
275                    public EOQualifier qualifierWithBindings(NSDictionary bindings, boolean requiresAll)
276                    {
277                            Enumeration e1 = values().objectEnumerator();
278                            boolean didSubstitute = false;
279                            NSMutableSet boundValues = new NSMutableSet(values().count());
280    
281                            while (e1.hasMoreElements()) {
282                                    NSArray objects = (NSArray) e1.nextElement();
283                                    NSMutableArray values = new NSMutableArray(objects.count());
284                                    Enumeration e2 = objects.objectEnumerator();
285    
286                                    while (e2.hasMoreElements()) {
287                                            Object object = e2.nextElement();
288    
289                                            if (object instanceof EOQualifierVariable) {
290                                                    Object value = null;
291    
292                                                    if (bindings != null) {
293                                                            value = bindings.valueForKeyPath(((EOQualifierVariable) object).key());
294                                                    }
295    
296                                                    if (value != null) {
297                                                            values.addObject(value);
298                                                            didSubstitute = true;
299    
300                                                            continue;
301                                                    }
302    
303                                                    if (requiresAll) {
304                                                            throw new EOQualifier.QualifierVariableSubstitutionException(
305                                                                            "Error in variable substitution: value for variable " + object
306                                                                                            + " not found");
307                                                    }
308                                            }
309                                            else {
310                                                    values.addObject(object);
311                                            }
312                                    }
313    
314                                    boundValues.addObject(values);
315                            }
316    
317                            if (didSubstitute) {
318                                    return new ArrayInSetQualifier(keyPathArray(), boundValues);
319                            }
320                            else {
321                                    return this;
322                            }
323                    }

373            public static class ArrayInSetQualifierSupport extends QualifierGenerationSupport
374            {
375                    /*
376                     * (non-Javadoc)
377                     * 
378                     * @see com.webobjects.eoaccess.EOQualifierSQLGeneration.Support#sqlStringForSQLExpression(com.webobjects.eocontrol.EOQualifier,
379                     *      com.webobjects.eoaccess.EOSQLExpression)
380                     */
381                    public String sqlStringForSQLExpression(EOQualifier qualifier, EOSQLExpression expression)
382                    {
383                            ArrayInSetQualifier aisQualifier = (ArrayInSetQualifier) qualifier;
384                            EOEntity entity = expression.entity();
385                            NSArray keyPathArray = aisQualifier.keyPathArray();
386                            int keyPathCount = keyPathArray.count();
387                            boolean hasMultipleElements = (keyPathCount > 1);
388                            int[] cutPoints = new int[keyPathCount];
389                            StringBuffer buffer = new StringBuffer();
390    
391                            if (hasMultipleElements) {
392                                    buffer.append("(");
393                            }
394    
395                            for (int i = 0; i < keyPathCount; i++) {
396                                    String keyPath = (String) keyPathArray.objectAtIndex(i);
397                                    String attributeString = expression.sqlStringForAttributeNamed(keyPath);
398    
399                                    if (attributeString == null) {
400                                            throw new IllegalStateException(
401                                                            "sqlStringForKeyValueQualifier: attempt to generate SQL for "
402                                                                            + aisQualifier.getClass().getName() + " " + aisQualifier
403                                                                            + " failed because attribute identified by key '" + keyPath
404                                                                            + "' was not reachable from from entity '" + entity.name()
405                                                                            + "'");
406                                    }
407    
408                                    if (i > 0) {
409                                            buffer.append(", ");
410                                    }
411    
412                                    buffer.append(expression.formatSQLString(attributeString, attributeForPath(entity,
413                                                    keyPath).readFormat()));
414    
415                                    // Initialize to prepare for nex loop
416                                    cutPoints[i] = -1;
417                            }
418    
419                            if (hasMultipleElements) {
420                                    buffer.append(")");
421                            }
422    
423                            buffer.append(" IN (");
424    
425                            Enumeration e = aisQualifier.values().objectEnumerator();
426                            boolean firstIteration = true;
427    
428                            while (e.hasMoreElements()) {
429                                    NSArray value = (NSArray) e.nextElement();
430    
431                                    if (value.count() != keyPathCount) {
432                                            throw new IllegalStateException(
433                                                            "sqlStringForKeyValueQualifier: key and value counts don't match");
434                                    }
435    
436                                    if (firstIteration) {
437                                            firstIteration = false;
438                                    }
439                                    else {
440                                            buffer.append(", ");
441                                    }
442    
443                                    if (hasMultipleElements) {
444                                            buffer.append("(");
445                                    }
446    
447                                    for (int i = 0; i < keyPathCount; i++) {
448                                            Object oneValue = value.objectAtIndex(i);
449                                            String keyPath = (String) keyPathArray.objectAtIndex(i);
450    
451                                            if (oneValue instanceof EOQualifierVariable) {
452                                                    throw new IllegalStateException(
453                                                                    "sqlStringForKeyValueQualifier: attempt to generate SQL for "
454                                                                                    + aisQualifier.getClass().getName() + " " + aisQualifier
455                                                                                    + " failed because the qualifier variable '$"
456                                                                                    + ((EOQualifierVariable) oneValue).key() + "' is unbound.");
457                                            }
458    
459                                            if (i > 0) {
460                                                    buffer.append(", ");
461                                            }
462    
463                                            // This is ugly: We generate SQL for a key-value qualifier
464                                            // and then extract
465                                            // the part representing the value. Unfortunately, there are
466                                            // case where the
467                                            // database adaptor must apply vendor specific magic to the
468                                            // value. One such
469                                            // transformation is padding values for qualifying against
470                                            // fixed width CHAR
471                                            // columns. These transformations are not publicly exposed
472                                            // by the adaptor.
473                                            //
474                                            // I guess the below code is the best we can get while still
475                                            // database vendor
476                                            // independant.
477    
478                                            EOKeyValueQualifier kvQualifier = new EOKeyValueQualifier(keyPath,
479                                                            EOQualifier.QualifierOperatorEqual, oneValue);
480                                            String sqlStringForQualifier = expression
481                                                            .sqlStringForKeyValueQualifier(kvQualifier);
482    
483                                            if (cutPoints[i] == -1) {
484                                                    String equalString = expression.sqlStringForSelector(
485                                                                    EOQualifier.QualifierOperatorEqual, value);
486    
487                                                    cutPoints[i] = 1 + sqlStringForQualifier.indexOf(equalString);
488                                            }
489    
490                                            buffer.append(sqlStringForQualifier.substring(cutPoints[i],
491                                                            sqlStringForQualifier.length()));
492                                    }
493    
494                                    if (hasMultipleElements) {
495                                            buffer.append(")");
496                                    }
497                            }
498    
499                            buffer.append(")");
500    
501                            return buffer.toString();
502                    }

539                    // Protected instance methods
540    
541                    protected EOQualifier nullValueAwareQualifier(ArrayInSetQualifier aisQualifier)
542                    {
543                            Enumeration values = aisQualifier.values().objectEnumerator();
544                            NSMutableSet plainValues = new NSMutableSet(aisQualifier.values().count());
545                            NSMutableArray otherQualifiers = new NSMutableArray();
546    
547                            while (values.hasMoreElements()) {
548                                    NSArray value = (NSArray) values.nextElement();
549    
550                                    if (value.containsObject(NSKeyValueCoding.NullValue)) {
551                                            otherQualifiers.addObject(EOQualifier
552                                                            .qualifierToMatchAllValues(new NSDictionary(value, aisQualifier
553                                                                            .keyPathArray())));
554                                    }
555                                    else {
556                                            plainValues.addObject(value);
557                                    }
558                            }
559    
560                            if (otherQualifiers.count() == 0) {
561                                    return aisQualifier;
562                            }
563                            else {
564                                    // Could optimize this by grouping into a minimal number of
565                                    // InSetQualifiers
566    
567                                    otherQualifiers.addObject(new ArrayInSetQualifier(aisQualifier.keyPathArray(),
568                                                    plainValues));
569    
570                                    return new EOAndQualifier(otherQualifiers);
571                            }
572                    }
