121        /**
122         * Test Reversibility
123         *
124         * For formatters to function properly, they must be able to call between format and parse without altering the data.
125         */
126        public void testReversibility()
127        {
128            testFormatter.setPattern("dd MM yyyy HH:mm:ss");
129    
130            // There is something around 1958 that makes this problem occur.
131            GregorianCalendar aDate = new GregorianCalendar(1957, 0, 1);
132            
133            for (int dayCount = 0; dayCount <= (3 * 366); dayCount++)
134            {
135                try
136                {
137                    NSTimestamp timestampFromDate = new NSTimestamp(aDate.getTime());
138                    String dateAsString = testFormatter.format(timestampFromDate);
139                    NSTimestamp stringAsTimestamp = (NSTimestamp) testFormatter.parseObject(dateAsString);
140                    assertTrue("Failed to reverse date " + aDate.getTime(), stringAsTimestamp.equals(timestampFromDate));
141                    aDate.add(Calendar.DAY_OF_YEAR , 1);
142                }
143                catch (ParseException e)
144                {
145                    fail("Failed  to handle " + aDate);
146                }
147            }
148        }

151        /**
152         * Test NSTimestampFormatter parsing with time zone Z format bug
153         *
154         * 
155         */
156        public void testNSTimestampFormatterParseZBug()
157        {
158    
159            String dateWithGMTOffset = "2007-10-25 2:53:27 GMT-06:00";
160            String dateWithRFC822Offset = "2007-10-25 2:53:27 -0600";
161            
162            String formatForGMTOffset = "yyyy-MM-dd HH:mm:ss z";
163            String formatForRFC822Offset = "yyyy-MM-dd HH:mm:ss Z";
164            
165            NSTimestampFormatter gmtOffsetFormatter = new NSTimestampFormatter(formatForGMTOffset);
166            NSTimestampFormatter rfc822OffsetFormatter = new NSTimestampFormatter(formatForRFC822Offset);
167            
168            Format gmtOffsetJavaFormatter = new SimpleDateFormat(formatForGMTOffset);
169            Format rfc822OffsetJavaFormatter = new SimpleDateFormat(formatForRFC822Offset);
170    
171            try
172            {
173                NSTimestamp timestampFromGMTOffset = (NSTimestamp) gmtOffsetFormatter.parseObject(dateWithGMTOffset);
174                NSTimestamp timestampFromRFC822Offset = (NSTimestamp) rfc822OffsetFormatter.parseObject(dateWithRFC822Offset);
175                NSTimestamp expectedTimestamp = new NSTimestamp(2007, 10, 25, 2, 53, 27, NSTimeZone.timeZoneForSecondsFromGMT(-6 * 60 * 60));
176    
177                assertTrue("GMT Offset with Java", gmtOffsetJavaFormatter.parseObject(dateWithGMTOffset).equals(expectedTimestamp));
178                assertTrue("RFC 822 Offset with Java", rfc822OffsetJavaFormatter.parseObject(dateWithRFC822Offset).equals(expectedTimestamp));
179                
180                assertTrue("GMT Offset", timestampFromGMTOffset.equals(expectedTimestamp));
181                
182                // This should be true.
183                assertFalse("RFC 822 Offset", timestampFromRFC822Offset.equals(expectedTimestamp));
184            }
185            catch (ParseException e)
186            {
187                fail(e.getMessage());
188            }
189        }

193        /**
194         * Test NSTimestampFormatter formatting with time zone Z format bug
195         *
196         * 
197         */
198        public void testNSTimestampFormatterFormatZBug()
199        {
200    
201            String dateWithGMTOffset = "2007-10-25 2:53:27 GMT-06:00";
202            String formatForGMTOffset = "yyyy-MM-dd HH:mm:ss z";
203            String formatForRFC822Offset = "yyyy-MM-dd HH:mm:ss Z";
204            
205            NSTimestampFormatter gmtOffsetNSTimestampFormatter = new NSTimestampFormatter(formatForGMTOffset);
206            NSTimestampFormatter rfc822OffsetNSTimestampFormatter = new NSTimestampFormatter(formatForRFC822Offset);
207            
208            Format gmtOffsetJavaFormatter = new SimpleDateFormat(formatForGMTOffset);
209            Format rfc822OffsetJavaFormatter = new SimpleDateFormat(formatForRFC822Offset);
210            
211            try
212            {
213                NSTimestamp timestampFromNSTimestampFormatter = (NSTimestamp) gmtOffsetNSTimestampFormatter.parseObject(dateWithGMTOffset);
214                Date timestampFromJavaFormatter = (Date) gmtOffsetJavaFormatter.parseObject(dateWithGMTOffset);
215                assertEquals("Parsed timestamps match", 
216                             timestampFromNSTimestampFormatter.getTime(),
217                             timestampFromJavaFormatter.getTime());
218                
219                // These SHOULD be the same, but NSTimestampFormatter is producing 2007-10-25 01:53:27 -0659
220                // instead of 2007-10-25 01:53:27 -0700 (-0700 is the local timezone).  Note that the timezone
221                // is off by one minute.
222                assertNotSame("RFC 822 Offset with Java", 
223                             rfc822OffsetNSTimestampFormatter.format(timestampFromNSTimestampFormatter),
224                             rfc822OffsetJavaFormatter.format(timestampFromNSTimestampFormatter));
225    
226                assertNotSame("RFC 822 Offset with Java", 
227                             rfc822OffsetNSTimestampFormatter.format(timestampFromNSTimestampFormatter),
228                             rfc822OffsetJavaFormatter.format(timestampFromJavaFormatter));
229            }
230            catch (ParseException e)
231            {
232                fail(e.getMessage());
233            }
234        }
