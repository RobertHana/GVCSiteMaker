014    public class EOFDebug
015    {
016    
017        /**
018         * Convenience method to aid in debugging tests.  Turns EOAdaptor SQL logging on or off.
019         *
020         * @param shouldLog - <code>true</code> if SQL logging should be turned on, <code>false</code> if it should be turned off.
021         */
022        public static void logSQL(boolean shouldLog)
023        {
024            if (shouldLog)
025            {
026                NSLog.allowDebugLoggingForGroups(NSLog.DebugGroupSQLGeneration
027                                                 | NSLog.DebugGroupDatabaseAccess
028                                                 | NSLog.DebugGroupEnterpriseObjects);
029                NSLog.debug.setAllowedDebugLevel(NSLog.DebugLevelDetailed);
030            }
031            else
032            {
033                NSLog.refuseDebugLoggingForGroups(NSLog.DebugGroupSQLGeneration
034                                                  | NSLog.DebugGroupDatabaseAccess
035                                                  | NSLog.DebugGroupEnterpriseObjects);
036                NSLog.debug.setAllowedDebugLevel(NSLog.DebugLevelOff);
037            }
038        }
039    }
