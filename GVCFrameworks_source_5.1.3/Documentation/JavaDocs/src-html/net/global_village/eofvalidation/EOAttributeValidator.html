017    public class EOAttributeValidator extends Object
018    {
019        protected final static int generatedPrimaryKeySize = 24;
020    
021    
022        /**
023         * Static methods only.  You'll never need to instantiate this class.
024         */
025        private EOAttributeValidator()
026        {
027            super();
028        }
029    
030    
031    
032        /**
033         * Returns <code>true</code> if EOAttribute's implementation would have raised an exception for this value for this attribute.  This is used only for the post condition of validateValueForAttribute(Object, EOAttribute)
034         *
035         * @param value value to validate for this attribute
036         * @param attribute EOAttribute to validate this value for
037         * @return <code>true</code> if EOAttribute would have raised an exception for this value for this attribute.
038         */
039        static protected boolean doesSuperRaise(Object value, EOAttribute attribute)
040        {
041            /** require [valid_param] attribute != null; **/
042    
043            boolean doesRaise = true;
044            try
045            {
046                attribute.validateValue(value);
047                doesRaise = false;
048            }
049            catch (Exception e) { }
050    
051            return doesRaise;
052        }
053    
054    
055    
056        /**
057         * This method oversees validating this value for this attribute.  If it is not valid the exception returned contains a customized, localized validation failure message.
058         *
059         * @param value value to validate for this attribute
060         * @param attribute EOAttribute to validate this value for
061         * @return null if the value is acceptable, or returns a new value if needed (e.g. coercion).
062         * @exception EOFValidationException if the value is not acceptable for this attribute
063         */
064        static public Object validateValue(Object value, EOAttribute attribute) throws EOFValidationException
065        {
066            /** require [valid_param] attribute != null; **/
067    
068            checkTypeCompatibility(value, attribute);
069            checkNullity(value, attribute);
070    
071            if (! EOEntityAdditions.isNull(value))
072            {
073                checkWidth(value, attribute);
074            }
075    
076            // We could also check the scale / precision of an NSNumber.  EOF does not appear to do this, and thus relying on database exceptions to indicate this.  FrontBase does not (as of 1.2w) raise an exception and ignores scale and precision.  Oracle rounds up decimals, but raises an exception if the scale is exceeded.
077    
078            return value;
079    
080            /** ensure [we_must_raise_if_super_raises] ! doesSuperRaise(value, attribute); **/
081        }
082    
083    
084    
085        /**
086         * Raises a validation exception if value is not compatible with the internal type (class) that this attribute is mapped to.  The value to is compatible in three cases:
087         * <ul>
088         * <li> it is of the same class as the value class for this attribute
089         * <li> it is of a sub-class of the value class for this attribute
090         * <li> it is null or EONullValue
091         * </ul>
092         *
093         * @param value value to validate for this attribute
094         * @param attribute EOAttribute to validate this value for
095         * @exception EOFValidationException if the value is not type compatible with this attribute
096         */
097        static public void checkTypeCompatibility(Object value, EOAttribute attribute) throws EOFValidationException
098        {
099            /** require [valid_param] attribute != null; **/
100    
101            // EOF allows conversion from non-numeric text to NaN as an acceptable value, we do not.  Looks like this is not an issue in Java.  How does Java handle this?
102            // boolean isInvalidNumber = (value instanceof BigDecimal) && (value.equals(NaN));
103    
104            if ( ! EOEntityAdditions.isNull(value))
105            {
106                Class desiredClass = classForAttribute(attribute);
107    
108                if ( ! desiredClass.isInstance(value))
109                {
110                    throw new EOFValidationException(null, attribute.entity().name(), attribute.name(), EOFValidation.InvalidValue, value);
111                }
112            }
113        }
114    
115    
116    
117        /**
118         * Raises a validation exception if this attribute is null and is not allowed to be, or if it is an empty / all white space string and empty strings are being considered as null.
119         *
120         * @param value value to validate for this key
121         * @param attribute EOAttribute to validate this value for
122         * @exception EOFValidationException if the value is not type compatible with this attribute
123         */
124        static public void checkNullity(Object value,
125                                        EOAttribute attribute) throws EOFValidationException
126        {
127            /** require [valid_param] attribute != null; **/
128    
129            boolean cantBeNull = ! (attribute.allowsNull() || isGeneratedPrimaryKey(attribute));
130    
131            boolean isDisallowedEmptyString = EOFValidation.shouldTreatEmptyStringsAsNull() &&
132                (value instanceof String) && (((String)value).trim().length() == 0);
133    
134            if (cantBeNull && (EOEntityAdditions.isNull(value) || isDisallowedEmptyString))
135            {
136                throw new EOFValidationException(null, attribute.entity().name(), attribute.name(), EOFValidation.NullNotAllowed, value);
137            }
138        }
139    
140    
141    
142        /**
143         * Raises a validation exception if this attribute is wider (longer) than allowed.  This only applied to NSString and NSData types.
144         *
145         * @param value value to validate for this key
146         * @param attribute EOAttribute to validate this value for
147         * @exception EOFValidationException        if the value is not type compatible with this attribute
148         */
149        static public void checkWidth(Object value,
150                                      EOAttribute attribute) throws EOFValidationException
151        {
152            /** require
153            [valid_attribute_param] attribute != null;
154            [valid_value_param] ! EOEntityAdditions.isNull(value); **/
155    
156            // The value passed here has been corerced by the adaptor so it is one of four known types. Two of these types, NSString and NSData, can have width restrictions and also have a length method.  Doesn't check width for attributes that have a width of 0, since those are likely to be LOBs.
157            Class classOfAttribute = classForAttribute(attribute);
158            if ((attribute.width() > 0) && ((classOfAttribute.equals(NSData.class)) || (classOfAttribute.equals(String.class))))
159            {
160                int actualWidth = 0;
161    
162                if (value instanceof NSData)
163                {
164                    actualWidth = ((NSData)value).length();
165                }
166                else
167                {
168                    actualWidth = ((String)value).length();
169                }
170    
171                if (actualWidth > attribute.width())
172                {
173                    throw new EOFValidationException(null, attribute.entity().name(), attribute.name(), EOFValidation.TooLong, value);
174                }
175            }
176        }
177    
178    
179    
180        /**
181         * Returns <code>true</code> if this attribute represents a primary key which will be generated by EOF. This attribute is considered to represent a primary key which will be generated if:
182         * <ul>
183         * <li>it is part of the primary key
184         * <li>there is only one column in the primary key (EOF does not generate for compound keys)
185         * <li>the attribute type is appropriate for generation (NSNumber or 12 byte NSData)
186         * </ul>
187         *
188         * @param attribute EOAttribute to check
189         * @return <code>true</code> if the attribute is a generatable primary key
190         */
191        static public boolean isGeneratedPrimaryKey(EOAttribute attribute)
192        {
193            /** require [valid_param] attribute != null; **/
194    
195            NSArray primaryKeysForEntity = attribute.entity().primaryKeyAttributes();
196            int columnsInPrimaryKey = primaryKeysForEntity.count();
197            boolean isPartOfPrimaryKey = primaryKeysForEntity.containsObject(attribute);
198            boolean isPrimaryKey = (columnsInPrimaryKey == 1) && isPartOfPrimaryKey;
199    
200            Class attributeValueClass = classForAttribute(attribute);
201            boolean isGeneratableType = (attributeValueClass.equals(Number.class)) || (attributeValueClass.equals(BigDecimal.class)) || (attributeValueClass.equals(NSData.class) && (attribute.width() == generatedPrimaryKeySize));
202    
203            return isPrimaryKey && isGeneratableType;
204        }

208        /**
209         * Returns the Java class that is used to implement this attribute.  This is based on the Objective-C name from the EOModel as determined by valueClassName() for this attribute.
210         *
211         * @param attribute EOAttribute to return class for
212         * @return the Java class that is used to implement this attribute
213         * @exception RuntimeException if the class for this attribute can not be located
214         *
215         */
216        static public Class classForAttribute(EOAttribute attribute)
217        {
218            /** require [valid_param] attribute != null; **/
219    
220            return EOEntityAdditions.safeClassForName(attribute.entity(), attribute.className());
221    
222            /** ensure [valid_result] Result != null; **/
223        }
224    
225    
226    
227        /**
228         * Convenience methods returning a localized display name for this attribute, or a prettified attribute name if none is available.
229         *
230         * @param attribute attribute to lookup display name for.
231         * @return a localized display name for this attribute, or a prettified version of the attribute name if none is available.
232         */
233        public static String displayNameForAttribute(EOAttribute attribute)
234        {
235            /** require [valid_param] attribute != null; **/
236    
237            return EOEntityAdditions.displayNameForPropertyNamed(attribute.name(), attribute.entity());
238    
239            /** ensure [valid_result] Result != null; **/
240        }
