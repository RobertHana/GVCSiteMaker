023    public class TestArchivableLookup extends VersionableArchivableLookup

025        // default shouldVersion inherited which returns false
026        
027        // and now for the various wrapper methods ...
028        
029        /**
030         * Fetchs all instances of this entity, qualified by aQualifier. 
031         *
032         * @param aQualifier the qualifier to use for the fetch
033         * @param editingContext the <code>EOEditingContext</code> to fetch from
034         * @return NSArray of the desired objects
035         */
036        static public NSArray objectsWithQualifier(EOQualifier aQualifier, EOEditingContext editingContext)
037        {
038            /** require [editingContext_not_null] editingContext != null; **/
039            
040            return EOEditingContextAdditions.objectsWithQualifier(editingContext, "TestArchivableLookup", aQualifier);
041    
042            /** ensure [Result_not_null] Result != null; **/
043        }
044    
045        
046    
047        /**
048         * Fetchs all active objects, qualified by aQualifier.  This method should be used for fetching to avoid returning inactive objects.
049         *
050         * @param aQualifier the qualifier to use for the fetch
051         * @param editingContext the <code>EOEditingContext</code> to fetch from
052         * @return NSArray all active objects from this Entity, qualified by aQualifier
053         */
054        static public NSArray activeObjectsWithQualifier(EOQualifier aQualifier, EOEditingContext editingContext)
055        {
056            /** require [editingContext_not_null] editingContext != null; **/
057            
058            return VersionableArchivableObject.activeObjectsWithQualifier( editingContext, "TestArchivableLookup", aQualifier);
059    
060            /** ensure [Result_not_null] Result != null; **/
061        }
062    
063        
064    
065        /**
066        * Fetchs all active objects, qualified by aQualifier, sorted by NameComparator.
067         *
068         * @param aQualifier the qualifier to use for the fetch
069         * @param editingContext the <code>EOEditingContext</code> to fetch from
070         * @return NSArray all active objects from this entity, qualified by aQualifier, sorted by compare()
071         */
072        static public NSArray orderedActiveObjectsWithQualifier(EOQualifier aQualifier, EOEditingContext editingContext)
073        {
074            /** require [editingContext_not_null] editingContext != null; **/
075            
076            return VersionableArchivableObject.orderedActiveObjectsWithQualifier(editingContext, "TestArchivableLookup", aQualifier, VersionableArchivableLookup.NameComparator );
077    
078            /** ensure [Result_not_null] Result != null; **/
079        }
080    
081        
082    
083        /**
084         *  Fetches all active objects sorted by compare(). 
085         *
086         * @param editingContext the <code>EOEditingContext</code> to fetch from
087         * @return NSArray of the desired objects
088         */
089        static public NSArray orderedActiveObjects(EOEditingContext editingContext)
090        {
091            /** require [editingContext_not_null] editingContext != null; **/
092            
093            return VersionableArchivableObject.orderedActiveObjects(editingContext, "TestArchivableLookup", VersionableArchivableLookup.NameComparator);
094    
095            /** ensure [Result_not_null] Result != null; **/
096        }
097    
098        
099    
100        /**
101         * Fetchs all objects sorted by compare().
102         *
103         * @param editingContext the <code>EOEditingContext</code> to fetch from
104         * @return NSArray of the desired objects
105         */
106        static public NSArray orderedObjects(EOEditingContext editingContext)
107        {
108            /** require [editingContext_not_null] editingContext != null; **/
109            
110            return VersionableArchivableObject.orderedObjects(editingContext, "TestArchivableLookup", VersionableArchivableLookup.NameComparator );
111    
112            /** ensure [Result_not_null] Result != null; **/
113        }
114

116        // compares the same way as superclass...
