074        /**
075         * Returns <code>true</code> if this exception is from an EOAdaptorOperation failure.
076         * This is detected by checking for the presence of EOFailedAdaptorOperationKey.
077         * If this key is present, then this exception has the expected keys and can be analysed as an adaptor failure exception.
078         *
079         * @param exception the exception to check.
080         * @return true if this exception is from an EOAdaptor failure
081         */
082        static public boolean isAdaptorOperationFailureException(EOGeneralAdaptorException exception)
083        {
084            /** require [valid_param] exception != null; **/
085    
086            boolean isAdaptorOperationFailureException = false;
087            if (exception.userInfo() != null)
088            {
089                isAdaptorOperationFailureException = exception.userInfo().objectForKey(EOAdaptorChannel.FailedAdaptorOperationKey) != null;
090            }
091    
092            return isAdaptorOperationFailureException;
093        }
094

097        /**
098         * Returns <code>true</code> if this exception is from a database base level failure.  This is the case if it is
099         * an JDBCAdaptorException or if it has an EOFailedAdaptorOperationKey.
100         *
101         * @param exception the exception to check
102         * @return true if this exception is from a database base level failure
103         */
104        static public boolean isDatabaseFailureException(EOGeneralAdaptorException exception)
105        {
106            /** require [valid_param] exception != null; **/
107            return isAdaptorOperationFailureException(exception) || exception instanceof JDBCAdaptorException;
108        }

112        /**
113         * Returns <code>true</code> if this exception is from an EOAdaptor failure.  This is detected by checking for
114         * the presence of EOAdaptorChannel.AdaptorFailureKey.  If this key is present, then this exception has the expected
115         * keys and can be analysed as an adaptor failure exception.
116         *
117         * @param exception the exception to check.
118         * @return <code>true</code> if this exception is an optimistic locking failure
119         */
120        static public boolean isOptimisticLockingFailure(EOGeneralAdaptorException exception)
121        {
122            /** require [valid_param] exception != null; **/
123    
124            boolean isOptimisticLockingFailure = false;
125    
126            if (exception.userInfo() != null)
127            {
128                String adaptorFailureKey = (String)exception.userInfo().objectForKey(EOAdaptorChannel.AdaptorFailureKey);
129                if (adaptorFailureKey != null)
130                {
131                    isOptimisticLockingFailure = adaptorFailureKey.equals(EOAdaptorChannel.AdaptorOptimisticLockingFailure);
132                }
133            }
134    
135            return isOptimisticLockingFailure;
136        }

140        /**
141         * Returns <code>true</code> if this exception is from an integrity constraint violation.
142         * This is detected by checking for an X/Open SQL CAE SQLState of 23XXX in the nested SQLException. See
143         * linked pages for lists of the error codes.
144         * <p>
145         * The SQLException documentation also states that this can be an SQL 99 error state.  SQL99 codes
146         * are not yet supported by this method.
147         * </p>
148         *
149         * @see <a href="http://www.roth.net/perl/odbc/SQLState/">SQLState 1</a>
150         * @see <a href="http://publib.boulder.ibm.com/infocenter/idshelp/v10/index.jsp?topic=/com.ibm.esqlc.doc/esqlc209.htm">SQLState 2</a>
151         * @see <a href="http://www.halfile.com/odbc.html">SQLState 3</a>
152         *
153         * @param exception the exception to check.
154         * @return <code>true</code> if this exception is an integrity constraint violation
155         */
156        static public boolean isIntegrityConstraintViolation(EOGeneralAdaptorException exception)
157        {
158            /** require [valid_param] exception != null;    **/
159            return isDatabaseFailureException(exception) && sqlException(exception) != null && sqlException(exception).getSQLState().startsWith("23");
160        }

182        /**
183         * Extracts and returns the SQLException that caused the EOGeneralAdaptorException.  If this happened on a specific
184         * SQL statement (non deferred constraint), it will be an EOAdaptorOperation that failed.  If this happened
185         * during commit (deferred constraint), there will be only the JDBCAdaptorException.
186         *
187         * @param exception the exception to extract the SQLException from
188         * @return the SQLException that caused the EOGeneralAdaptorException
189         */
190        static public SQLException sqlException(EOGeneralAdaptorException exception)
191        {
192            /** require [valid_param] exception != null;
193                        [isDatabaseFailureException] isDatabaseFailureException(exception);       **/
194    
195            JDBCAdaptorException jdbcException = null;
196            if (isAdaptorOperationFailureException(exception))
197            {
198                EOAdaptorOperation failedOperation = (EOAdaptorOperation)exception.userInfo().objectForKey(EOAdaptorChannel.FailedAdaptorOperationKey);
199                Throwable opException = failedOperation.exception();
200                if (opException == null || ! (opException instanceof com.webobjects.jdbcadaptor.JDBCAdaptorException))
201                {
202                    throw new NSForwardException(exception, "EOGeneralAdaptorException failed operation has unexpected exception: " + opException);
203                }
204                jdbcException = (JDBCAdaptorException)opException;
205            }
206            else
207            {
208                jdbcException = (JDBCAdaptorException)exception;
209            }
210    
211            return jdbcException.sqlException();
212            /** ensure [valid_result] Result != null;  **/
213        }

255        /**
256         * Returns the entity associated with the database failure exception.  If this is
257         * an EOAdaptorOperation level failure, the enity can be determined directly.  If this is
258         * a JDDBAdaptor exception, the entity must be inferred from the exception returned from
259         * the database.  This is not nearly as exact.
260         *
261         * @see DataBaseExceptionInterpreter#tableNameFromSQLException(SQLException)
262         * @see ERXEOAccessUtilities#entityUsingTable(EOEditingContext, String)
263         *
264         * @param exception the exception to check
265         * @param ec to use to find entity if this is a JDBCAdaptorException
266         * @return the entity for the object that the adaptor failure exception was raised for
267         */
268        static public EOEntity entitySaveFailedOn(EOGeneralAdaptorException exception, com.webobjects.eocontrol.EOEditingContext ec)
269        {
270            /** require
271            [valid_exception] exception != null;
272            [valid_ec] ec != null;
273            [is_adaptor_failure_exception] isDatabaseFailureException(exception); **/
274    
275            if (isAdaptorOperationFailureException(exception))
276            {
277                EODatabaseOperation operation = databaseOperation(exception);
278                return operation.entity();
279            }
280    
281            /* We have a JDCBAdaptorException here.  We have only the SQLException, so all that we can
282             * do is to try and extract the table name from the message and reverse that into an entity.
283             * As one table can be used for multiple entities, this is at best approximate.
284             */
285            String tableName = exceptionInterpreterFor(databaseTypeFromException(exception)).tableNameFromSQLException(sqlException(exception));
286            EOEntity entity = ERXEOAccessUtilities.entityUsingTable(ec, tableName);
287            if (entity == null)
288            {
289                throw new IllegalArgumentException("Can't determine EOEntity for table named " + tableName + " from exception " + exception);
290            }
291            return entity;
292            /** ensure [valid_result] Result != null; **/
293        }

297        /**
298         * Returns the adaptor operation that was being performed when the adaptor failure exception occurred.
299         *
300         * @param exception the exception to check.
301         * @return the adaptor operation that was being performed when the adaptor failure exception occurred.
302         */
303        static public int failedAdaptorOperator(EOGeneralAdaptorException exception)
304        {
305            /** require
306            [valid_param] exception != null;
307            [is_adaptor_failure_exception] isAdaptorOperationFailureException(exception); **/
308    
309            // Although the name of the key EOFailedDatabaseOperationKey would suggest that a singular result is returned, the result is in fact an array.  The array has a single element which is a number (Adaptor Operators are defined as ints).
310            NSArray adaptorOperators = databaseOperation(exception).adaptorOperations();
311            return ((EOAdaptorOperation)adaptorOperators.objectAtIndex(0)).adaptorOperator();
312        }

338        /**
339         * Extracts the type of database (e.g. FrontBase, Oracle, etc.) from the exception.  This is converted
340         * to a standard form using DatabaseTypeNamesForJDBCTypes to translate the name extracted from the connection
341         * URL (e.g. jdbc:FrontBase:...)
342         *
343         * @param exception EOGeneralAdaptorException to use to determine the database type
344         * @return the type of database from the exception
345         */
346        public static String databaseTypeFromException(EOGeneralAdaptorException exception)
347        {
348            /** require [valid_param] exception != null;
349                        [isDatabaseFailureException] isDatabaseFailureException(exception);    **/
350    
351            String databaseType = null;
352    
353            if (isAdaptorOperationFailureException(exception))
354            {
355                EOAdaptorOperation failedOperation = (EOAdaptorOperation)exception.userInfo().objectForKey(EOAdaptorChannel.FailedAdaptorOperationKey);
356                String dbURL = (String)failedOperation.entity().model().connectionDictionary().objectForKey(JDBCAdaptor.URLKey);
357                int firstColon = dbURL.indexOf(':');
358                int secondColon = dbURL.indexOf(':', firstColon + 1);
359                String jdbcType = dbURL.substring(firstColon + 1, secondColon);
360                databaseType = (String)DatabaseTypeNamesForJDBCTypes.objectForKey(jdbcType.toUpperCase());
361                if (databaseType == null)
362                {
363                    throw new RuntimeException("DatabaseTypeNamesForJDBCTypes does not have a name defined for " + jdbcType);
364                }
365            }
366    
367            /* OK, this part is nasty.
368             * We have a JDCBAdaptorException here.  We don't know the JDCBAdaptor, only the SQLException.
369             * To work around this, we could implement the EOAdaptorContext.Delegate to track the last context to try to
370             * commit in this thread and record the adaptor in the thread for use here.  We would then extract the
371             * connectionDictionaryURL() from the adaptor.
372             *
373             * In the meantime, we will cheat by guessing from the message.
374             */
375            else if (exception.getMessage().startsWith("Exception condition"))
376            {
377                databaseType = FrontBase;
378            }
379            else {
380                throw new RuntimeException("Can't determine database type from message " + exception.getMessage());
381            }
382    
383            return databaseType;
384            /** ensure [valid_result] Result != null;  **/
385        }

389        /**
390         * Locates and returns a subclass of DataBaseExceptionInterpreter appropriate to the database type.
391         * The database type is intended to be from databaseTypeFromException(EOGeneralAdaptorException), though
392         * this is not required.
393         *
394         * @see #databaseTypeFromException(EOGeneralAdaptorException)
395         * @param dbType String identifying the database, dbType + "ExceptionInterpreter" must be a valid class
396         * @return a subclass of DataBaseExceptionInterpreter appropriate to the database type
397         */
398        public static DataBaseExceptionInterpreter exceptionInterpreterFor(String dbType)
399        {
400            /** require [non_null_type] dbType != null;   **/
401            DataBaseExceptionInterpreter exceptionInterpreter = (DataBaseExceptionInterpreter) ExceptionInterpreters.objectForKey(dbType);
402    
403            // Lazy creation
404            if (exceptionInterpreter == null)
405            {
406                Class exceptionInterpreterClass = _NSUtilities.classWithName(dbType + "ExceptionInterpreter");
407                if (exceptionInterpreterClass == null)
408                {
409                    throw new RuntimeException("Class not found for " + dbType + "ExceptionInterpreter");
410                }
411    
412                try
413                {
414                    exceptionInterpreter = (DataBaseExceptionInterpreter)exceptionInterpreterClass.newInstance();
415                    ExceptionInterpreters.setObjectForKey(exceptionInterpreter, dbType);
416                }
417                catch (Exception e)
418                {
419                    throw new ExceptionConverter(e);
420                }
421            }
422    
423            return exceptionInterpreter;
424            /** ensure [valid_result] Result != null;  **/
425        }
