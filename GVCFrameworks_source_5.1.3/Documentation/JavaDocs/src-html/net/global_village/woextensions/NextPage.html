009    /**
010     * Holds all the information needed to go to a given page.  Use this to go back to the original page
011     * when you have a page that can be accessed from multiple pages.
012     *

015     * a copy of which has been included with this distribution in the LICENSE.TXT file.
016     * @version $Revision: 3$
017     */
018    public class NextPage
019    {
020        protected String nextPageName;
021        protected NSArray orderedDictionariesOfKeysAndValuesToSetInNextPage;
022    
023    
024        /**
025         * Designated constructor.
026         *
027         * @param aNextPageName the name of next page to display
028         * @param aValuesToSetInNextPage ordered array of dictionaries containing keys and values to set in the next page
029         */
030        public NextPage(String aNextPageName, NSArray aValuesToSetInNextPage)
031        {
032            super();
033            /** require
034            [valid_aNextPageName_param] aNextPageName != null;
035            [valid_aValuesToSetInNextPage_param] aValuesToSetInNextPage != null; **/
036    
037            nextPageName = aNextPageName;
038            orderedDictionariesOfKeysAndValuesToSetInNextPage = aValuesToSetInNextPage;
039        }
040    
041    
042        /**
043         * Cover constructor with a dictionary of parameters.
044         *
045         * @param aNextPageName the name of next page to display
046         * @param aValuesToSetInNextPage keys and values to set in the next page
047         */
048        public NextPage(String aNextPageName, NSDictionary aValuesToSetInNextPage)
049        {
050            super();
051            /** require
052            [valid_aNextPageName_param] aNextPageName != null;
053            [valid_aValuesToSetInNextPage_param] aValuesToSetInNextPage != null; **/
054    
055            nextPageName = aNextPageName;
056            orderedDictionariesOfKeysAndValuesToSetInNextPage = new NSArray(aValuesToSetInNextPage);
057        }
058    
059    
060        /**
061         * Cover constructor with no values to set in next page.
062         *
063         * @param aNextPageName the name of next page to display.
064         */
065        public NextPage(String aNextPageName)
066        {
067            this(aNextPageName, NSDictionary.EmptyDictionary);
068            /** require [valid_param] aNextPageName != null; **/
069        }
070    
071    
072    
073        /**
074         * Returns the next page to display.  Instantiates the page with the application.
075         *
076         * @param aContext the context in which to display this page
077         * @return the next page to display
078         */
079        public com.webobjects.appserver.WOComponent nextPage(WOContext aContext)
080        {
081            /** require [valid_param] aContext != null; **/
082            com.webobjects.appserver.WOComponent nextPage = WOApplication.application().pageWithName(nextPageName, aContext);
083    
084            Enumeration dictionariesToSetInNextPageEnumerator = orderedDictionariesOfKeysAndValuesToSetInNextPage.objectEnumerator();
085            while (dictionariesToSetInNextPageEnumerator.hasMoreElements())
086            {
087                NSDictionary valuesToSetInNextPage = (NSDictionary)dictionariesToSetInNextPageEnumerator.nextElement();
088                Enumeration valuesToSetInNextPageEnumerator = valuesToSetInNextPage.keyEnumerator();
089                while (valuesToSetInNextPageEnumerator.hasMoreElements())
090                {
091                    String key = (String)valuesToSetInNextPageEnumerator.nextElement();
092                    Object value = valuesToSetInNextPage.valueForKey(key);
093                    nextPage.takeValueForKeyPath(value, key);
094                }
095            }
096    
097            return nextPage;
098        }
099    
100    
101        /** invariant
102        [has_nextPageName] nextPageName != null;
103        [has_valuesToSetInNextPage] orderedDictionariesOfKeysAndValuesToSetInNextPage != null; **/
104    
105    
106    }
