018    public class InSetFormatterTest extends TestCase
019    {
020        
021        NSArray arrayOfStrings = new NSArray(new String[] {"0", "1" , "2"});
022        NSArray arrayOfIntegers = new NSArray(new Integer[] {new Integer(0), new Integer(1) , new Integer(2)});
023    
024        
025        
026        /**
027         * Constructor.
028         * @param name of test
029         */
030        public InSetFormatterTest(String name)
031        {
032            super(name);
033        }
034    
035    
036    
037        /*
038         * Test for StringBuffer format(Object, StringBuffer, FieldPosition)
039         */
040        public void testFormatObjectStringBufferFieldPosition()
041        {
042            Format formatter = new InSetFormatter(NSArray.EmptyArray, NullFormatter.Formatter);
043            StringBuffer sb = formatter.format("12345", new StringBuffer(), new FieldPosition(0));
044            assertTrue(sb.toString().equals("12345"));
045        }
046    
047    
048    
049        /*
050         * Test for java.lang.String format(java.lang.Object)
051         */
052        public void testFormatObject()
053        {
054            Format formatter = new InSetFormatter(NSArray.EmptyArray, NullFormatter.Formatter);
055            assertTrue(formatter.format("12345").equals("12345"));
056        }
057        
058        
059        
060        /*
061         * Test for Object parseObject(String, ParsePosition)
062         */
063        public void testParseObjectStringParsePosition()
064        {
065            Format formatter = new InSetFormatter(arrayOfStrings, NullFormatter.Formatter);
066            ParsePosition position = new ParsePosition(0);
067            assertTrue(formatter.parseObject("1", position).equals("1"));
068            assertTrue(position.getIndex() == 1);
069            
070            position = new ParsePosition(0);
071            assertTrue(formatter.parseObject("12", position) == null);
072            assertTrue(position.getIndex() == 0);
073            
074            formatter = new InSetFormatter(arrayOfIntegers, IntegerFormatter.Formatter);
075            position = new ParsePosition(0);
076            assertTrue(formatter.parseObject("1", position).equals(new Integer(1)));
077            assertTrue(position.getIndex() == 1);
078            
079            position = new ParsePosition(0);
080            assertTrue(formatter.parseObject("12", position) == null);
081            assertTrue(position.getIndex() == 0);
082        }
083    
084    
085    
086        /*
087         * Test for java.lang.Object parseObject(java.lang.String)
088         */
089        public void testParseObjectString()
090        {
091            Format formatter = new InSetFormatter(arrayOfStrings, NullFormatter.Formatter);
092            try
093            {
094                assertTrue(formatter.parseObject("1").equals("1"));
095                formatter = new InSetFormatter(arrayOfIntegers, IntegerFormatter.Formatter);
096                assertTrue(formatter.parseObject("2").equals(new Integer(2)));
097             }
098            catch (ParseException e)
099            {
100                fail(e.getMessage());
101            }
102    
103            try
104            {
105                assertTrue(formatter.parseObject("9").equals(new Integer(9)));
106                fail("accepted unaccepable value");
107            }
108            catch (ParseException e) {}
109            
110        }
111    
112    }
