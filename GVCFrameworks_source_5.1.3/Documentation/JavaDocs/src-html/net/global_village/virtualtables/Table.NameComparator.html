015    /**
016     * The root object for a virtual table, which is a set of objects that can be created easily at runtime, that, together, act as if they were a single object (see VirtualRow) with a persistent store.<p>
017     *
018     * Table is the only object that you actually have to insert into the editing context.  Columns and Rows are created by using <code>newColumn()</code> and <code>newRow()</code> respectively, and these methods automatically insert the newly created column or row into the table's editing context.<p>
019     *
020     * Usage is simple: choose the correct subclass of Table (only one to choose from right now, so that is simple ;-) and instantiate it and insert it into an editing context (there is a convenience method on <code>VirtualTable</code> to help you with this). Then use <code>newColumn()</code> to create columns on the new table and <code>newRow</code> to make new rows, and finally, use <code>valueForFieldNamed()</code> and <code>takeValueForFieldNamed()</code> or KVC to populate row data.  For example:
021     * <code>
022     * testTable = VirtualTable.createVirtualTable(editingContext(), "Test Table");
023     * numberColumn = testTable.newColumn("number column", ColumnType.NumberColumnType);
024     * row = testTable.newRow();
025     * row.takeValueForKey(new Integer(1), "number column");
026     * </code>
027     *
028     * @author Copyright (c) 2001-2005  Global Village Consulting, Inc.  All rights reserved.
029     * This software is published under the terms of the Educational Community License (ECL) version 1.0,
030     * a copy of which has been included with this distribution in the LICENSE.TXT file.
031     * @version $Revision: 12$
032     */

196        {
197            /** require [valid_param] name != null; **/
198

200            return ! ((name.length() == 0) ||
201                      ( ! Character.isLetter(name.charAt(0))) ||
202                      (invalidColumnNames().containsObject(name)) ||
203                      (invalidColumnNames().containsObject(Column.normalizeStringForColumnNames(name))) );

411         *
412         * @return the list of other tables which may refer to data in this table.
413         */
414        public NSArray referringTables()
415        {
416            NSMutableSet referringTables = new NSMutableSet();
417    
418            Enumeration columnEnumerator = columns().objectEnumerator();
419            while (columnEnumerator.hasMoreElements())
420            {
421                Column aColumn = (Column) columnEnumerator.nextElement();
422    
423                Enumeration referringColumnEnumerator = aColumn.referringColumns().objectEnumerator();
424                while (referringColumnEnumerator.hasMoreElements())
425                {
426                    Column referringColumn = (Column) referringColumnEnumerator.nextElement();
427                    referringTables.addObject(referringColumn.table());
428                }
429            }
430    
431            return referringTables.allObjects();
432        }

663         * @param sortOrderings NSArray of EOSortOrder to control the order in which rows are exported, or null if no specific order is required.
664         * @param stream the stream to write the data to.  The stream is <em>not</em> closed at the end of this method - you'll have to do it yourself
665         * @param columnsToExport an ordered list of column names that we will export.  The first name in this array cooresponds to the first field in the data file, the second name to the second field, and so on
666         * @param delimiter the string to use as the delimiter
667         * @param writeHeaderLine should we write the header line, which will contain the names of the columns we are exporting
668         */
669        public void exportTable(EOQualifier qualifier, NSArray sortOrderings, OutputStream stream, NSArray columnsToExport,

671        {
672            /** require
673            [valid_stream_param] stream != null;
