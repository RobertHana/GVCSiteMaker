004    /**
005     * This class is added in this test package as a work-around for the tests to successfully run.
006     * Without this class, an error "Class 'Session' exists (class javax.mail.Session) but is not a subclass
007     * of WOSession", will be raised. The reason for this is that, when it tries to create a session for
008     * these tests, it looks for any class named Application.

012     * a copy of which has been included with this distribution in the LICENSE.TXT file.
013    
014     */
015    public class Application extends net.global_village.woextensions.WOApplication
016    {
017    
018        /**
019         * Defines required properties with no defaults.
020         */
021        protected void loadConfigurationInfo()
022        {
023            super.loadConfigurationInfo();
024            properties().addPropertyForKey("/cgi-bin/WebObjects", "WOAdaptorURL");
025            properties().addPropertyForKey("smtp.local.host", "SMTPHost");
026            properties().addPropertyForKey("FrontBase", "DBType");
027            properties().addPropertyForKey("Main", "DefaultPageName");
028    
029        }
030    
031    
032    }
