018    public class StringArrayFormatterTest extends TestCase
019    {
020        
021        NSArray arrayOfStrings = new NSArray(new String[] {"0", "1" , "2"});
022        NSArray arrayOfIntegers = new NSArray(new Integer[] {new Integer(0), new Integer(1) , new Integer(2)});
023        String stringArray = "0, 1,2";
024        String cleanStringArray = "0,1,2";
025        
026        
027        /**
028         * Constructor.
029         * @param name of test
030         */
031        public StringArrayFormatterTest(String name)
032        {
033            super(name);
034        }
035    
036    
037    
038        /*
039         * Test for StringBuffer format(Object, StringBuffer, FieldPosition)
040         */
041        public void testFormatObjectStringBufferFieldPosition()
042        {
043            Format formatter = new StringArrayFormatter(",", NullFormatter.Formatter);
044            StringBuffer sb = formatter.format("12345", new StringBuffer(), new FieldPosition(0));
045            assertTrue(sb.toString().equals("12345"));
046            
047            sb = formatter.format(arrayOfStrings, new StringBuffer(), new FieldPosition(0));
048            assertTrue(sb.toString().equals(cleanStringArray));
049            
050            formatter = new StringArrayFormatter(",", IntegerFormatter.Formatter);
051            sb = formatter.format(arrayOfIntegers, new StringBuffer(), new FieldPosition(0));
052            assertTrue(sb.toString().equals(cleanStringArray));
053        }
054    
055    
056    
057        /*
058         * Test for java.lang.String format(java.lang.Object)
059         */
060        public void testFormatObject()
061        {
062            Format formatter = new StringArrayFormatter(",", NullFormatter.Formatter);
063            assertTrue(formatter.format("12345").equals("12345"));
064            assertTrue(formatter.format(arrayOfStrings).equals(cleanStringArray));
065            assertTrue(formatter.format(arrayOfIntegers).equals(cleanStringArray));
066            assertTrue(formatter.format(NSArray.EmptyArray).equals(""));
067        }
068        
069        
070        
071        /*
072         * Test for Object parseObject(String, ParsePosition)
073         */
074        public void testParseObjectStringParsePosition()
075        {
076            Format formatter = new StringArrayFormatter(",", NullFormatter.Formatter);
077            ParsePosition position = new ParsePosition(0);
078            assertTrue(formatter.parseObject("12345", position).equals(new NSArray("12345")));
079            assertTrue(position.getIndex() == "12345".length());
080            
081            position = new ParsePosition(0);
082            assertTrue(formatter.parseObject(stringArray, position).equals(arrayOfStrings));
083            assertTrue(position.getIndex() == stringArray.length());
084            
085            formatter = new StringArrayFormatter(",", IntegerFormatter.Formatter);
086            position = new ParsePosition(0);
087            assertTrue(formatter.parseObject(stringArray, position).equals(arrayOfIntegers));
088            assertTrue(position.getIndex() == stringArray.length());
089            
090            position = new ParsePosition(0);
091            assertTrue(formatter.parseObject("", position).equals(NSArray.EmptyArray));
092            assertTrue(position.getIndex() == 0);
093            
094            position = new ParsePosition(0);
095            assertTrue(formatter.parseObject("A,B,C", position) ==  null);
096            assertTrue(position.getIndex() == 0);
097        }
098    
099    
100    
101        /*
102         * Test for java.lang.Object parseObject(java.lang.String)
103         */
104        public void testParseObjectString()
105        {
106            Format formatter = new StringArrayFormatter(",", NullFormatter.Formatter);
107            try
108            {
109                assertTrue(formatter.parseObject("12345").equals(new NSArray("12345")));
110                assertTrue(formatter.parseObject(stringArray).equals(arrayOfStrings));
111                
112                formatter = new StringArrayFormatter(",", IntegerFormatter.Formatter);
113                assertTrue(formatter.parseObject(stringArray).equals(arrayOfIntegers));
114                assertTrue(formatter.parseObject("").equals(NSArray.EmptyArray));
115            }
116            catch (ParseException e)
117            {
118                fail(e.getMessage());
119            }
120    
121            try
122            {
123                assertTrue(formatter.parseObject("A,B,C").equals(NSArray.EmptyArray));
124                fail("accepted invalid string");
125            }
126            catch (ParseException e) {}
127            
128        }
