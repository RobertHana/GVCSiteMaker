020    public class EOCustomObject extends com.webobjects.eocontrol.EOCustomObject
021    {
022    
023        /**
024         * Standard EOF constuctor.
025         */
026        public EOCustomObject()
027        {
028            super();
029        }
030    
031    
032    
033        /**
034         * Hack for bug (?) in EOF where this is NOT calling into the class description!
035         */
036        public Object validateValueForKey( Object value, String key)
037        {
038            Object tempObject = classDescription().validateValueForKey(value, key);
039    
040            // We do this just for the call to validate<em>Key</em>.  In the future, we should change this to an invocation of that method directly.
041            Object tempObject2 = super.validateValueForKey(tempObject, key);
042    
043            return tempObject2;
044        }
045

048    }
